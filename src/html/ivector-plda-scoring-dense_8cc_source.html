29 bool EstPca(const Matrix<BaseFloat> &ivector_mat, BaseFloat target_energy,
30  Matrix<BaseFloat> *mat) {
31  int32 num_rows = ivector_mat.NumRows(),
32  num_cols = ivector_mat.NumCols();
33  Vector<BaseFloat> sum;
34  SpMatrix<BaseFloat> sumsq;
35  sum.Resize(num_cols);
36  sumsq.Resize(num_cols);
37  sum.AddRowSumMat(1.0, ivector_mat);
38  sumsq.AddMat2(1.0, ivector_mat, kTrans, 1.0);
39  sum.Scale(1.0 / num_rows);
40  sumsq.Scale(1.0 / num_rows);
41  sumsq.AddVec2(-1.0, sum); // now sumsq is centered covariance.
42  int32 full_dim = sum.Dim();
43 
44  Matrix<BaseFloat> P(full_dim, full_dim);
45  Vector<BaseFloat> s(full_dim);
46 
47  try {
48  if (num_rows > num_cols)
49  sumsq.Eig(&s, &P);
50  else
51  Matrix<BaseFloat>(sumsq).Svd(&s, &P, NULL);
52  } catch (...) {
53  return false;
54  }
55 
56  SortSvd(&s, &P);
57 
58  Matrix<BaseFloat> transform(P, kTrans); // Transpose of P. This is what
59  // appears in the transform.
60 
61  // We want the PCA transform to retain target_energy amount of the total
62  // energy.
63  BaseFloat total_energy = s.Sum();
64  BaseFloat energy = 0.0;
65  int32 dim = 1;
66  while (energy / total_energy <= target_energy) {
67  energy += s(dim-1);
68  dim++;
69  }
70  Matrix<BaseFloat> transform_float(transform);
71  mat->Resize(transform.NumCols(), transform.NumRows());
72  mat->CopyFromMat(transform);
73  mat->Resize(dim, transform_float.NumCols(), kCopyData);
74  return true;
75 }

77 // Transforms i-vectors using the PLDA model.
78 void TransformIvectors(const Matrix<BaseFloat> &ivectors_in,
79  const PldaConfig &plda_config, const Plda &plda,
80  Matrix<BaseFloat> *ivectors_out) {
81  int32 dim = plda.Dim();
82  ivectors_out->Resize(ivectors_in.NumRows(), dim);
83  for (int32 i = 0; i < ivectors_in.NumRows(); i++) {
84  Vector<BaseFloat> transformed_ivector(dim);
85  plda.TransformIvector(plda_config, ivectors_in.Row(i), 1.0,
86  &transformed_ivector);
87  ivectors_out->Row(i).CopyFromVec(transformed_ivector);
88  }
89 }

91 // Transform the i-vectors using the recording-dependent PCA matrix.
92 void ApplyPca(const Matrix<BaseFloat> &ivectors_in,
93  const Matrix<BaseFloat> &pca_mat, Matrix<BaseFloat> *ivectors_out) {
94  int32 transform_cols = pca_mat.NumCols(),
95  transform_rows = pca_mat.NumRows(),
96  feat_dim = ivectors_in.NumCols();
97  ivectors_out->Resize(ivectors_in.NumRows(), transform_rows);
98  KALDI_ASSERT(transform_cols == feat_dim);
99  ivectors_out->AddMatMat(1.0, ivectors_in, kNoTrans,
100  pca_mat, kTrans, 0.0);
101 }

105 int main(int argc, char *argv[]) {
106  using namespace kaldi;
107  typedef kaldi::int32 int32;
108  try {
109  const char *usage =
110  "Perform PLDA scoring for speaker diarization. The input reco2utt\n"
111  "should be of the form <recording-id> <seg1> <seg2> ... <segN> and\n"
112  "there should be one iVector for each segment. PLDA scoring is\n"
113  "performed between all pairs of iVectors in a recording and outputs\n"
114  "an archive of score matrices, one for each recording-id. The rows\n"
115  "and columns of the the matrix correspond the sorted order of the\n"
116  "segments.\n"
117  "Usage: ivector-plda-scoring-dense [options] <plda> <reco2utt>"
118  " <ivectors-rspecifier> <scores-wspecifier>\n"
119  "e.g.: \n"
120  " ivector-plda-scoring-dense plda reco2utt scp:ivectors.scp"
121  " ark:scores.ark ark,t:ivectors.1.ark\n";
122 
123  ParseOptions po(usage);
124  BaseFloat target_energy = 0.5;
125  PldaConfig plda_config;
126  plda_config.Register(&po);
127 
128  po.Register("target-energy", &target_energy,
129  "Reduce dimensionality of i-vectors using a recording-dependent"
130  " PCA such that this fraction of the total energy remains.");
131  KALDI_ASSERT(target_energy <= 1.0);
132 
133  po.Read(argc, argv);
134 
135  if (po.NumArgs() != 4) {
136  po.PrintUsage();
137  exit(1);
138  }
139 
140  std::string plda_rxfilename = po.GetArg(1),
141  reco2utt_rspecifier = po.GetArg(2),
142  ivector_rspecifier = po.GetArg(3),
143  scores_wspecifier = po.GetArg(4);
144 
145  Plda plda;
146  ReadKaldiObject(plda_rxfilename, &plda);
147 
148  SequentialTokenVectorReader reco2utt_reader(reco2utt_rspecifier);
149  RandomAccessBaseFloatVectorReader ivector_reader(ivector_rspecifier);
150  BaseFloatMatrixWriter scores_writer(scores_wspecifier);
151  int32 num_reco_err = 0,
152  num_reco_done = 0;
153  for (; !reco2utt_reader.Done(); reco2utt_reader.Next()) {
154  Plda this_plda(plda);
155  std::string reco = reco2utt_reader.Key();
156 
157  std::vector<std::string> uttlist = reco2utt_reader.Value();
158  std::vector<Vector<BaseFloat> > ivectors;
159 
160  for (size_t i = 0; i < uttlist.size(); i++) {
161  std::string utt = uttlist[i];
162 
163  if (!ivector_reader.HasKey(utt)) {
164  KALDI_ERR << "No iVector present in input for utterance " << utt;
165  }
166 
167  Vector<BaseFloat> ivector = ivector_reader.Value(utt);
168  ivectors.push_back(ivector);
169  }
170  if (ivectors.size() == 0) {
171  KALDI_WARN << "Not producing output for recording " << reco
172  << " since no segments had iVectors";
173  num_reco_err++;
174  } else {
175  Matrix<BaseFloat> ivector_mat(ivectors.size(), ivectors[0].Dim()),
176  ivector_mat_pca,
177  ivector_mat_plda,
178  pca_transform,
179  scores(ivectors.size(), ivectors.size());
180 
181  for (size_t i = 0; i < ivectors.size(); i++) {
182  ivector_mat.Row(i).CopyFromVec(ivectors[i]);
183  }
184  if (EstPca(ivector_mat, target_energy, &pca_transform)) {
185  // Apply the PCA transform to the raw i-vectors.
186  ApplyPca(ivector_mat, pca_transform, &ivector_mat_pca);
187 
188  // Apply the PCA transform to the parameters of the PLDA model.
189  this_plda.ApplyTransform(Matrix<double>(pca_transform));
190 
191  // Now transform the i-vectors using the reduced PLDA model.
192  TransformIvectors(ivector_mat_pca, plda_config, this_plda,
193  &ivector_mat_plda);
194  } else {
195  KALDI_WARN << "Unable to compute conversation dependent PCA for"
196  << " recording " << reco << ".";
197  TransformIvectors(ivector_mat, plda_config, this_plda,
198  &ivector_mat_plda);
199  }
200  for (int32 i = 0; i < ivector_mat_plda.NumRows(); i++) {
201  for (int32 j = 0; j < ivector_mat_plda.NumRows(); j++) {
202  scores(i, j) = this_plda.LogLikelihoodRatio(Vector<double>(
203  ivector_mat_plda.Row(i)), 1.0,
204  Vector<double>(ivector_mat_plda.Row(j)));
205  }
206  }
207  scores_writer.Write(reco, scores);
208  num_reco_done++;
209  }
210  }
211  KALDI_LOG << "Processed " << num_reco_done << " recordings, "
212  << num_reco_err << " had errors.";
213  return (num_reco_done != 0 ? 0 : 1 );
214  } catch(const std::exception &e) {
215  std::cerr << e.what();
216  return -1;
217  }
218 }
