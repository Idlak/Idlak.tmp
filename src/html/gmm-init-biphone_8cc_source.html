29 // This function reads a file like:
30 // 1 2 3
31 // 4 5
32 // 6 7 8
33 // where each line is a list of integer id's of phones (that should have their pdfs shared).
34 void ReadSharedPhonesList(std::string rxfilename, std::vector<std::vector<int32> > *list_out) {
35  list_out->clear();
36  Input input(rxfilename);
37  std::istream &is = input.Stream();
38  std::string line;
39  while (std::getline(is, line)) {
40  list_out->push_back(std::vector<int32>());
41  if (!SplitStringToIntegers(line, " \t\r", true, &(list_out->back())))
42  KALDI_ERR << "Bad line in shared phones list: " << line << " (reading "
43  << PrintableRxfilename(rxfilename) << ")";
44  std::sort(list_out->rbegin()->begin(), list_out->rbegin()->end());
45  if (!IsSortedAndUniq(*(list_out->rbegin())))
46  KALDI_ERR << "Bad line in shared phones list (repeated phone): " << line
47  << " (reading " << PrintableRxfilename(rxfilename) << ")";
48  }
49 }

52 *GetFullBiphoneStubMap(const std::vector<std::vector<int32> > &phone_sets,
53  const std::vector<int32> &phone2num_pdf_classes,
54  const std::vector<bool> &share_roots) {
55 
56  { // Check the inputs
57  KALDI_ASSERT(!phone_sets.empty() && share_roots.size() == phone_sets.size());
58  std::set<int32> all_phones;
59  for (size_t i = 0; i < phone_sets.size(); i++) {
60  KALDI_ASSERT(IsSortedAndUniq(phone_sets[i]));
61  KALDI_ASSERT(!phone_sets[i].empty());
62  for (size_t j = 0; j < phone_sets[i].size(); j++) {
63  KALDI_ASSERT(all_phones.count(phone_sets[i][j]) == 0); // Check not present.
64  all_phones.insert(phone_sets[i][j]);
65  }
66  }
67  }
68 
69  int32 numpdfs_per_phone = phone2num_pdf_classes[1];
70  int32 current_pdfid = 0;
71  std::map<EventValueType, EventMap*> level1_map; // key is 1
72  for (size_t i = 0; i < phone_sets.size(); i++) {
73 
74  if (numpdfs_per_phone == 1) {
75  // Create an event map for level2:
76  std::map<EventValueType, EventAnswerType> level2_map; // key is 0
77  level2_map[0] = current_pdfid++; // no-left-context case
78  for (size_t j = 0; j < phone_sets.size(); j++) {
79  int32 pdfid = current_pdfid++;
80  std::vector<int32> pset = phone_sets[j]; // All these will have a
81  // shared pdf with id=pdfid
82  for (size_t k = 0; k < pset.size(); k++)
83  level2_map[pset[k]] = pdfid;
84  }
85  std::vector<int32> pset = phone_sets[i]; // All these will have a
86  // shared event-map child
87  for (size_t k = 0; k < pset.size(); k++)
88  level1_map[pset[k]] = new TableEventMap(0, level2_map);
89  } else {
90  KALDI_ASSERT(numpdfs_per_phone == 2);
91  int32 base_pdfid = current_pdfid;
92  std::vector<int32> pset = phone_sets[i]; // All these will have a shared
93  // event-map child
94  for (size_t k = 0; k < pset.size(); k++) {
95  // Create an event map for level2:
96  std::map<EventValueType, EventMap*> level2_map; // key is 0
97  {
98  std::map<EventValueType, EventAnswerType> level3_map; // key is kPdfClass
99  level3_map[0] = current_pdfid++;
100  level3_map[1] = current_pdfid++;
101  level2_map[0] = new TableEventMap(kPdfClass, level3_map); // no-left-context case
102  }
103  for (size_t j = 0; j < phone_sets.size(); j++) {
104  std::map<EventValueType, EventAnswerType> level3_map; // key is -1
105  level3_map[0] = current_pdfid++;
106  level3_map[1] = current_pdfid++;
107 
108  std::vector<int32> ipset = phone_sets[j]; // All these will have a
109  // shared subtree with 2 pdfids
110  for (size_t ik = 0; ik < ipset.size(); ik++) {
111  level2_map[ipset[ik]] = new TableEventMap(kPdfClass, level3_map);
112  }
113  }
114  level1_map[pset[k]] = new TableEventMap(0, level2_map);
115  if (k != pset.size() - 1)
116  current_pdfid = base_pdfid;
117  }
118  }
119 
120  }
121  return new TableEventMap(1, level1_map);
122 }

124 ContextDependency*
125 BiphoneContextDependencyFull(const std::vector<std::vector<int32> > phone_sets,
126  const std::vector<int32> phone2num_pdf_classes) {
127  std::vector<bool> share_roots(phone_sets.size(), false); // Don't share roots
128  // N is context size, P = position of central phone (must be 0).
129  int32 P = 1, N = 2;
130  EventMap *pdf_map = GetFullBiphoneStubMap(phone_sets,
131  phone2num_pdf_classes, share_roots);
132  return new ContextDependency(N, P, pdf_map);
133 }

137 int main(int argc, char *argv[]) {
138  try {
139  using namespace kaldi;
140  using kaldi::int32;
141 
142  const char *usage =
143  "Initialize a biphone context-dependency tree with all the\n"
144  "leaves (i.e. a full tree). Intended for end-to-end tree-free models.\n"
145  "Usage: gmm-init-biphone <topology-in> <dim> <model-out> <tree-out> \n"
146  "e.g.: \n"
147  " gmm-init-biphone topo 39 bi.mdl bi.tree\n";
148 
149  bool binary = true;
150  std::string shared_phones_rxfilename;
151  ParseOptions po(usage);
152  po.Register("binary", &binary, "Write output in binary mode");
153  po.Register("shared-phones", &shared_phones_rxfilename,
154  "rxfilename containing, on each line, a list of phones "
155  "whose pdfs should be shared.");
156  po.Read(argc, argv);
157 
158  if (po.NumArgs() != 4) {
159  po.PrintUsage();
160  exit(1);
161  }
162 
163 
164  std::string topo_filename = po.GetArg(1);
165  int dim = 0;
166  if (!ConvertStringToInteger(po.GetArg(2), &dim) || dim <= 0 || dim > 10000)
167  KALDI_ERR << "Bad dimension:" << po.GetArg(2)
168  << ". It should be a positive integer.";
169  std::string model_filename = po.GetArg(3);
170  std::string tree_filename = po.GetArg(4);
171 
172  Vector<BaseFloat> glob_inv_var(dim);
173  glob_inv_var.Set(1.0);
174  Vector<BaseFloat> glob_mean(dim);
175  glob_mean.Set(1.0);
176 
177  HmmTopology topo;
178  bool binary_in;
179  Input ki(topo_filename, &binary_in);
180  topo.Read(ki.Stream(), binary_in);
181 
182  const std::vector<int32> &phones = topo.GetPhones();
183 
184  std::vector<int32> phone2num_pdf_classes(1 + phones.back());
185  for (size_t i = 0; i < phones.size(); i++) {
186  phone2num_pdf_classes[phones[i]] = topo.NumPdfClasses(phones[i]);
187  // For now we only support 1 or 2 pdf's per phone
188  KALDI_ASSERT(phone2num_pdf_classes[phones[i]] == 1 ||
189  phone2num_pdf_classes[phones[i]] == 2);
190  }
191 
192  // Now the tree:
193  ContextDependency *ctx_dep = NULL;
194  std::vector<std::vector<int32> > shared_phones;
195  if (shared_phones_rxfilename == "") {
196  shared_phones.resize(phones.size());
197  for (size_t i = 0; i < phones.size(); i++)
198  shared_phones[i].push_back(phones[i]);
199  } else {
200  ReadSharedPhonesList(shared_phones_rxfilename, &shared_phones);
201  // ReadSharedPhonesList crashes on error.
202  }
203  ctx_dep = BiphoneContextDependencyFull(shared_phones, phone2num_pdf_classes);
204 
205  int32 num_pdfs = ctx_dep->NumPdfs();
206 
207  AmDiagGmm am_gmm;
208  DiagGmm gmm;
209  gmm.Resize(1, dim);
210  { // Initialize the gmm.
211  Matrix<BaseFloat> inv_var(1, dim);
212  inv_var.Row(0).CopyFromVec(glob_inv_var);
213  Matrix<BaseFloat> mu(1, dim);
214  mu.Row(0).CopyFromVec(glob_mean);
215  Vector<BaseFloat> weights(1);
216  weights.Set(1.0);
217  gmm.SetInvVarsAndMeans(inv_var, mu);
218  gmm.SetWeights(weights);
219  gmm.ComputeGconsts();
220  }
221 
222  for (int i = 0; i < num_pdfs; i++)
223  am_gmm.AddPdf(gmm);
224 
225  // Now the transition model:
226  TransitionModel trans_model(*ctx_dep, topo);
227 
228  {
229  Output ko(model_filename, binary);
230  trans_model.Write(ko.Stream(), binary);
231  am_gmm.Write(ko.Stream(), binary);
232  }
233 
234  // Now write the tree.
235  ctx_dep->Write(Output(tree_filename, binary).Stream(),
236  binary);
237 
238  delete ctx_dep;
239  return 0;
240  } catch(const std::exception &e) {
241  std::cerr << e.what();
242  return -1;
243  }
244 }
