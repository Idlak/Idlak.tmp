34 const char* TxpPos::GetPos(const char* ptag, const char* word) {
35  PosRgxVector::iterator it;
36  LookupMap::iterator lkp;
37  const char* current_tag;
38  current_tag = most_common_.c_str();
39  // Regex tagger
40  for (it = rgxtagger_.begin(); it != rgxtagger_.end(); ++it) {
41  if ((*it).pos == TXPPOSRGX_POS_WORD &&
42  !strcmp((*it).pattern.c_str(), word)) {
43  current_tag = (*it).tag.c_str();
44  break;
45  } else if ((*it).pos == TXPPOSRGX_POS_PREFIX &&
46  !strncmp((*it).pattern.c_str(), word, (*it).pattern.length())) {
47  current_tag = (*it).tag.c_str();
48  break;
49  } else if ((*it).pos == TXPPOSRGX_POS_SUFFIX &&
50  (*it).pattern.length() < strlen(word) &&
51  !strcmp((*it).pattern.c_str(),
52  word + (strlen(word) - (*it).pattern.length()))) {
53  current_tag = (*it).tag.c_str();
54  break;
55  }
56  }
57  // Pattern tagger: unigram
58  lkp = patterntagger_.find(std::string(word));
59  if (lkp != patterntagger_.end()) {
60  current_tag = (lkp->second).c_str();
61  }
62  // bigram
63  if (*ptag) {
64  lkp = patterntagger_.find(std::string(ptag) + "_" + std::string(word));
65  if (lkp != patterntagger_.end()) {
66  current_tag = lkp->second.c_str();
67  }
68  }
69  return current_tag;
70 }

72 void TxpPos::StartElement(const char* name, const char** atts) {
73  std::string pos;
74  std::string tag;
75  std::string ptag;
76  TxpPosRgx rgx;
77  if (!strcmp(name, "most_common")) {
78  SetAttribute("tag", atts, &most_common_);
79  }
80  if (!strcmp(name, "r")) {
81  SetAttribute("pos", atts, &pos);
82  if (pos == "WORD") rgx.pos = TXPPOSRGX_POS_WORD;
83  else if (pos == "PREFIX") rgx.pos = TXPPOSRGX_POS_PREFIX;
84  else if (pos == "SUFFIX") rgx.pos = TXPPOSRGX_POS_SUFFIX;
85  SetAttribute("pat", atts, &(rgx.pattern));
86  SetAttribute("tag", atts, &(rgx.tag));
87  rgxtagger_.push_back(rgx);
88  } else if (!strcmp(name, "w")) {
89  SetAttribute("name", atts, &word_);
90  } else if (!strcmp(name, "p")) {
91  SetAttribute("tag", atts, &tag);
92  SetAttribute("ptag", atts, &ptag);
93  if (ptag.empty())
94  patterntagger_.insert(LookupItem(word_, tag));
95  else
96  patterntagger_.insert(LookupItem(word_, tag + "_" + word_));
97  }
98 }
