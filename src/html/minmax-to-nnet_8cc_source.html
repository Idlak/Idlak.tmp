60  Input ki(transform_rxfilename, &binary_read);
61  minmax_stats.Read(ki.Stream(), binary_read);

64  KALDI_ASSERT(minmax_stats.NumCols() > 1);

67  double count = minmax_stats(0, minmax_stats.NumCols()-1);

69  // buffers for shift and scale

72 
73  // Make this configurable?

75  double da = 0.99;
76 
77  // compute the shift and scale per each dimension
78  for(int32 d = 0; d < num_dims; d++) {
79  BaseFloat min = minmax_stats(0, d);
80  BaseFloat max = minmax_stats(1, d);
81  if (min + 1e-7 >= max) max = min + 1e-7;
82  shift(d) = (max * ca - min * da) / (max - min);
83  scale(d) = (da - ca) / (max - min);
84  std::cerr << d << ": " << shift(d) << "," << scale(d) << '\n';
85  }

103  }
