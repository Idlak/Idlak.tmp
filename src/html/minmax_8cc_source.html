36 void AccMinmaxStats(const VectorBase<BaseFloat> &feats, BaseFloat weight, MatrixBase<double> *stats) {
37  int32 dim = feats.Dim();
38  KALDI_ASSERT(stats != NULL);
39  KALDI_ASSERT(stats->NumRows() == 2 && stats->NumCols() == dim + 1);
40  // Remove these __restrict__ modifiers if they cause compilation problems.
41  // It's just an optimization.
42  double *__restrict__ min_ptr = stats->RowData(0),
43  *__restrict__ max_ptr = stats->RowData(1),
44  *__restrict__ count_ptr = min_ptr + dim;
45  const BaseFloat * __restrict__ feats_ptr = feats.Data();
46  *count_ptr += weight;
47  // Careful-- if we change the format of the matrix, the "mean_ptr < count_ptr"
48  // statement below might become wrong.
49  for (; min_ptr < count_ptr; min_ptr++, max_ptr++, feats_ptr++) {
50  double a = *feats_ptr * weight;
51  if (a < *min_ptr) *min_ptr = a;
52  if (a > *max_ptr) *max_ptr = a;
53  }
54 }

71 void ApplyMinmax(const MatrixBase<double> &stats,
72  bool var_norm,
73  MatrixBase<BaseFloat> *feats) {
74  KALDI_ASSERT(feats != NULL);
75  int32 dim = stats.NumCols() - 1;
76  if (stats.NumRows() > 2 || stats.NumRows() < 1 || feats->NumCols() != dim) {
77  KALDI_ERR << "Dim mismatch in ApplyCmvn: cmvn "
78  << stats.NumRows() << 'x' << stats.NumCols()
79  << ", feats " << feats->NumRows() << 'x' << feats->NumCols();
80  }
81  if (stats.NumRows() == 1)
82  KALDI_ERR << "You requested variance normalization but no variance stats "
83  << "are supplied.";
84 
85  double count = stats(0, dim);
86  // Do not change the threshold of 1.0 here: in the balanced-cmvn code, when
87  // computing an offset and representing it as stats, we use a count of one.
88  if (count < 1.0)
89  KALDI_ERR << "Insufficient stats for cepstral mean and variance normalization: "
90  << "count = " << count;
91 
92  Matrix<BaseFloat> norm(2, dim); // norm(0, d) = mean offset
93  // norm(1, d) = scale, e.g. x(d) <-- x(d)*norm(1, d) + norm(0, d).
94  // TODO: make these values configurable as parameters
95  double ca = 0.01;
96  double da = 0.99;
97  for (int32 d = 0; d < dim; d++) {
98  double min, max, offset, scale;
99  min = stats(0, d);
100  max = stats(1, d);
101  //if (!var_norm) {
102  //scale = 1.0;
103  offset = (max * ca - min * da) / (max - min);
104  scale = (da - ca) / (max - min);
105  norm(0, d) = offset;
106  norm(1, d) = scale;
107  }
108  int32 num_frames = feats->NumRows();
109 
110  // Apply the normalization.
111  for (int32 i = 0; i < num_frames; i++) {
112  for (int32 d = 0; d < dim; d++) {
113  BaseFloat &f = (*feats)(i, d);
114  f = norm(0, d) + f*norm(1, d);
115  }
116  }
117 }
