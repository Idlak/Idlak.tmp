48 class OnlineGenericBaseFeature: public OnlineBaseFeature {
49  public:
50  //
51  // First, functions that are present in the interface:
52  //
53  virtual int32 Dim() const { return computer_.Dim(); }
54 
55  // Note: IsLastFrame() will only ever return true if you have called
56  // InputFinished() (and this frame is the last frame).
57  virtual bool IsLastFrame(int32 frame) const {
58  return input_finished_ && frame == NumFramesReady() - 1;
59  }
60  virtual BaseFloat FrameShiftInSeconds() const {
61  return computer_.GetFrameOptions().frame_shift_ms / 1000.0f;
62  }
63 
64  virtual int32 NumFramesReady() const { return features_.size(); }
65 
66  virtual void GetFrame(int32 frame, VectorBase<BaseFloat> *feat);
67 
68  // Next, functions that are not in the interface.
69 
70 
71  // Constructor from options class
72  explicit OnlineGenericBaseFeature(const typename C::Options &opts);
73 
74  // This would be called from the application, when you get
75  // more wave data. Note: the sampling_rate is only provided so
76  // the code can assert that it matches the sampling rate
77  // expected in the options.
78  virtual void AcceptWaveform(BaseFloat sampling_rate,
79  const VectorBase<BaseFloat> &waveform);
80 
81 
82  // InputFinished() tells the class you won't be providing any
83  // more waveform. This will help flush out the last frame or two
84  // of features, in the case where snip-edges == false; it also
85  // affects the return value of IsLastFrame().
86  virtual void InputFinished() {
87  input_finished_ = true;
88  ComputeFeatures();
89  }
90 
91  ~OnlineGenericBaseFeature() {
92  DeletePointers(&features_);
93  }
94 
95  private:
96  // This function computes any additional feature frames that it is possible to
97  // compute from 'waveform_remainder_', which at this point may contain more
98  // than just a remainder-sized quantity (because AcceptWaveform() appends to
99  // waveform_remainder_ before calling this function). It adds these feature
100  // frames to features_, and shifts off any now-unneeded samples of input from
101  // waveform_remainder_ while incrementing waveform_offset_ by the same amount.
102  void ComputeFeatures();
103 
104  C computer_; // class that does the MFCC or PLP or filterbank computation
105 
106  FeatureWindowFunction window_function_;
107 
108  // features_ is the Mfcc or Plp or Fbank features that we have already computed.
109 
110  std::vector<Vector<BaseFloat>*> features_;
111 
112  // True if the user has called "InputFinished()"
113  bool input_finished_;
114 
115  // The sampling frequency, extracted from the config. Should
116  // be identical to the waveform supplied.
117  BaseFloat sampling_frequency_;
118 
119  // waveform_offset_ is the number of samples of waveform that we have
120  // already discarded, i.e. thatn were prior to 'waveform_remainder_'.
121  int64 waveform_offset_;
122 
123  // waveform_remainder_ is a short piece of waveform that we may need to keep
124  // after extracting all the whole frames we can (whatever length of feature
125  // will be required for the next phase of computation).
126  Vector<BaseFloat> waveform_remainder_;
127 };

166 // Note the similarity with SlidingWindowCmnOptions, but there
167 // are also differences. One which doesn't appear in the config
168 // itself, because it's a difference between the setups, is that
169 // in OnlineCmn, we carry over data from the previous utterance,
170 // or, if no previous utterance is available, from global stats,
171 // or, if previous utterances are available but the total amount
172 // of data is less than prev_frames, we pad with up to "global_frames"
173 // frames from the global stats.
174 struct OnlineCmvnOptions {
175  int32 cmn_window;
176  int32 speaker_frames; // must be <= cmn_window
177  int32 global_frames; // must be <= speaker_frames.
178  bool normalize_mean; // Must be true if normalize_variance==true.
179  bool normalize_variance;
180 
181  int32 modulus; // not configurable from command line, relates to how the
182  // class computes the cmvn internally. smaller->more
183  // time-efficient but less memory-efficient. Must be >= 1.
184  int32 ring_buffer_size; // not configurable from command line; size of ring
185  // buffer used for caching CMVN stats.
186  std::string skip_dims; // Colon-separated list of dimensions to skip normalization
187  // of, e.g. 13:14:15.
188 
189  OnlineCmvnOptions():
190  cmn_window(600),
191  speaker_frames(600),
192  global_frames(200),
193  normalize_mean(true),
194  normalize_variance(false),
195  modulus(20),
196  ring_buffer_size(20),
197  skip_dims("") { }
198 
199  void Check() {
200  KALDI_ASSERT(speaker_frames <= cmn_window && global_frames <= speaker_frames
201  && modulus > 0);
202  }
203 
204  void Register(ParseOptions *po) {
205  po->Register("cmn-window", &cmn_window, "Number of frames of sliding "
206  "context for cepstral mean normalization.");
207  po->Register("global-frames", &global_frames, "Number of frames of "
208  "global-average cepstral mean normalization stats to use for "
209  "first utterance of a speaker");
210  po->Register("speaker-frames", &speaker_frames, "Number of frames of "
211  "previous utterance(s) from this speaker to use in cepstral "
212  "mean normalization");
213  // we name the config string "norm-vars" for compatibility with
214  // ../featbin/apply-cmvn.cc
215  po->Register("norm-vars", &normalize_variance, "If true, do "
216  "cepstral variance normalization in addition to cepstral mean "
217  "normalization ");
218  po->Register("norm-means", &normalize_mean, "If true, do mean normalization "
219  "(note: you cannot normalize the variance but not the mean)");
220  po->Register("skip-dims", &skip_dims, "Dimensions to skip normalization of "
221  "(colon-separated list of integers)");}
222 };

292  public:
293 
294  //
295  // First, functions that are present in the interface:
296  //
297  virtual int32 Dim() const { return src_->Dim(); }
298 
299  virtual bool IsLastFrame(int32 frame) const {
300  return src_->IsLastFrame(frame);
301  }
302  virtual BaseFloat FrameShiftInSeconds() const {
303  return src_->FrameShiftInSeconds();
304  }
305 
306  // The online cmvn does not introduce any additional latency.
307  virtual int32 NumFramesReady() const { return src_->NumFramesReady(); }
308 
309  virtual void GetFrame(int32 frame, VectorBase<BaseFloat> *feat);
310 
311  //
312  // Next, functions that are not in the interface.
313  //
314 
323  OnlineCmvn(const OnlineCmvnOptions &opts,
324  const OnlineCmvnState &cmvn_state,
325  OnlineFeatureInterface *src);
326 
329  OnlineCmvn(const OnlineCmvnOptions &opts,
330  OnlineFeatureInterface *src);
331 
332  // Outputs any state information from this utterance to "cmvn_state".
333  // The value of "cmvn_state" before the call does not matter: the output
334  // depends on the value of OnlineCmvnState the class was initialized
335  // with, the input feature values up to cur_frame, and the effects
336  // of the user possibly having called Freeze().
337  // If cur_frame is -1, it will just output the unmodified original
338  // state that was supplied to this object.
339  void GetState(int32 cur_frame,
340  OnlineCmvnState *cmvn_state);
341 
342  // This function can be used to modify the state of the CMVN computation
343  // from outside, but must only be called before you have processed any data
344  // (otherwise it will crash). This "state" is really just the information
345  // that is propagated between utterances, not the state of the computation
346  // inside an utterance.
347  void SetState(const OnlineCmvnState &cmvn_state);
348 
349  // From this point it will freeze the CMN to what it would have been if
350  // measured at frame "cur_frame", and it will stop it from changing
351  // further. This also applies retroactively for this utterance, so if you
352  // call GetFrame() on previous frames, it will use the CMVN stats
353  // from cur_frame; and it applies in the future too if you then
354  // call OutputState() and use this state to initialize the next
355  // utterance's CMVN object.
356  void Freeze(int32 cur_frame);
357 
358  virtual ~OnlineCmvn();
359  private:
360 
365  static void SmoothOnlineCmvnStats(const MatrixBase<double> &speaker_stats,
366  const MatrixBase<double> &global_stats,
367  const OnlineCmvnOptions &opts,
368  MatrixBase<double> *stats);
369 
372  void GetMostRecentCachedFrame(int32 frame,
373  int32 *cached_frame,
374  Matrix<double> *stats);
375 
377  void CacheFrame(int32 frame, const Matrix<double> &stats);
378 
380  inline void InitRingBufferIfNeeded();
381 
385  void ComputeStatsForFrame(int32 frame,
386  MatrixBase<double> *stats);
387 
388 
389  OnlineCmvnOptions opts_;
390  std::vector<int32> skip_dims_; // Skip CMVN for these dimensions. Derived from opts_.
391  OnlineCmvnState orig_state_; // reflects the state before we saw this
392  // utterance.
393  Matrix<double> frozen_state_; // If the user called Freeze(), this variable
394  // will reflect the CMVN state that we froze
395  // at.
396 
397  // The variable below reflects the raw (count, x, x^2) statistics of the
398  // input, computed every opts_.modulus frames. raw_stats_[n / opts_.modulus]
399  // contains the (count, x, x^2) statistics for the frames from
400  // std::max(0, n - opts_.cmn_window) through n.
401  std::vector<Matrix<double>*> cached_stats_modulo_;
402  // the variable below is a ring-buffer of cached stats. the int32 is the
403  // frame index.
404  std::vector<std::pair<int32, Matrix<double> > > cached_stats_ring_;
405 
406  OnlineFeatureInterface *src_; // Not owned here
407 };
