33 bool TxpTokenise::Process(pugi::xml_document* input) {
34  const char* p;
35  int32 n = 0;
36  int32 offset, col = 0;
37  std::string token, wspace, tmp;
38  pugi::xml_node tkroot, tk, tkcopy, ws, ntxt, lex;
39  TxpAbbrevInfo * abbrev_info;
40  // all text nodes are tokenised
41  pugi::xpath_node_set text =
42  input->document_element().select_nodes("//text()");
43  // iterate through text nodes
44  text.sort();
45  for (pugi::xpath_node_set::const_iterator it = text.begin();
46  it != text.end();
47  ++it) {
48  pugi::xml_node node = (*it).node();
49  p = node.value();
50  while (*p) {
51  // break off tokens and spacing and add as an element
52  p = trules_.ConsumeWhitespaceToken(p, &token, &wspace);
53  if (token.length()) {
54  col += token.length();
55  tkroot = tk = node.parent().insert_child_before("tk", node);
56  ntxt = tk.append_child(pugi::node_pcdata);
57  ntxt.set_value(token.c_str());
58  n += 1;
59  trules_.ReplaceUtf8Punc(token, &tmp);
62  abbrev_info = abbrev_.LookupAbbrev(token.c_str());
63  if (abbrev_info) {
64  for(int32 i = 0; i < abbrev_info->expansions.size(); i++) {
65  if (!i) {
66  tk.append_attribute("tknorm");
67  tk.attribute("tknorm").set_value(abbrev_info->expansions[0].c_str());
68  if (!abbrev_info->lexentries[0].empty()) {
69  lex = tk.parent().insert_child_after("lex", tk);
70  lex.append_attribute("type").set_value(abbrev_info->lexentries[0].c_str());
71  tkcopy = lex.append_copy(tk);
72  tk.parent().remove_child(tk);
73  tk = tkcopy;
74  }
75  }
76  else {
77  if (!abbrev_info->lexentries[i].empty()) {
78  lex = tk.parent().insert_child_after("lex", tk);
79  lex.append_attribute("type").set_value(abbrev_info->lexentries[i].c_str());
80  tk = lex.append_child("tk");
81  tk.append_attribute("tknorm").set_value(abbrev_info->expansions[i].c_str());
82  tk = lex; //so we add new tokens after the lex tag not inside it
83  }
84  else {
85  tk = tk.parent().insert_child_after("tk", tk);
86  tk.append_attribute("tknorm").set_value(abbrev_info->expansions[i].c_str());
87  }
88  }
89  }
90  }
91  // if no full match of an abbreviation unpack punctuation
92  else {
93  SetPuncCaseInfo(&tmp, &tk);
94  }
95  }
96  if (wspace.length()) {
97  if (token.length())
98  ws = tk.parent().insert_child_after("ws", tk);
99  else
100  ws = node.parent().insert_child_before("ws", node);
101  ntxt = ws.append_child(pugi::node_pcdata);
102  ntxt.set_value(wspace.c_str());
103  // Set column position for carriage returns and newlines
104  // Used to decide on soft hyphenation and phrasing
105  offset = wspace.find("\n");
106  if (offset == std::string::npos) offset = wspace.find("\r");
107  if (offset != std::string::npos) {
108  std::stringstream sstream;
109  sstream << col + offset;
110  ws.append_attribute("col").set_value(sstream.str().c_str());
111  col = wspace.length() - offset;
112  } else {
113  col += wspace.length();
114  }
115  }
116  }
117  // Delete original text node
118  node.parent().remove_child(node);
119  }
120  return true;
121 }

123 // Analyses the characters and sets flags giving case, foriegn character info
124 int32 TxpTokenise::SetPuncCaseInfo(std::string* tkin, pugi::xml_node* tk) {
125  const char* p;
126  TxpTrulesCaseInfo caseinfo;
127  pugi::xml_node node, lex, tkcopy;
128  int32 n = 0;
129  std::string token;
130  std::string prepunc;
131  std::string pstpunc;
132  TxpAbbrevInfo * abbrev_info;
133  p = tkin->c_str();
134  while (*p) {
135  p = trules_.ConsumePunc(p, &prepunc);
136  p = trules_.ConsumeToken(p, &token);
137  p = trules_.ConsumePunc(p, &pstpunc);
138  if (n) {
139  *tk = tk->parent().insert_child_after("tk", *tk);
140  }
141  // check to see if there is an abbreviation that matches the token and
142  // completely or partially matches the punctuation
143  abbrev_info = abbrev_.LookupAbbrev(token.c_str(), prepunc.c_str(), pstpunc.c_str());
144  if (abbrev_info) {
145  // trim punctuation as appropriate
146  prepunc = prepunc.substr(0, prepunc.size() - abbrev_.CheckPrePunc(prepunc.c_str(), abbrev_info));
147  pstpunc = pstpunc.substr(abbrev_.CheckPstPunc(pstpunc.c_str(), abbrev_info),
148  pstpunc.size() - abbrev_.CheckPstPunc(pstpunc.c_str(),
149  abbrev_info));
150  for(int32 i = 0; i < abbrev_info->expansions.size(); i++) {
151  if (!i) {
152  tk->append_attribute("tknorm");
153  tk->attribute("tknorm").set_value(abbrev_info->expansions[0].c_str());
154  if (!abbrev_info->lexentries[0].empty()) {
155  lex = tk->parent().insert_child_after("lex", *tk);
156  lex.append_attribute("type").set_value(abbrev_info->lexentries[0].c_str());
157  tkcopy = lex.append_copy(*tk);
158  tk->parent().remove_child(*tk);
159  *tk = tkcopy;
160  }
161  if (prepunc.length()) {
162  tk->append_attribute("prepunc");
163  tk->attribute("prepunc").set_value(prepunc.c_str());
164  }
165  }
166  else {
167  if (!abbrev_info->lexentries[i].empty()) {
168  lex = tk->parent().insert_child_after("lex", *tk);
169  lex.append_attribute("type").set_value(abbrev_info->lexentries[i].c_str());
170  *tk = lex.append_child("tk");
171  tk->append_attribute("tknorm").set_value(abbrev_info->expansions[i].c_str());
172  *tk = lex;
173  }
174  else {
175  *tk = tk->parent().insert_child_after("tk", *tk);
176  tk->append_attribute("tknorm").set_value(abbrev_info->expansions[i].c_str());
177  }
178  }
179  }
180  if (pstpunc.length()) {
181  tk->append_attribute("pstpunc");
182  tk->attribute("pstpunc").set_value(pstpunc.c_str());
183  }
184  }
185  else {
186  if (prepunc.length()) {
187  tk->append_attribute("prepunc");
188  tk->attribute("prepunc").set_value(prepunc.c_str());
189  }
190  if (token.length()) {
191  tk->append_attribute("tknorm");
192  trules_.NormCaseCharacter(&token, caseinfo);
193  tk->attribute("tknorm").set_value(token.c_str());
194  }
195  if (pstpunc.length()) {
196  tk->append_attribute("pstpunc");
197  tk->attribute("pstpunc").set_value(pstpunc.c_str());
198  }
199  if (caseinfo.lowercase) tk->append_attribute("lc").set_value("true");
200  if (caseinfo.uppercase) tk->append_attribute("uc").set_value("true");
201  if (caseinfo.foreign) tk->append_attribute("foreign").set_value("true");
202  if (caseinfo.symbols) tk->append_attribute("symbols").set_value("true");
203  if (caseinfo.capitalised) tk->append_attribute("caps").set_value("true");
204  }
205  n++;
206  }
207  return true;
208 }
