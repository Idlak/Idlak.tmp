33 bool TxpPbreak::GetPbreak(const char* punc,
34  enum TXPPBREAK_POS pos,
35  TxpPbreakInfo &info) {
36  PbreakMap::iterator lkp;
37  const char* p;
38  TxpUtf8 utf8;
39  int32 clen;
40  bool found = false;
41  p = punc;
42  while (*p) {
43  clen = utf8.Clen(p);
44  if (pos == TXPPBREAK_POS_PRE) {
45  lkp = pre_pbreak_.find(std::string(p, clen));
46  if (lkp != pre_pbreak_.end()) found = true;
47  } else {
48  lkp = pst_pbreak_.find(std::string(p, clen));
49  if (lkp != pst_pbreak_.end()) found = true;
50  }
51  if (found && lkp->second.time > info.time) info.time = lkp->second.time;
52  if (found && lkp->second.type > info.type) info.type = lkp->second.type;
53  p += clen;
54  }
55  return found;
56 }

58 void TxpPbreak::GetWhitespaceBreaks(const char* ws, int32 col,
59  bool hzone,
60  int32 hzone_start, int32 hzone_end,
61  bool* newline, bool* newline2) {
62  const char* p = ws;
63  TxpUtf8 utf8;
64  int32 clen;
65  int32 nls = 0;
66  bool wspstnl = false;
67  *newline = false;
68  *newline2 = false;
69  while (*p) {
70  clen = utf8.Clen(p);
71  if (!strncmp(p, "\r", clen)) {
72  if (!strncmp(p + 1, "\n", 1)) p++;
73  nls++;
74  } else if (!strncmp(p, "\n", clen)) {
75  if (!strncmp(p + 1, "\r", 1)) p++;
76  nls++;
77  }
78  if (nls && !strncmp(p, " ", 1)) wspstnl = true;
79  p += clen;
80  }
81  if (nls > 1) *newline2 = true;
82  if (nls == 1) {
83  if (!wspstnl && hzone && col > hzone_start && col < hzone_end)
84  *newline = false;
85  else
86  *newline = true;
87  }
88 }
