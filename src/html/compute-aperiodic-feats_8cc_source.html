24 int main(int argc, char *argv[]) {
25  try {
26  using namespace kaldi;
27  const char *usage =
28  "Apply Aperiodic Energy extractor, starting from wav input.\n"
29  "Output is n-dimensional features consisting of log energy\n"
30  "from a set of warped frequency bands. This correspond to the\n"
31  "energy bands of the noise spectrum as a ratio over the original\n"
32  "spectrum. The values on each band will always be negative or 0.\n"
33  "Usage: compute-aperiodic-feats [options...] <wav-rspecifier> "
34  "<f0-rspecifier> <feats-wspecifier>\n"
35  "e.g. compute-aperiodic-feats scp:wav.scp scp:f0.scp ark:- \n"
36  "\n"
37  "See also: compute-kaldi-pitch-feats\n";
38 
39  bool f0_first = false;
40  ParseOptions po(usage);
41  AperiodicEnergyOptions aperiodic_opts;
42 
43  aperiodic_opts.Register(&po);
44  po.Register("pitch-first", &f0_first, "Assume first column is pitch, "
45  "second is POV, for compatibility with get_f0. ");
46 
47  po.Read(argc, argv);
48 
49  if (po.NumArgs() != 3) {
50  po.PrintUsage();
51  exit(1);
52  }
53 
54  std::string wav_rspecifier = po.GetArg(1),
55  f0_rspecifier = po.GetArg(2),
56  feat_wspecifier = po.GetArg(3);
57 
58  SequentialTableReader<WaveHolder> wav_reader(wav_rspecifier);
59  RandomAccessBaseFloatMatrixReader f0_reader(f0_rspecifier);
60  BaseFloatMatrixWriter feat_writer(feat_wspecifier);
61 
62  int32 num_done = 0, num_err = 0;
63  int f0_col = 1, pov_col = 0;
64  // Reverse column order if f0_first
65  if (f0_first) {
66  f0_col = 0;
67  pov_col = 1;
68  }
69  for (; !wav_reader.Done(); wav_reader.Next()) {
70  // TODO(BP): check keys are matching
71  std::string utt = wav_reader.Key();
72  const WaveData &wave_data = wav_reader.Value();
73  const Matrix<BaseFloat> &pitch_features = f0_reader.Value(utt);
74 
75  AperiodicEnergy ap_energy(aperiodic_opts);
76 
77  // Get first channel wave data
78  SubVector<BaseFloat> waveform(wave_data.Data(), 0);
79  int32 num_frames = NumFrames(waveform.Dim(), aperiodic_opts.frame_opts);
80  int32 num_frames_f0 = pitch_features.NumRows();
81  KALDI_LOG << "NF = " << num_frames << "; NR = " << num_frames_f0;
82  /*KALDI_ASSERT(num_frames_f0 == num_frames);*/
83 
84  // F0 is read from second or first column
85  Vector<BaseFloat> f0(num_frames_f0);
86  f0.CopyColFromMat(pitch_features, f0_col);
87  // Pov is read from the other column
88  Vector<BaseFloat> pov(num_frames_f0);
89  pov.CopyColFromMat(pitch_features, pov_col);
90  if (num_frames_f0 != num_frames) {
91  f0.Resize(num_frames, kCopyData);
92  pov.Resize(num_frames, kCopyData);
93  }
94 
95  // Output data
96  Matrix<BaseFloat> features;
97  try {
98  ap_energy.Compute(waveform, pov, f0, &features);
99  } catch(...) {
100  KALDI_WARN << "Failed to compute bndap for utterance "
101  << utt;
102  num_err++;
103  continue;
104  }
105 
106  feat_writer.Write(utt, features);
107  if (num_done % 50 == 0 && num_done != 0)
108  KALDI_VLOG(2) << "Processed " << num_done << " utterances";
109  num_done++;
110  }
111  KALDI_LOG << "Done " << num_done << " utterances, " << num_err
112  << " with errors.";
113  return (num_done != 0 ? 0 : 1);
114  } catch(const std::exception &e) {
115  std::cerr << e.what();
116  return -1;
117  }
118 }
