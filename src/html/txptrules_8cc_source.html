68 bool TxpTrules::Parse(const std::string &tpdb) {
69  TxpPcre pcre_funcs;
70  const pcre* rgx;
71  bool r;
72  r = TxpXmlData::Parse(tpdb);
73  if (r) {
74  rgx = GetRgx("whitespace");
75  if (rgx) rgxwspace_ = rgx;
76  rgx = GetRgx("punctuation");
77  if (rgx) rgxpunc_ = rgx;
78  rgx = GetRgx("alpha");
79  if (rgx) rgxalpha_ = rgx;
80  // default to tokenising on whitespace only
81  tokrgxs_.push_back(pcre_funcs.Compile("^(.*)"));
82  } else {
83  KALDI_WARN << "Error reading normaliser rule file: " << tpdb;
84  }
85  return r;
86 }

111 // Tokenise on whitespace
112 const char* TxpTrules::ConsumeWhitespaceToken(const char* input,
113  std::string* token,
114  std::string* wspace) {
115  TxpPcre pcre;
116  TxpUtf8 utf8;
117  int32 clen;
118  const char *p, *w;
119  // Check for initial whitespace
120  *wspace = "";
121  *token = "";
122  w = pcre.Consume(rgxwspace_, input);
123  if (w) {
124  pcre.SetMatch(0, wspace);
125  return w;
126  }
127  // Go through utf8 chars until we find whitespace
128  p = input;
129  while (*p) {
130  clen = utf8.Clen(p);
131  token->append(p, clen);
132  // increment to next char
133  p += clen;
134  w = pcre.Consume(rgxwspace_, p);
135  if (w) {
136  pcre.SetMatch(0, wspace);
137  return w;
138  }
139  }
140  return p;
141 }

175 const char* TxpTrules::ConsumeToken(const char* input,
176  std::string* token) {
177  TxpPcre pcre;
178  const char* p;
179  TxpUtf8 utf8;
180  int32 clen;
181  RgxVector::iterator it;
182  // go through token regexes until we match
183  for(it = tokrgxs_.begin(); it != tokrgxs_.end(); it++) {
184  p = pcre.Consume(*(it), input);
185  if (p) {
186  pcre.SetMatch(0, token);
187  return p;
188  }
189  }
190  // otherwise throw out the first character as a token
191  clen = utf8.Clen(input);
192  *token = std::string(input, clen);
193  return input + clen;
194 }

197 void TxpTrules::NormCaseCharacter(std::string* norm,
198  TxpTrulesCaseInfo & caseinfo) {
199  const char* p;
200  const std::string* r;
201  std::string c;
202  std::string result;
203  int32 clen, j = 0;
204  bool alpha;
205  TxpUtf8 utf8;
206  TxpPcre pcre;
207  p = norm->c_str();
208  while (*p) {
209  alpha = false;
210  clen = utf8.Clen(p);
211  c = std::string(p, clen);
212  if (pcre.Execute(GetRgx(std::string("alpha")), c)) {
213  alpha = true;
214  caseinfo.lowercase = true;
215  }
216  r = Lkp(std::string("convertillegal"), c);
217  if (r) {
218  c = *r;
219  caseinfo.foreign = true;
220  alpha = true;
221  }
222  r = Lkp(std::string("downcase"), c);
223  if (r) {
224  caseinfo.uppercase = true;
225  alpha = true;
226  c = *r;
227  }
228  if (r && j > 0) caseinfo.capitalised = false;
229  if (!r && j == 0) caseinfo.capitalised = false;
230  if (!alpha) {
231  caseinfo.symbols = true;
232  caseinfo.capitalised = false;
233  }
234  result.append(c);
235  p += clen;
236  j += 1;
237  }
238  *norm = result;
239 }

294 // private member to format lookup table from xml cdata
295 int32 TxpTrules::MakeLkp(LookupMapMap *lkps,
296  const std::string &name,
297  const std::string &cdata) {
298  TxpPcre pcre;
299  int32 pplen;
300  std::string key;
301  std::string val;
302  std::string tmp;
303  const char* pp, *npp;
304 
305  LookupMapMap::iterator itmap;
306  // see if we have already created the LookupMap
307  itmap = lkps->find(name);
308  if (itmap != lkps->end()) {
309  // if so delete it and replace with the new one
310  delete (itmap->second);
311  lkps->erase(itmap);
312  }
313  // add a new lookup table
314  lkps->insert(LookupMapItem(name, new LookupMap));
315  itmap = lkps->find(name);
316 
317  pp = cdata.c_str();
318  pplen = cdata.length();
319  // consume opening {
320  pp = pcre.Consume(lkp_open_, pp, pplen);
321  // no bracket! ignore
322  if (!pp) pp = cdata.c_str();
323  while (*pp) {
324  npp = pcre.Consume(lkp_item_, pp, pplen);
325  if (npp) {
326  pcre.SetMatch(1, &key);
327  pcre.SetMatch(2, &val);
328  (itmap->second)->insert(LookupItem(key, val));
329  pplen -= npp - pp;
330  pp = npp;
331  } else {
332  return false;
333  }
334  }
335  return true;
336 }
