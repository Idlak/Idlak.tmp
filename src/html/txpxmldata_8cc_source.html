65 static const char * _get_str_field(std::string work, char sep, int pos) {
66  /*std::string work = new std::string(str);*/
67  int i, direction = 0;
68  size_t n = 0;
69  size_t next;
70  if (pos < 0) {
71  direction = 1;
72  pos = -pos - 1;
73  n = work.size() - 1;
74  }
75  for (i = 0; i < pos; i++) {
76  if (direction) {
77  next = work.rfind(sep, n);
78  if (next == n) {
79  n = n - 1;
80  continue;
81  }
82  } else {
83  next = work.find(sep, n);
84  if (next == n) {
85  n = n + 1;
86  continue;
87  }
88  }
89  if (next == std::string::npos) return NULL;
90  }
91  size_t sz = static_cast<size_t>(abs(static_cast<long int>(next) - static_cast<long int>(n)));
92  return work.substr(std::min(n, next), sz).c_str();
93  // return work.substr(std::min(n, next), abs(next - n)).c_str();
94 }

100 bool TxpXmlData::Parse(const std::string &tpdb, const std::string &fname) {
101  const char *lang, *region, *acc, *spk;
102  bool binary, fileopen = false;
103  enum XML_Status r;
104  Input ki;
105  std::string dataroot;
106  tpdb_ = tpdb;
107  fname_.clear();
108  // if no fname specified use architecture and default type name
109  if (!fname.empty()) fname_ = fname;
110  else fname_ = type_ + "-" + name_ + ".xml";
111  // get general settings from the configuration
112  lang = GetOptValue("lang");
113  region = GetOptValue("region");
114  acc = GetOptValue("acc");
115  spk = GetOptValue("spk");
116  if (!lang) {
117  KALDI_ERR << "No language specified: i.e. --general-lang=en";
118  }
119  // if fname specified override architecture and default type name
120  // and only accept this file (no default names)
121  if (!fname.empty()) {
122  fname_ = fname;
123  } else {
124  // first search for architecture file <type_>-<name_>.xml
125  fname_ = type_ + "-" + name_ + ".xml";
126  }
127 
128  // If we are in a kaldi flat directory structure look for the file
129  // first as fname_ and second as default value
130  if (ki.Open((tpdb + "/idlak-data-flat").c_str())) {
131  path_ = tpdb_;
132  // search in this directory only
133  if (!ki.Open((path_ + "/" + fname_).c_str(), &binary)) {
134  KALDI_WARN << "Can't find xml data:" << path_ << "/" << fname_;
135  }
136  } else {
137  if (spk && *spk) {
138  path_ = tpdb_ + "/" + lang + "/" + acc + "/" + spk;
139  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
140  }
141  if (!fileopen && acc && *acc) {
142  path_ = tpdb_ + "/" + lang + "/" + acc;
143  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
144  }
145  if (!fileopen && region && *region) {
146  path_ = tpdb_ + "/" + lang + "/" + region;
147  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
148  }
149  if (!fileopen) {
150  path_ = tpdb_ + "/" + lang;
151  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
152  }
153  if (!fileopen) KALDI_WARN << "Can't find xml data in " <<
154  "speaker/accent/region/language directories:" <<
155  "tpdb root:" << path_ << " fname:" << fname_;
156  }
157 
158  // if still not found and fname is not overridden
159  // try to open a default data file
160  if (!fileopen && fname.empty() && name_ != "default") {
161  fname_.clear();
162  fname_ = type_ + "-default.xml";
163  if (ki.Open((tpdb + "/idlak-data-flat").c_str())) {
164  path_ = tpdb_;
165  // search in this directory only
166  if (!ki.Open((path_ + "/" + fname_).c_str(), &binary)) {
167  KALDI_ERR << "Can't find xml data:" << path_ << "/" << fname_;
168  }
169  } else {
170  if (spk && *spk) {
171  path_ = tpdb_ + "/" + lang + "/" + acc + "/" + spk;
172  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
173  }
174  if (!fileopen && acc && *acc) {
175  path_ = tpdb_ + "/" + lang + "/" + acc;
176  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
177  }
178  if (!fileopen && region && *region) {
179  path_ = tpdb_ + "/" + lang + "/" + region;
180  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
181  }
182  if (!fileopen) {
183  path_ = tpdb_ + "/" + lang;
184  fileopen = ki.Open((path_ + "/" + fname_).c_str(), &binary);
185  }
186  }
187  }
188 
189  if (!fileopen) KALDI_ERR << "Can't find xml data in speaker/accent/region/language directories:" <<
190  "tpdb root:" << tpdb_ << " fname:" << fname_;
191 
192  KALDI_VLOG(1) << "Loading: " << path_ << "/" << fname_;
193  while (getline(ki.Stream(), buffer_)) {
194  // Reappend line break to get correct error reporting
195  buffer_.append("\n");
196  r = XML_Parse(parser_, buffer_.c_str(), buffer_.length(), false);
197  if (r == XML_STATUS_ERROR) {
198  KALDI_WARN << "Expat XML Parse error: " <<
199  XML_ErrorString(::XML_GetErrorCode(parser_))
200  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
201  << " Col:" << XML_GetCurrentColumnNumber(parser_);
202  return false;
203  }
204  }
205  XML_Parse(parser_, "", 0, true);
206  ki.Close();
207  return true;
208 }
