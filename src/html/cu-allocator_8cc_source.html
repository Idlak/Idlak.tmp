47 void* CuMemoryAllocator::Malloc(size_t size) {
48  Timer tim;
49  if (!opts_.cache_memory) {
50  void *ans;
51  CU_SAFE_CALL(cudaMalloc(&ans, size));
52  double elapsed = tim.Elapsed();
53  tot_time_taken_ += elapsed;
54  malloc_time_taken_ += elapsed;
55  t_++;
56  return ans;
57  }
58 
59  // We could perhaps change this to KALDI_PARANOID_ASSERT to save time.
60  KALDI_ASSERT(size != 0);
61 
62  // Round up 'size' to a multiple of 256; this ensures the right kind of
63  // memory alignment.
64  size = (size + 255) & ~((size_t)255);
65  void *ans = MallocInternal(size);
66  tot_time_taken_ += tim.Elapsed();
67  return ans;
68 }

71 CuMemoryAllocator::MemoryBlock *CuMemoryAllocator::SplitBlock(
72  MemoryBlock *block, size_t size) {
73  SubRegion *subregion = block->subregion;
74  // new_block will become the right-most part of 'block', and 'block' will
75  // be the left-most part.
76  MemoryBlock *new_block = new MemoryBlock;
77  bool return_new_block;
78  char *new_begin;
79 
80  // We now decide whether to make the left part of 'block' be of size ('size')
81  // and return it (the 'if' branch of the if-else block below), or the right
82  // part (the 'else' branch). We decide this based on heuristics. Basically,
83  // we want to allocate the sub-block that's either next to the edge of the
84  // MemoryRegion, or next to something that was allocated long ago (and which,
85  // we assume won't be deallocated for a relatively long time). That is: we
86  // want to leave the un-allocated memory next to a memory block that was
87  // recently allocated (and thus is likely to be freed sooner), so that when
88  // that block is freed we can merge it with the still-unallocated piece into a
89  // larger block; this will reduce fragmentation. But if this block spans
90  // multiple sub-regions we don't want to do that, as that would be against our
91  // heuristic of, where possible, allocating memory from lower-numbered
92  // sub-regions.
93  //
94  // Bear in mind that we can assume block->next and block->prev, if they are
95  // non-NULL, are both currently allocated, since 'block' is un-allocated and
96  // we would have merged any adjacent un-allocated sub-regions.
97  if (block->next != NULL && block->prev != NULL &&
98  block->prev->t < block->next->t &&
99  block->next->subregion == subregion) {
100  // We'll allocate the right part of the block, since the left side is next
101  // to a relatively recently-allocated block.
102  return_new_block = true;
103  new_begin = block->end - size;
104  } else {
105  // We'll allocate the left part of the block.
106  return_new_block = false;
107  new_begin = block->begin + size;
108  }
109 
110  // The following code makes sure the SubRegion for 'new_block' is correct,
111  // i.e. its 'begin' is >= the 'begin' of the subregion and < the 'end' of the
112  // subregion. If the following loop segfaults, it indicates a bug somewhere
113  // else.
114  while (new_begin >= subregion->end)
115  subregion = subregion->next;
116  MemoryBlock *next_block = block->next;
117  new_block->begin = new_begin;
118  new_block->end = block->end;
119  new_block->subregion = subregion;
120  new_block->allocated = false;
121  new_block->thread_id = block->thread_id;
122  new_block->t = block->t;
123  new_block->next = next_block;
124  new_block->prev = block;
125  if (next_block)
126  next_block->prev = new_block;
127  block->next = new_block;
128  block->end = new_begin;
129 
130  // Add the split-up piece that we won't be allocating, to the
131  // 'free_blocks' member of its subregion.
132  if (return_new_block) {
133  AddToFreeBlocks(block);
134  return new_block;
135  } else {
136  AddToFreeBlocks(new_block);
137  return block;
138  }
139 }

142 void CuMemoryAllocator::RemoveFromFreeBlocks(MemoryBlock *block) {
143  SubRegion *subregion = block->subregion;
144  size_t block_size = block->end - block->begin;
145  std::pair<size_t, MemoryBlock*> p(block_size, block);
146  size_t num_removed = subregion->free_blocks.erase(p);
147  KALDI_ASSERT(num_removed != 0);
148  // Update largest_free_block_, if needed.
149  size_t subregion_index = subregion->subregion_index;
150  if (block_size == largest_free_block_[subregion_index]) {
151  if (subregion->free_blocks.empty())
152  largest_free_block_[subregion_index] = 0;
153  else
154  largest_free_block_[subregion_index] =
155  subregion->free_blocks.begin()->first;
156  }
157 }

159 void CuMemoryAllocator::AddToFreeBlocks(MemoryBlock *block) {
160  SubRegion *subregion = block->subregion;
161  KALDI_PARANOID_ASSERT(block->begin >= subregion->begin &&
162  block->begin < subregion->end);
163  size_t block_size = block->end - block->begin,
164  subregion_index = subregion->subregion_index;
165  // Update largest_free_block_, if needed.
166  if (block_size > largest_free_block_[subregion_index]) {
167  largest_free_block_[subregion_index] = block_size;
168  }
169  subregion->free_blocks.insert(std::pair<size_t, MemoryBlock*>(block_size, block));
170 }

173 void* CuMemoryAllocator::MallocFromSubregion(SubRegion *subregion,
174  size_t size) {
175  // NULL is implementation defined and doesn't have to be zero so we can't
176  // guarantee that NULL will be <= a valid pointer-- so we cast to a pointer
177  // from zero instead of using NULL.
178  std::pair<size_t, MemoryBlock*> p(size, (MemoryBlock*)0);
179 
180  std::set<std::pair<size_t, MemoryBlock*> >::iterator iter =
181  subregion->free_blocks.lower_bound(p);
182  // so now 'iter' is the first member of free_blocks whose size_t value is >=
183  // size. If 'iter' was equal to the end() of that multi_map, it would be a
184  // bug because the calling code checked that the largest free block in this
185  // region was sufficiently large. We don't check this; if it segfaults, we'll
186  // debug.
187 
188  MemoryBlock *block = iter->second;
189  // Erase 'block' from its subregion's free blocks list... the next lines are
190  // similar to RemoveFromFreeBlocks(), but we code it directly as we have the
191  // iterator here, and it would be wasteful to do another lookup.
192  subregion->free_blocks.erase(iter);
193  // Update largest_free_block_, if needed. The following few lines of code also appear
194  // in RemoveFromFreeBlocks().
195  size_t block_size = block->end - block->begin,
196  subregion_index = subregion->subregion_index;
197  if (block_size == largest_free_block_[subregion_index]) {
198  if (subregion->free_blocks.empty())
199  largest_free_block_[subregion_index] = 0;
200  else
201  largest_free_block_[subregion_index] =
202  subregion->free_blocks.begin()->first;
203  }
204 
205  KALDI_PARANOID_ASSERT(block_size >= size && block->allocated == false);
206 
207  // the most memory we allow to be 'wasted' by failing to split a block, is the
208  // smaller of: 1/16 of the size we're allocating, or half a megabyte.
209  size_t allowed_extra_size = std::min<size_t>(size >> 4, 524288);
210  if (block_size > size + allowed_extra_size) {
211  // If the requested block is substantially larger than what was requested,
212  // split it so we don't waste memory.
213  block = SplitBlock(block, size);
214  }
215 
216  if (std::this_thread::get_id() != block->thread_id &&
217  block->t > synchronize_gpu_t_) {
218  // see NOTE ON SYNCHRONIZATION in the header.
219  SynchronizeGpu();
220  synchronize_gpu_t_ = t_;
221  num_synchronizations_++;
222  }
223  block->allocated = true;
224  block->t = t_;
225  allocated_block_map_[block->begin] = block;
226  return block->begin;
227 }

229 // By the time MallocInternal is called, we will have ensured that 'size' is
230 // a nonzero multiple of 256 (for memory aligment reasons).
231 // inline
232 void* CuMemoryAllocator::MallocInternal(size_t size) {
233 start:
234  std::vector<size_t>::const_iterator iter = largest_free_block_.begin(),
235  end = largest_free_block_.end();
236  size_t subregion_index = 0;
237  for (; iter != end; ++iter, ++subregion_index) {
238  if (*iter > size) {
239  return MallocFromSubregion(subregions_[subregion_index], size);
240  }
241  }
242  // We dropped off the loop without finding a subregion with enough memory
243  // to satisfy the request -> allocate a new region.
244  AllocateNewRegion(size);
245  // An infinite loop shouldn't be possible because after calling
246  // AllocateNewRegion(size), there should always be a SubRegion
247  // with that size available.
248  goto start;
249 }

251 // Returns max(0, floor(log_2(i))). Not tested independently.
252 static inline size_t IntegerLog2(size_t i) {
253  size_t ans = 0;
254  while (i > 256) {
255  i >>= 8;
256  ans += 8;
257  }
258  while (i > 16) {
259  i >>= 4;
260  ans += 4;
261  }
262  while (i > 1) {
263  i >>= 1;
264  ans++;
265  }
266  return ans;
267 }

269 std::string GetFreeGpuMemory(int64* free, int64* total) {
270 #ifdef _MSC_VER
271  size_t mem_free, mem_total;
272  cuMemGetInfo_v2(&mem_free, &mem_total);
273 #else
274  // define the function signature type
275  size_t mem_free, mem_total;
276  {
277  // we will load cuMemGetInfo_v2 dynamically from libcuda.so
278  // pre-fill ``safe'' values that will not cause problems
279  mem_free = 1; mem_total = 1;
280  // open libcuda.so
281  void* libcuda = dlopen("libcuda.so", RTLD_LAZY);
282  if (NULL == libcuda) {
283  KALDI_WARN << "cannot open libcuda.so";
284  } else {
285  // define the function signature type
286  // and get the symbol
287  typedef CUresult (*cu_fun_ptr)(size_t*, size_t*);
288  cu_fun_ptr dl_cuMemGetInfo = (cu_fun_ptr)dlsym(libcuda,"cuMemGetInfo_v2");
289  if (NULL == dl_cuMemGetInfo) {
290  KALDI_WARN << "cannot load cuMemGetInfo from libcuda.so";
291  } else {
292  // call the function
293  dl_cuMemGetInfo(&mem_free, &mem_total);
294  }
295  // close the library
296  dlclose(libcuda);
297  }
298  }
299 #endif
300  // copy the output values outside
301  if (NULL != free) *free = mem_free;
302  if (NULL != total) *total = mem_total;
303  // prepare the text output
304  std::ostringstream os;
305  os << "free:" << mem_free/(1024*1024) << "M, "
306  << "used:" << (mem_total-mem_free)/(1024*1024) << "M, "
307  << "total:" << mem_total/(1024*1024) << "M, "
308  << "free/total:" << mem_free/(float)mem_total;
309  return os.str();
310 }

312 void CuMemoryAllocator::PrintMemoryUsage() const {
313  if (!opts_.cache_memory) {
314  KALDI_LOG << "Not caching allocations; time taken in "
315  << "malloc/free is " << malloc_time_taken_
316  << "/" << (tot_time_taken_ - malloc_time_taken_)
317  << ", num operations is " << t_
318  << "; device memory info: "
319  << GetFreeGpuMemory(NULL, NULL);
320  return;
321  }
322 
323  size_t num_blocks_allocated = 0, num_blocks_free = 0,
324  memory_allocated = 0, memory_held = 0,
325  largest_free_block = 0, largest_allocated_block = 0;
326 
327  for (size_t i = 0; i < memory_regions_.size(); i++) {
328  MemoryBlock *m = memory_regions_[i].block_begin;
329  KALDI_ASSERT(m->begin == memory_regions_[i].begin);
330  for (; m != NULL; m = m->next) {
331  size_t size = m->end - m->begin;
332  if (m->allocated) {
333  num_blocks_allocated++;
334  memory_allocated += size;
335  if (size > largest_allocated_block)
336  largest_allocated_block = size;
337  } else {
338  num_blocks_free++;
339  if (size > largest_free_block)
340  largest_free_block = size;
341  }
342  memory_held += size;
343  // The following is just some sanity checks; this code is rarely called so
344  // it's a reasonable place to put them.
345  if (m->next) {
346  KALDI_ASSERT(m->next->prev == m && m->end == m->next->begin);
347  } else {
348  KALDI_ASSERT(m->end == memory_regions_[m->subregion->memory_region].end);
349  }
350  }
351  }
352  KALDI_LOG << "Memory usage: " << memory_allocated << "/"
353  << memory_held << " bytes currently allocated/total-held; "
354  << num_blocks_allocated << "/" << num_blocks_free
355  << " blocks currently allocated/free; largest "
356  << "free/allocated block sizes are "
357  << largest_allocated_block << "/" << largest_free_block
358  << "; time taken total/cudaMalloc is "
359  << tot_time_taken_ << "/" << malloc_time_taken_
360  << ", synchronized the GPU " << num_synchronizations_
361  << " times out of " << (t_/2) << " frees; "
362  << "device memory info: " << GetFreeGpuMemory(NULL, NULL);
363 }

400 void CuMemoryAllocator::Free(void *ptr) {
401  CuTimer tim;
402  if (!opts_.cache_memory) {
403  CU_SAFE_CALL(cudaFree(ptr));
404  tot_time_taken_ += tim.Elapsed();
405  t_++;
406  return;
407  }
408  t_++;
409  unordered_map<void*, MemoryBlock*>::iterator iter =
410  allocated_block_map_.find(ptr);
411  if (iter == allocated_block_map_.end()) {
412  KALDI_ERR << "Attempt to free CUDA memory pointer that was not allocated: "
413  << ptr;
414  }
415  MemoryBlock *block = iter->second;
416  allocated_block_map_.erase(iter);
417  block->t = t_;
418  block->thread_id = std::this_thread::get_id();
419  block->allocated = false;
420 
421  // If this is not the first block of the memory region and the previous block
422  // is not allocated, merge this block into the previous block.
423  MemoryBlock *prev_block = block->prev;
424  if (prev_block != NULL && !prev_block->allocated) {
425  RemoveFromFreeBlocks(prev_block);
426  prev_block->end = block->end;
427  if (prev_block->thread_id != block->thread_id) {
428  // the two blocks we're merging were freed by different threads, so we
429  // give the 'nonexistent thread' as their thread, which means that
430  // whichever thread requests that block, we force synchronization. We can
431  // assume that prev_block was previously allocated (prev_block->t > 0)
432  // because we always start from the left when allocating blocks, and we
433  // know that this block was previously allocated.
434  prev_block->thread_id = std::thread::id();
435  }
436  prev_block->t = t_;
437  prev_block->next = block->next;
438  if (block->next)
439  block->next->prev = prev_block;
440  delete block;
441  block = prev_block;
442  }
443 
444  // If this is not the last block of the memory region and the next block is
445  // not allocated, merge the next block into this block.
446  MemoryBlock *next_block = block->next;
447  if (next_block != NULL && !next_block->allocated) {
448  // merge next_block into 'block', deleting 'next_block'. Note: at this
449  // point, if we merged with the previous block, the variable 'block' may now
450  // be pointing to that previous block, so it would be a 3-way merge.
451  RemoveFromFreeBlocks(next_block);
452  block->end = next_block->end;
453  if (next_block->thread_id != block->thread_id && next_block->t > 0) {
454  // the two blocks we're merging were freed by different threads, so we
455  // give the 'nonexistent thread' as their thread, which means that
456  // whichever thread requests that block, we force synchronization. there
457  // is no need to do this if next_block->t == 0, which would mean it had
458  // never been allocated.
459  block->thread_id = std::thread::id();
460  }
461  // We don't need to inspect the 't' value of next_block; it can't be
462  // larger than t_ because t_ is now.
463  block->next = next_block->next;
464  if (block->next)
465  block->next->prev = block;
466  delete next_block;
467  }
468  AddToFreeBlocks(block);
469  tot_time_taken_ += tim.Elapsed();
470 }

472 void CuMemoryAllocator::AllocateNewRegion(size_t size) {
473  int64 free_memory, total_memory;
474  std::string mem_info = GetFreeGpuMemory(&free_memory, &total_memory);
475  opts_.Check();
476  size_t region_size = static_cast<size_t>(free_memory * opts_.memory_proportion);
477  if (region_size < size)
478  region_size = size;
479  // Round up region_size to an exact multiple of 1M (note: we expect it will
480  // be much larger than that). 1048575 is 2^20 - 1.
481  region_size = (region_size + 1048575) & ~((size_t)1048575);
482 
483  if (!memory_regions_.empty()) {
484  // If this is not the first region allocated, print some information.
485  KALDI_LOG << "About to allocate new memory region of " << region_size
486  << " bytes; current memory info is: " << mem_info;
487  }
488  void *memory_region;
489  cudaError_t e;
490  {
491  Timer tim;
492  e = cudaMalloc(&memory_region, region_size);
493  malloc_time_taken_ += tim.Elapsed();
494  }
495  if (e != cudaSuccess) {
496  PrintMemoryUsage();
497  if (!CuDevice::Instantiate().IsComputeExclusive()) {
498  KALDI_ERR << "Failed to allocate a memory region of " << region_size
499  << " bytes. Possibly this is due to sharing the GPU. Try "
500  << "switching the GPUs to exclusive mode (nvidia-smi -c 3) and using "
501  << "the option --use-gpu=wait to scripts like "
502  << "steps/nnet3/chain/train.py. Memory info: "
503  << mem_info;
504  } else {
505  KALDI_ERR << "Failed to allocate a memory region of " << region_size
506  << " bytes. Possibly smaller minibatch size would help. "
507  << "Memory info: " << mem_info;
508  }
509  }
510  // this_num_subregions would be approximately 'opts_.num_subregions' if
511  // 'region_size' was all the device's memory. (We add one to round up).
512  // We're aiming to get a number of sub-regions approximately equal to
513  // opts_.num_subregions by the time we allocate all the device's memory.
514  size_t this_num_subregions = 1 +
515  (region_size * opts_.num_subregions) / total_memory;
516 
517  size_t memory_region_index = memory_regions_.size();
518  memory_regions_.resize(memory_region_index + 1);
519  MemoryRegion &this_region = memory_regions_.back();
520 
521  this_region.begin = static_cast<char*>(memory_region);
522  this_region.end = this_region.begin + region_size;
523  // subregion_size will be hundreds of megabytes.
524  size_t subregion_size = region_size / this_num_subregions;
525 
526  std::vector<SubRegion*> new_subregions;
527  char* subregion_begin = static_cast<char*>(memory_region);
528  for (size_t i = 0; i < this_num_subregions; i++) {
529  SubRegion *subregion = new SubRegion();
530  subregion->memory_region = memory_region_index;
531  subregion->begin = subregion_begin;
532  if (i + 1 == this_num_subregions) {
533  subregion->end = this_region.end;
534  KALDI_ASSERT(subregion->end > subregion->begin);
535  } else {
536  subregion->end = subregion_begin + subregion_size;
537  subregion_begin = subregion->end;
538  }
539  subregion->next = NULL;
540  if (i > 0) {
541  new_subregions.back()->next = subregion;
542  }
543  new_subregions.push_back(subregion);
544  }
545  // Initially the memory is in a single block, owned by
546  // the first subregion. It will be split up gradually.
547  MemoryBlock *block = new MemoryBlock();
548  block->begin = this_region.begin;
549  block->end = this_region.end;
550  block->subregion = new_subregions.front();
551  block->allocated = false;
552  block->t = 0; // was never allocated.
553  block->next = NULL;
554  block->prev = NULL;
555  for (size_t i = 0; i < this_num_subregions; i++)
556  subregions_.push_back(new_subregions[i]);
557  SortSubregions();
558  this_region.block_begin = block;
559 
560  AddToFreeBlocks(block);
561 }

563 // We sort the sub-regions according to the distance between the start of the
564 // MemoryRegion of which they are a part, and the start of the SubRegion. This
565 // will generally mean that the highest-numbered SubRegion-- the one we keep
566 // free at all costs-- will be the end of the first block which we allocated
567 // (which under most situations will be the largest block).
568 void CuMemoryAllocator::SortSubregions() {
569  largest_free_block_.resize(subregions_.size());
570 
571  std::vector<std::pair<size_t, SubRegion*> > pairs;
572  for (size_t i = 0; i < subregions_.size(); i++) {
573  SubRegion *subregion = subregions_[i];
574  MemoryRegion &memory_region = memory_regions_[subregion->memory_region];
575  size_t distance = subregion->begin - memory_region.begin;
576  pairs.push_back(std::pair<size_t, SubRegion*>(distance, subregion));
577  }
578  std::sort(pairs.begin(), pairs.end());
579  for (size_t i = 0; i < subregions_.size(); i++) {
580  subregions_[i] = pairs[i].second;
581  subregions_[i]->subregion_index = i;
582  if (subregions_[i]->free_blocks.empty())
