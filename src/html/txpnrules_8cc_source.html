127 // get intial token and following whitespace from input and return the rest
128 const char* TxpNRules::ConsumeToken(const char* input,
129  std::string* token,
130  std::string* wspace) {
131  TxpPcre pcre;
132  TxpUtf8 utf8;
133  int32 clen;
134  const char *p, *s, *w, *pre_s;
135  // Check for initial whitespace
136  *wspace = "";
137  *token = "";
138  w = pcre.Consume(rgxwspace_, input);
139  if (w) {
140  pcre.SetMatch(0, wspace);
141  return w;
142  }
143  // Check for initial separator + whitespace
144  clen = utf8.Clen(input);
145  s = pcre.Consume(rgxsep_, input, clen);
146  if (s) {
147  pcre.SetMatch(0, token);
148  w = pcre.Consume(rgxwspace_, s);
149  if (w) {
150  pcre.SetMatch(0, wspace);
151  return w;
152  } else {
153  return s;
154  }
155  }
156  // Go through utf8 chars until we find whitespace or a separator
157  p = input;
158  while (*p) {
159  clen = utf8.Clen(p);
160  token->append(p, clen);
161  // increment to next char
162  p += clen;
163  w = pcre.Consume(rgxwspace_, p);
164  if (w) {
165  pcre.SetMatch(0, wspace);
166  return w;
167  }
168  pre_s = p;
169  clen = utf8.Clen(p);
170  s = pcre.Consume(rgxsep_, p, clen);
171  if (s) return pre_s; // Next token
172  }
173  return p;
174 }

177  std::string* tkout) {

195 const char* TxpNRules::ConsumePunc(const char* input,
196  std::string* prepunc,
197  std::string* token,
198  std::string* pstpunc) {
199  TxpPcre pcre;
200  const char* p;
201  p = pcre.Consume(rgxpunc_, input);
202  if (p) {
203  pcre.SetMatch(0, prepunc);
204  pcre.SetMatch(1, token);
205  pcre.SetMatch(2, pstpunc);
206  return p;
207  } else {
208  *prepunc = input;
209  *token = "";
210  *pstpunc = "";
211  return input += strlen(input);
212  }

215 void TxpNRules::NormCaseCharacter(std::string* norm, TxpCaseInfo & caseinfo) {
216  const char* p;
217  const std::string* r;
218  std::string c;
219  std::string result;
220  int32 clen, j = 0;
221  bool alpha;
222  TxpUtf8 utf8;
223  TxpPcre pcre;
224  p = norm->c_str();
225  while (*p) {
226  alpha = false;
227  clen = utf8.Clen(p);
228  c = std::string(p, clen);
229  if (pcre.Execute(GetRgx(std::string("alpha")), c)) {
230  alpha = true;
231  caseinfo.lowercase = true;
232  }
233  r = Lkp(std::string("convertillegal"), c);
234  if (r) {
235  c = *r;
236  caseinfo.foreign = true;
237  alpha = true;
238  }
239  r = Lkp(std::string("downcase"), c);
240  if (r) {
241  caseinfo.uppercase = true;
242  alpha = true;
243  c = *r;
244  }
245  if (r && j > 0) caseinfo.capitalised = false;
246  if (!r && j == 0) caseinfo.capitalised = false;
247  if (!alpha) {
248  caseinfo.symbols = true;
249  caseinfo.capitalised = false;
250  }
251  result.append(c);
252  p += clen;
253  j += 1;
254  }
255  *norm = result;
256 }

298 // private member to format lookup table from xml cdata
299 int32 TxpNRules::MakeLkp(LookupMapMap *lkps,
300  const std::string &name,
301  const std::string &cdata) {
302  TxpPcre pcre;
303  int32 pplen;
304  std::string key;
305  std::string val;
306  std::string tmp;
307  const char* pp, *npp;
308 
309  LookupMapMap::iterator itmap;
310  // see if we have already created the LookupMap
311  itmap = lkps->find(name);
312  if (itmap == lkps->end()) {
313  // if not add one
314  lkps->insert(LookupMapItem(name, new LookupMap));
315  itmap = lkps->find(name);
316  }
317  pp = cdata.c_str();
318  pplen = cdata.length();
319  // consume opening {
320  pp = pcre.Consume(lkp_open_, pp, pplen);
321  // no bracket! ignore
322  if (!pp) pp = cdata.c_str();
323  while (*pp) {
324  npp = pcre.Consume(lkp_item_, pp, pplen);
325  if (npp) {
326  pcre.SetMatch(1, &key);
327  pcre.SetMatch(2, &val);
328  (itmap->second)->insert(LookupItem(key, val));
329  pplen -= npp - pp;
330  pp = npp;
331  } else {
332  return false;
333  }
334  }
335  return true;
336 }
