39 void TxpLexicon:: EndElement(const char* name) {
40  LookupMap::iterator it;
41  if (!strcmp(name, "lex")) {
42  inlex_ = false;
43  // if word as delimiter char : in it we can't add it
44  if (word_.find(":") != std::string::npos) {
45  KALDI_ERR << "Lexicon error with forbidden ':' character: " << word_;
46  return;
47  }
48  // Only one default for each word is valid
49  if (isdefault_ == "true" || isdefault_ == "True" || isdefault_ == "TRUE") {
50  it = lookup_.find(word_ + std::string(":default"));
51  if (it == lookup_.end()) {
52  lookup_.insert(LookupItem(word_ + std::string(":default"), pron_));
53  } else {
54  KALDI_WARN << "Lexicon error multiple default entries for word: "
55  << word_;
56  return;
57  }
58  }
59  // If entry types repeat only the first will normally be accessed
60  lookup_.insert(LookupItem(word_ + std::string(":") + entry_, pron_));
61  }
62 }

64 int TxpLexicon::GetPron(const std::string &word,
65  const std::string &entry,
66  TxpLexiconLkp* lkp) {
67  LookupMap::iterator it;
68  std::size_t pos;
69  if (!entry.empty())
70  it = lookup_.find(word + std::string(":") + entry);
71  else
72  it = lookup_.find(word + std::string(":default"));
73  if (it != lookup_.end()) {
74  lkp->pron += it->second;
75  // Extract other pronunciations
76  it = lookup_.find(word + std::string(":default"));
77  ++it;
78  pos = (it->first).find(":");
79  while ((it->first).substr(0, pos) == word) {
80  lkp->altprons.push_back(it->second);
81  ++it;
82  if (it == lookup_.end()) break;
83  pos = (it->first).find(":");
84  }
85  return true;
86  } else {
87  return false;
88  }
89 }
