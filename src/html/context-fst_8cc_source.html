26 InverseContextFst::InverseContextFst(
27  Label subsequential_symbol,
28  const vector<int32>& phones,
29  const vector<int32>& disambig_syms,
30  int32 context_width,
31  int32 central_position):
32  context_width_(context_width),
33  central_position_(central_position),
34  phone_syms_(phones),
35  disambig_syms_(disambig_syms),
36  subsequential_symbol_(subsequential_symbol) {
37 
38  { // This block checks the inputs.
39  KALDI_ASSERT(subsequential_symbol != 0
40  && disambig_syms_.count(subsequential_symbol) == 0
41  && phone_syms_.count(subsequential_symbol) == 0);
42  if (phone_syms_.empty())
43  KALDI_WARN << "Context FST created but there are no phone symbols: probably "
44  "input FST was empty.";
45  KALDI_ASSERT(phone_syms_.count(0) == 0 && disambig_syms_.count(0) == 0 &&
46  central_position_ >= 0 && central_position_ < context_width_);
47  for (size_t i = 0; i < phones.size(); i++) {
48  KALDI_ASSERT(disambig_syms_.count(phones[i]) == 0);
49  }
50  }
51 
52  // empty vector, will be the ilabel_info vector that corresponds to epsilon,
53  // in case our FST needs to output epsilons.
54  vector<int32> empty_vec;
55  Label epsilon_label = FindLabel(empty_vec);
56 
57  // epsilon_vec is the phonetic context window we have at the very start of a
58  // sequence, meaning "no real phones have been seen yet".
59  vector<int32> epsilon_vec(context_width_ - 1, 0);
60  StateId start_state = FindState(epsilon_vec);
61 
62  KALDI_ASSERT(epsilon_label == 0 && start_state == 0);
63 
64  if (context_width_ > central_position_ + 1 && !disambig_syms_.empty()) {
65  // We add a symbol whose sequence representation is [ 0 ], and whose
66  // symbol-id is 1. This is treated as a disambiguation symbol, we call it
67  // #-1 in printed form. It is necessary to ensure that all determinizable
68  // LG's will have determinizable CLG's. The problem it fixes is quite
69  // subtle-- it relates to reordering of disambiguation symbols (they appear
70  // earlier in CLG than in LG, relative to phones), and the fact that if a
71  // disambig symbol appears at the very start of a sequence in CLG, it's not
72  // clear exatly where it appeared on the corresponding sequence at the input
73  // of LG.
74  vector<int32> pseudo_eps_vec;
75  pseudo_eps_vec.push_back(0);
76  pseudo_eps_symbol_= FindLabel(pseudo_eps_vec);
77  KALDI_ASSERT(pseudo_eps_symbol_ == 1);
78  } else {
79  pseudo_eps_symbol_ = 0; // use actual epsilon.
80  }
81 }

92 void InverseContextFst::GetFullPhoneSequence(
93  const std::vector<int32> &seq, Label label,
94  std::vector<int32> *full_phone_sequence) {
95  int32 context_width = context_width_;
96  full_phone_sequence->reserve(context_width);
97  full_phone_sequence->insert(full_phone_sequence->end(),
98  seq.begin(), seq.end());
99  full_phone_sequence->push_back(label);
100  for (int32 i = central_position_ + 1; i < context_width; i++) {
101  if ((*full_phone_sequence)[i] == subsequential_symbol_) {
102  (*full_phone_sequence)[i] = 0;
103  }
104  }
105 }

108 InverseContextFst::Weight InverseContextFst::Final(StateId s) {
109  KALDI_ASSERT(static_cast<size_t>(s) < state_seqs_.size());
110 
111  const vector<int32> &phone_context = state_seqs_[s];
112 
113  KALDI_ASSERT(phone_context.size() == context_width_ - 1);
114 
115  bool has_final_prob;
116 
117  if (central_position_ < context_width_ - 1) {
118  has_final_prob = (phone_context[central_position_] == subsequential_symbol_);
119  // if phone_context[central_position_] != subsequential_symbol_ then we have
120  // pending phones-in-context that we still need to output, so we need to
121  // consume more subsequential symbols before we can terminate.
122  } else {
123  has_final_prob = true;
124  }
125  return has_final_prob ? Weight::One() : Weight::Zero();
126 }

128 bool InverseContextFst::GetArc(StateId s, Label ilabel, Arc *arc) {
129  KALDI_ASSERT(ilabel != 0 && static_cast<size_t>(s) < state_seqs_.size() &&
130  state_seqs_[s].size() == context_width_ - 1);
131 
132  if (IsDisambigSymbol(ilabel)) {
133  // A disambiguation-symbol self-loop arc.
134  CreateDisambigArc(s, ilabel, arc);
135  return true;
136  } else if (IsPhoneSymbol(ilabel)) {
137  const vector<int32> &seq = state_seqs_[s];
138  if (!seq.empty() && seq.back() == subsequential_symbol_) {
139  return false; // A real phone is not allowed to follow the subsequential
140  // symbol.
141  }
142 
143  // next_seq will be 'seq' shifted left by 1, with 'ilabel' appended.
144  vector<int32> next_seq(seq);
145  ShiftSequenceLeft(ilabel, &next_seq);
146 
147  // full-seq will be the full context window of size context_width_.
148  vector<int32> full_seq;
149  GetFullPhoneSequence(seq, ilabel, &full_seq);
150 
151  StateId next_s = FindState(next_seq);
152 
153  CreatePhoneOrEpsArc(s, next_s, ilabel, full_seq, arc);
154  return true;
155  } else if (ilabel == subsequential_symbol_) {
156  const vector<int32> &seq = state_seqs_[s];
157 
158  if (central_position_ + 1 == context_width_ ||
159  seq[central_position_] == subsequential_symbol_) {
160  // We already had "enough" subsequential symbols in a row and don't want to
161  // accept any more, or we'd be making the subsequential symbol the central phone.
162  return false;
163  }
164 
165  // full-seq will be the full context window of size context_width_.
166  vector<int32> full_seq;
167  GetFullPhoneSequence(seq, ilabel, &full_seq);
168 
169  vector<int32> next_seq(seq);
170  ShiftSequenceLeft(ilabel, &next_seq);
171  StateId next_s = FindState(next_seq);
172 
173  CreatePhoneOrEpsArc(s, next_s, ilabel, full_seq, arc);
174  return true;
175  } else {
176  KALDI_ERR << "ContextFst: CreateArc, invalid ilabel supplied [confusion "
177  << "about phone list or disambig symbols?]: " << ilabel;
178  }
179  return false; // won't get here. suppress compiler error.
180 }

183 void InverseContextFst::CreateDisambigArc(StateId s, Label ilabel, Arc *arc) {
184  // Creates a self-loop arc corresponding to the disambiguation symbol.
185  vector<int32> label_info; // This will be a vector containing just [ -olabel ].
186  label_info.push_back(-ilabel); // olabel is a disambiguation symbol. Use its negative
187  // so we can more easily distinguish them from phones.
188  Label olabel = FindLabel(label_info);
189  arc->ilabel = ilabel;
190  arc->olabel = olabel;
191  arc->weight = Weight::One();
192  arc->nextstate = s; // self-loop.
193 }

195 void InverseContextFst::CreatePhoneOrEpsArc(StateId src, StateId dest,
196  Label ilabel,
197  const vector<int32> &phone_seq,
198  Arc *arc) {
199  KALDI_PARANOID_ASSERT(phone_seq[central_position_] != subsequential_symbol_);
200 
201  arc->ilabel = ilabel;
202  arc->weight = Weight::One();
203  arc->nextstate = dest;
204  if (phone_seq[central_position_] == 0) {
205  // This can happen at the beginning of the graph. In this case we don't
206  // output a real phone, we createdt an epsilon arc (but sometimes we need to
207  // use a special disambiguation symbol instead of epsilon).
208  arc->olabel = pseudo_eps_symbol_;
209  } else {
210  // We have a phone in the central position.
211  arc->olabel = FindLabel(phone_seq);
212  }
213 }

215 StdArc::StateId InverseContextFst::FindState(const vector<int32> &seq) {
216  // Finds state-id corresponding to this vector of phones. Inserts it if
217  // necessary.
218  KALDI_ASSERT(static_cast<int32>(seq.size()) == context_width_ - 1);
219  VectorToStateMap::const_iterator iter = state_map_.find(seq);
220  if (iter == state_map_.end()) { // Not already in map.
221  StateId this_state_id = (StateId)state_seqs_.size();
222  state_seqs_.push_back(seq);
223  state_map_[seq] = this_state_id;
224  return this_state_id;
225  } else {
226  return iter->second;
227  }
228 }

245 void ComposeContext(const vector<int32> &disambig_syms_in,
246  int32 context_width, int32 central_position,
247  VectorFst<StdArc> *ifst,
248  VectorFst<StdArc> *ofst,
249  vector<vector<int32> > *ilabels_out,
250  bool project_ifst) {
251  KALDI_ASSERT(ifst != NULL && ofst != NULL);
252  KALDI_ASSERT(context_width > 0);
253  KALDI_ASSERT(central_position >= 0);
254  KALDI_ASSERT(central_position < context_width);
255 
256  vector<int32> disambig_syms(disambig_syms_in);
257  std::sort(disambig_syms.begin(), disambig_syms.end());
258 
259  vector<int32> all_syms;
260  GetInputSymbols(*ifst, false/*no eps*/, &all_syms);
261  std::sort(all_syms.begin(), all_syms.end());
262  vector<int32> phones;
263  for (size_t i = 0; i < all_syms.size(); i++)
264  if (!std::binary_search(disambig_syms.begin(),
265  disambig_syms.end(), all_syms[i]))
266  phones.push_back(all_syms[i]);
267 
268  // Get subsequential symbol that does not clash with
269  // any disambiguation symbol or symbol in the FST.
270  int32 subseq_sym = 1;
271  if (!all_syms.empty())
272  subseq_sym = std::max(subseq_sym, all_syms.back() + 1);
273  if (!disambig_syms.empty())
274  subseq_sym = std::max(subseq_sym, disambig_syms.back() + 1);
275 
276  // if central_position == context_width-1, it's left-context, and no
277  // subsequential symbol is needed.
278  if (central_position != context_width-1) {
279  AddSubsequentialLoop(subseq_sym, ifst);
280  if (project_ifst) {
281  fst::Project(ifst, fst::PROJECT_INPUT);
282  }
283  }
284 
285  InverseContextFst inv_c(subseq_sym, phones, disambig_syms,
286  context_width, central_position);
287 
288  // The following statement is equivalent to the following
289  // (if FSTs had the '*' operator for composition):
290  // (*ofst) = inv(inv_c) * (*ifst)
291  ComposeDeterministicOnDemandInverse(*ifst, &inv_c, ofst);
292 
293  inv_c.SwapIlabelInfo(ilabels_out);
294 }

296 void AddSubsequentialLoop(StdArc::Label subseq_symbol,
297  MutableFst<StdArc> *fst) {
298  typedef StdArc Arc;
299  typedef typename Arc::StateId StateId;
300  typedef typename Arc::Weight Weight;
301 
302  vector<StateId> final_states;
303  for (StateIterator<MutableFst<Arc> > siter(*fst); !siter.Done(); siter.Next()) {
304  StateId s = siter.Value();
305  if (fst->Final(s) != Weight::Zero()) final_states.push_back(s);
306  }
307 
308  StateId superfinal = fst->AddState();
309  Arc arc(subseq_symbol, 0, Weight::One(), superfinal);
310  fst->AddArc(superfinal, arc); // loop at superfinal.
311  fst->SetFinal(superfinal, Weight::One());
312 
313  for (size_t i = 0; i < final_states.size(); i++) {
314  StateId s = final_states[i];
315  fst->AddArc(s, Arc(subseq_symbol, 0, fst->Final(s), superfinal));
316  // No, don't remove the final-weights of the original states..
317  // this is so we can add the subsequential loop in cases where
318  // there is no context, and it won't hurt.
319  // fst->SetFinal(s, Weight::Zero());
320  arc.nextstate = final_states[i];
321  }
322 }

344 SymbolTable *CreateILabelInfoSymbolTable(const vector<vector<int32> > &info,
345  const SymbolTable &phones_symtab,
346  std::string separator,
347  std::string initial_disambig) { // e.g. separator = "/", initial-disambig="#-1"
348  KALDI_ASSERT(!info.empty() && !info[0].empty());
349  SymbolTable *ans = new SymbolTable("ilabel-info-symtab");
350  int64 s = ans->AddSymbol(phones_symtab.Find(static_cast<int64>(0)));
351  assert(s == 0);
352  for (size_t i = 1; i < info.size(); i++) {
353  if (info[i].size() == 0) {
354  KALDI_ERR << "Invalid ilabel-info";
355  }
356  if (info[i].size() == 1 &&
357  info[i][0] <= 0) {
358  if (info[i][0] == 0) { // special symbol at start that we want to call #-1.
359  s = ans->AddSymbol(initial_disambig);
360  if (s != i) {
361  KALDI_ERR << "Disambig symbol " << initial_disambig
362  << " already in vocab";
363  }
364  } else {
365  std::string disambig_sym = phones_symtab.Find(-info[i][0]);
366  if (disambig_sym == "") {
367  KALDI_ERR << "Disambig symbol " << -info[i][0]
368  << " not in phone symbol-table";
369  }
370  s = ans->AddSymbol(disambig_sym);
371  if (s != i) {
372  KALDI_ERR << "Disambig symbol " << disambig_sym
373  << " already in vocab";
374  }
375  }
376  } else {
377  // is a phone-context-window.
378  std::string newsym;
379  for (size_t j = 0; j < info[i].size(); j++) {
380  std::string phonesym = phones_symtab.Find(info[i][j]);
381  if (phonesym == "") {
382  KALDI_ERR << "Symbol " << info[i][j]
383  << " not in phone symbol-table";
384  }
385  if (j != 0) newsym += separator;
386  newsym += phonesym;
387  }
388  int64 s = ans->AddSymbol(newsym);
389  if (s != static_cast<int64>(i)) {
390  KALDI_ERR << "Some problem with duplicate symbols";
391  }
392  }
393  }
394  return ans;
395 }
