37 struct TxpLexCompare {
38  public:
39  bool operator()(const std::string x, const std::string y) const {
40  std::size_t p1, p2;
41  p1 = x.find(":");
42  p2 = y.find(":");
43  if (!x.substr(0, p1).compare(y.substr(0, p2))) {
45  if (x.find(":default") != std::string::npos &&
46  y.find(":default") == std::string::npos) {
47  return true;
48  } else if (x.find(":default") == std::string::npos &&
49  y.find(":default") != std::string::npos) {
50  return false;
52  } else {
53  return x.compare(y) < 0;
54  }
55  } else {
57  return (x.substr(0, p1).compare(y.substr(0, p2)) < 0);
58  }
59  }
60 };
