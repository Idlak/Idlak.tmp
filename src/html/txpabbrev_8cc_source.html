58 int32 TxpAbbrev::CheckPrePunc(const char * prepunc, TxpAbbrevInfo * abb) {
59  // no prepunc specified
60  if (abb->prepunc.empty()) return 0;
61  // prepunc matches
62  if (!strcmp(prepunc + strlen(prepunc) - abb->prepunc.size(),
63  abb->prepunc.c_str()))
64  return abb->prepunc.size();
65  // prepunc does not match
66  return NO_INDEX;
67 }

69 int32 TxpAbbrev::CheckPstPunc(const char * pstpunc, TxpAbbrevInfo * abb) {
70  // no pstpunc specified
71  if (abb->pstpunc.empty()) return 0;
72  // pstpunc matches
73  if (!strncmp(pstpunc, abb->pstpunc.c_str(), abb->pstpunc.size()))
74  return abb->pstpunc.size();
75  // prepunc does not match
76  return NO_INDEX;

114 void TxpAbbrev::CharHandler(const char* data, int32 len) {
115  const char * p = NULL, * word = NULL;
116  int32 i = 0, wordlen = 0;
117  p = data;
118  if (!p) return;
119  for (i = 0; i < len; p++, i++) {
120  if (!*p) break;
121  if (*p == ' ' || *p == '\n') {
122  if (word) {
123  current_abbrev_->AddExpansionToken(word, wordlen,
124  current_lexentry_.c_str());
125  word = NULL;
126  wordlen = 0;
127  }
128  } else {
129  if (!word) word = p;
130  wordlen++;
131  }
132  }
133  if (word) current_abbrev_->AddExpansionToken(word, wordlen,
134  current_lexentry_.c_str());
135 }
