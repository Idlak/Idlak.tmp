40 template<class C>
41 void OnlineGenericBaseFeature<C>::AcceptWaveform(BaseFloat sampling_rate,
42  const VectorBase<BaseFloat> &waveform) {
43  BaseFloat expected_sampling_rate = computer_.GetFrameOptions().samp_freq;
44  if (sampling_rate != expected_sampling_rate)
45  KALDI_ERR << "Sampling frequency mismatch, expected "
46  << expected_sampling_rate << ", got " << sampling_rate;
47  if (waveform.Dim() == 0)
48  return; // Nothing to do.
49  if (input_finished_)
50  KALDI_ERR << "AcceptWaveform called after InputFinished() was called.";
51  // append 'waveform' to 'waveform_remainder_.'
52  Vector<BaseFloat> appended_wave(waveform_remainder_.Dim() + waveform.Dim());
53  if (waveform_remainder_.Dim() != 0)
54  appended_wave.Range(0, waveform_remainder_.Dim()).CopyFromVec(
55  waveform_remainder_);
56  appended_wave.Range(waveform_remainder_.Dim(), waveform.Dim()).CopyFromVec(
57  waveform);
58  waveform_remainder_.Swap(&appended_wave);
59  ComputeFeatures();
60 }

62 template<class C>
63 void OnlineGenericBaseFeature<C>::ComputeFeatures() {
64  const FrameExtractionOptions &frame_opts = computer_.GetFrameOptions();
65  int64 num_samples_total = waveform_offset_ + waveform_remainder_.Dim();
66  int32 num_frames_old = features_.size(),
67  num_frames_new = NumFrames(num_samples_total, frame_opts,
68  input_finished_);
69  KALDI_ASSERT(num_frames_new >= num_frames_old);
70  features_.resize(num_frames_new, NULL);
71 
72  Vector<BaseFloat> window;
73  bool need_raw_log_energy = computer_.NeedRawLogEnergy();
74  for (int32 frame = num_frames_old; frame < num_frames_new; frame++) {
75  BaseFloat raw_log_energy = 0.0;
76  ExtractWindow(waveform_offset_, waveform_remainder_, frame,
77  frame_opts, window_function_, &window,
78  need_raw_log_energy ? &raw_log_energy : NULL);
79  Vector<BaseFloat> *this_feature = new Vector<BaseFloat>(computer_.Dim(),
80  kUndefined);
81  // note: this online feature-extraction code does not support VTLN.
82  BaseFloat vtln_warp = 1.0;
83  computer_.Compute(raw_log_energy, vtln_warp, &window, this_feature);
84  features_[frame] = this_feature;
85  }
86  // OK, we will now discard any portion of the signal that will not be
87  // necessary to compute frames in the future.
88  int64 first_sample_of_next_frame = FirstSampleOfFrame(num_frames_new,
89  frame_opts);
90  int32 samples_to_discard = first_sample_of_next_frame - waveform_offset_;
91  if (samples_to_discard > 0) {
92  // discard the leftmost part of the waveform that we no longer need.
93  int32 new_num_samples = waveform_remainder_.Dim() - samples_to_discard;
94  if (new_num_samples <= 0) {
95  // odd, but we'll try to handle it.
96  waveform_offset_ += waveform_remainder_.Dim();
97  waveform_remainder_.Resize(0);
98  } else {
99  Vector<BaseFloat> new_remainder(new_num_samples);
100  new_remainder.CopyFromVec(waveform_remainder_.Range(samples_to_discard,
101  new_num_samples));
102  waveform_offset_ += samples_to_discard;
103  waveform_remainder_.Swap(&new_remainder);
104  }
105  }
106 }

161 void OnlineCmvn::GetMostRecentCachedFrame(int32 frame,
162  int32 *cached_frame,
163  Matrix<double> *stats) {
164  KALDI_ASSERT(frame >= 0);
165  InitRingBufferIfNeeded();
166  // look for a cached frame on a previous frame as close as possible in time
167  // to "frame". Return if we get one.
168  for (int32 t = frame; t >= 0 && t >= frame - opts_.ring_buffer_size; t--) {
169  if (t % opts_.modulus == 0) {
170  // if this frame should be cached in cached_stats_modulo_, then
171  // we'll look there, and we won't go back any further in time.
172  break;
173  }
174  int32 index = t % opts_.ring_buffer_size;
175  if (cached_stats_ring_[index].first == t) {
176  *cached_frame = t;
177  *stats = cached_stats_ring_[index].second;
178  return;
179  }
180  }
181  int32 n = frame / opts_.modulus;
182  if (n >= cached_stats_modulo_.size()) {
183  if (cached_stats_modulo_.size() == 0) {
184  *cached_frame = -1;
185  stats->Resize(2, this->Dim() + 1);
186  return;
187  } else {
188  n = static_cast<int32>(cached_stats_modulo_.size() - 1);
189  }
190  }
191  *cached_frame = n * opts_.modulus;
192  KALDI_ASSERT(cached_stats_modulo_[n] != NULL);
193  *stats = *(cached_stats_modulo_[n]);
194 }

205 void OnlineCmvn::CacheFrame(int32 frame, const Matrix<double> &stats) {
206  KALDI_ASSERT(frame >= 0);
207  if (frame % opts_.modulus == 0) { // store in cached_stats_modulo_.
208  int32 n = frame / opts_.modulus;
209  if (n >= cached_stats_modulo_.size()) {
210  // The following assert is a limitation on in what order you can call
211  // CacheFrame. Fortunately the calling code always calls it in sequence,
212  // which it has to because you need a previous frame to compute the
213  // current one.
214  KALDI_ASSERT(n == cached_stats_modulo_.size());
215  cached_stats_modulo_.push_back(new Matrix<double>(stats));
216  } else {
217  KALDI_WARN << "Did not expect to reach this part of code.";
218  // do what seems right, but we shouldn't get here.
219  cached_stats_modulo_[n]->CopyFromMat(stats);
220  }
221  } else { // store in the ring buffer.
222  InitRingBufferIfNeeded();
223  if (!cached_stats_ring_.empty()) {
224  int32 index = frame % cached_stats_ring_.size();
225  cached_stats_ring_[index].first = frame;
226  cached_stats_ring_[index].second.CopyFromMat(stats);
227  }
228  }
229 }

237 void OnlineCmvn::ComputeStatsForFrame(int32 frame,
238  MatrixBase<double> *stats_out) {
239  KALDI_ASSERT(frame >= 0 && frame < src_->NumFramesReady());
240 
241  int32 dim = this->Dim(), cur_frame;
242  Matrix<double> stats(2, dim + 1);
243  GetMostRecentCachedFrame(frame, &cur_frame, &stats);
244 
245  Vector<BaseFloat> feats(dim);
246  Vector<double> feats_dbl(dim);
247  while (cur_frame < frame) {
248  cur_frame++;
249  src_->GetFrame(cur_frame, &feats);
250  feats_dbl.CopyFromVec(feats);
251  stats.Row(0).Range(0, dim).AddVec(1.0, feats_dbl);
252  stats.Row(1).Range(0, dim).AddVec2(1.0, feats_dbl);
253  stats(0, dim) += 1.0;
254  // it's a sliding buffer; a frame at the back may be
255  // leaving the buffer so we have to subtract that.
256  int32 prev_frame = cur_frame - opts_.cmn_window;
257  if (prev_frame >= 0) {
258  // we need to subtract frame prev_f from the stats.
259  src_->GetFrame(prev_frame, &feats);
260  feats_dbl.CopyFromVec(feats);
261  stats.Row(0).Range(0, dim).AddVec(-1.0, feats_dbl);
262  stats.Row(1).Range(0, dim).AddVec2(-1.0, feats_dbl);
263  stats(0, dim) -= 1.0;
264  }
265  CacheFrame(cur_frame, stats);
266  }
267  stats_out->CopyFromMat(stats);
268 }

271 // static
272 void OnlineCmvn::SmoothOnlineCmvnStats(const MatrixBase<double> &speaker_stats,
273  const MatrixBase<double> &global_stats,
274  const OnlineCmvnOptions &opts,
275  MatrixBase<double> *stats) {
276  int32 dim = stats->NumCols() - 1;
277  double cur_count = (*stats)(0, dim);
278  // If count exceeded cmn_window it would be an error in how "window_stats"
279  // was accumulated.
280  KALDI_ASSERT(cur_count <= 1.001 * opts.cmn_window);
281  if (cur_count >= opts.cmn_window) return;
282  if (speaker_stats.NumRows() != 0) { // if we have speaker stats..
283  double count_from_speaker = opts.cmn_window - cur_count,
284  speaker_count = speaker_stats(0, dim);
285  if (count_from_speaker > opts.speaker_frames)
286  count_from_speaker = opts.speaker_frames;
287  if (count_from_speaker > speaker_count)
288  count_from_speaker = speaker_count;
289  if (count_from_speaker > 0.0)
290  stats->AddMat(count_from_speaker / speaker_count,
291  speaker_stats);
292  cur_count = (*stats)(0, dim);
293  }
294  if (cur_count >= opts.cmn_window) return;
295  if (global_stats.NumRows() != 0) {
296  double count_from_global = opts.cmn_window - cur_count,
297  global_count = global_stats(0, dim);
298  KALDI_ASSERT(global_count > 0.0);
299  if (count_from_global > opts.global_frames)
300  count_from_global = opts.global_frames;
301  if (count_from_global > 0.0)
302  stats->AddMat(count_from_global / global_count,
303  global_stats);
304  } else {
305  KALDI_ERR << "Global CMN stats are required";
306  }
307 }

309 void OnlineCmvn::GetFrame(int32 frame,
310  VectorBase<BaseFloat> *feat) {
311  src_->GetFrame(frame, feat);
312  KALDI_ASSERT(feat->Dim() == this->Dim());
313  int32 dim = feat->Dim();
314  Matrix<double> stats(2, dim + 1);
315  if (frozen_state_.NumRows() != 0) { // the CMVN state has been frozen.
316  stats.CopyFromMat(frozen_state_);
317  } else {
318  // first get the raw CMVN stats (this involves caching..)
319  this->ComputeStatsForFrame(frame, &stats);
320  // now smooth them.
321  SmoothOnlineCmvnStats(orig_state_.speaker_cmvn_stats,
322  orig_state_.global_cmvn_stats,
323  opts_,
324  &stats);
325  }
326 
327  if (!skip_dims_.empty())
328  FakeStatsForSomeDims(skip_dims_, &stats);
329 
330  // call the function ApplyCmvn declared in ../transform/cmvn.h, which
331  // requires a matrix.
332  Matrix<BaseFloat> feat_mat(1, dim);
333  feat_mat.Row(0).CopyFromVec(*feat);
334  // the function ApplyCmvn takes a matrix, so form a one-row matrix to give it.
335  if (opts_.normalize_mean)
336  ApplyCmvn(stats, opts_.normalize_variance, &feat_mat);
337  else
338  KALDI_ASSERT(!opts_.normalize_variance);
339  feat->CopyFromVec(feat_mat.Row(0));
340 }

355 void OnlineCmvn::GetState(int32 cur_frame,
356  OnlineCmvnState *state_out) {
357  *state_out = this->orig_state_;
358  { // This block updates state_out->speaker_cmvn_stats
359  int32 dim = this->Dim();

392 void OnlineSpliceFrames::GetFrame(int32 frame, VectorBase<BaseFloat> *feat) {
393  KALDI_ASSERT(left_context_ >= 0 && right_context_ >= 0);
394  KALDI_ASSERT(frame >= 0 && frame < NumFramesReady());
395  int32 dim_in = src_->Dim();
396  KALDI_ASSERT(feat->Dim() == dim_in * (1 + left_context_ + right_context_));
397  int32 T = src_->NumFramesReady();
398  for (int32 t2 = frame - left_context_; t2 <= frame + right_context_; t2++) {
399  int32 t2_limited = t2;
400  if (t2_limited < 0) t2_limited = 0;
401  if (t2_limited >= T) t2_limited = T - 1;
402  int32 n = t2 - (frame - left_context_); // 0 for left-most frame,
403  // increases to the right.
404  SubVector<BaseFloat> part(*feat, n * dim_in, dim_in);
405  src_->GetFrame(t2_limited, &part);
406  }
407 }

450 void OnlineDeltaFeature::GetFrame(int32 frame,
451  VectorBase<BaseFloat> *feat) {
452  KALDI_ASSERT(frame >= 0 && frame < NumFramesReady());
453  KALDI_ASSERT(feat->Dim() == Dim());
454  // We'll produce a temporary matrix containing the features we want to
455  // compute deltas on, but truncated to the necessary context.
456  int32 context = opts_.order * opts_.window;
457  int32 left_frame = frame - context,
458  right_frame = frame + context,
459  src_frames_ready = src_->NumFramesReady();
460  if (left_frame < 0) left_frame = 0;
461  if (right_frame >= src_frames_ready)
462  right_frame = src_frames_ready - 1;
463  KALDI_ASSERT(right_frame >= left_frame);
464  int32 temp_num_frames = right_frame + 1 - left_frame,
465  src_dim = src_->Dim();
466  Matrix<BaseFloat> temp_src(temp_num_frames, src_dim);
467  for (int32 t = left_frame; t <= right_frame; t++) {
468  SubVector<BaseFloat> temp_row(temp_src, t - left_frame);
469  src_->GetFrame(t, &temp_row);
470  }
471  int32 temp_t = frame - left_frame; // temp_t is the offset of frame "frame"
472  // within temp_src
473  delta_features_.Process(temp_src, temp_t, feat);
474 }
