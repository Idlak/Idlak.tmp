99 // ForwardLinks are the links from a token to a token on the next frame.
100 // or sometimes on the current frame (for input-epsilon links).
101 template <typename Token>
102 struct ForwardLink {
103  using Label = fst::StdArc::Label;
104 
105  Token *next_tok; // the next token [or NULL if represents final-state]
106  Label ilabel; // ilabel on arc
107  Label olabel; // olabel on arc
108  BaseFloat graph_cost; // graph cost of traversing arc (contains LM, etc.)
109  BaseFloat acoustic_cost; // acoustic cost (pre-scaled) of traversing arc
110  ForwardLink *next; // next in singly-linked list of forward arcs (arcs
111  // in the state-level lattice) from a token.
112  inline ForwardLink(Token *next_tok, Label ilabel, Label olabel,
113  BaseFloat graph_cost, BaseFloat acoustic_cost,

121 struct StdToken {
122  using ForwardLinkT = ForwardLink<StdToken>;
123  using Token = StdToken;
124 
125  // Standard token type for LatticeFasterDecoder. Each active HCLG
126  // (decoding-graph) state on each frame has one token.
127 
128  // tot_cost is the total (LM + acoustic) cost from the beginning of the
129  // utterance up to this point. (but see cost_offset_, which is subtracted
130  // to keep it in a good numerical range).
131  BaseFloat tot_cost;
132 
133  // exta_cost is >= 0. After calling PruneForwardLinks, this equals
134  // the minimum difference between the cost of the best path, and the cost of
135  // this is on, and the cost of the absolute best path, under the assumption
136  // that any of the currently active states at the decoding front may
137  // eventually succeed (e.g. if you were to take the currently active states
138  // one by one and compute this difference, and then take the minimum).
139  BaseFloat extra_cost;
140 
141  // 'links' is the head of singly-linked list of ForwardLinks, which is what we
142  // use for lattice generation.
143  ForwardLinkT *links;
144 
145  //'next' is the next in the singly-linked list of tokens for this frame.
146  Token *next;
147 
148  // This function does nothing and should be optimized out; it's needed
149  // so we can share the regular LatticeFasterDecoderTpl code and the code
150  // for LatticeFasterOnlineDecoder that supports fast traceback.
151  inline void SetBackpointer (Token *backpointer) { }
152 
153  // This constructor just ignores the 'backpointer' argument. That argument is
154  // needed so that we can use the same decoder code for LatticeFasterDecoderTpl
155  // and LatticeFasterOnlineDecoderTpl (which needs backpointers to support a
156  // fast way to obtain the best path).
157  inline StdToken(BaseFloat tot_cost, BaseFloat extra_cost, ForwardLinkT *links,
158  Token *next, Token *backpointer):
159  tot_cost(tot_cost), extra_cost(extra_cost), links(links), next(next) { }
160 };

372  // FindOrAddToken either locates a token in hash of toks_, or if necessary
373  // inserts a new, empty token (i.e. with no forward links) for the current
374  // frame. [note: it's inserted if necessary into hash toks_ and also into the
375  // singly linked list of tokens active on this frame (whose head is at
376  // active_toks_[frame]). The frame_plus_one argument is the acoustic frame
377  // index plus one, which is used to index into the active_toks_ array.
378  // Returns the Token pointer. Sets "changed" (if non-NULL) to true if the
379  // token was newly created or the cost changed.
380  // If Token == StdToken, the 'backpointer' argument has no purpose (and will

386  // prunes outgoing links for all tokens in active_toks_[frame]
387  // it's called by PruneActiveTokens
388  // all links, that have link_extra_cost > lattice_beam are pruned
389  // delta is the amount by which the extra_costs must change
390  // before we set *extra_costs_changed = true.
391  // If delta is larger, we'll tend to go back less far
392  // toward the beginning of the file.
393  // extra_costs_changed is set to true if extra_cost was changed for any token
394  // links_pruned is set to true if any link in any token was pruned
395  void PruneForwardLinks(int32 frame_plus_one, bool *extra_costs_changed,
396  bool *links_pruned,
397  BaseFloat delta);
398 
399  // This function computes the final-costs for tokens active on the final
400  // frame. It outputs to final-costs, if non-NULL, a map from the Token*
401  // pointer to the final-prob of the corresponding state, for all Tokens
402  // that correspond to states that have final-probs. This map will be
403  // empty if there were no final-probs. It outputs to
404  // final_relative_cost, if non-NULL, the difference between the best
405  // forward-cost including the final-prob cost, and the best forward-cost
406  // without including the final-prob cost (this will usually be positive), or
407  // infinity if there were no final-probs. [c.f. FinalRelativeCost(), which
408  // outputs this quanitity]. It outputs to final_best_cost, if
409  // non-NULL, the lowest for any token t active on the final frame, of
410  // forward-cost[t] + final-cost[t], where final-cost[t] is the final-cost in
411  // the graph of the state corresponding to token t, or the best of
412  // forward-cost[t] if there were no final-probs active on the final frame.
413  // You cannot call this after FinalizeDecoding() has been called; in that
414  // case you should get the answer from class-member variables.
415  void ComputeFinalCosts(unordered_map<Token*, BaseFloat> *final_costs,
416  BaseFloat *final_relative_cost,
417  BaseFloat *final_best_cost) const;
