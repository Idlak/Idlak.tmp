26 void AgglomerativeClusterer::Cluster() {
27  KALDI_VLOG(2) << "Initializing cluster assignments.";
28  Initialize();
29 
30  KALDI_VLOG(2) << "Clustering...";
31  // This is the main algorithm loop. It moves through the queue merging
32  // clusters until a stopping criterion has been reached.
33  while (num_clusters_ > min_clust_ && !queue_.empty()) {
34  std::pair<BaseFloat, std::pair<uint16, uint16> > pr = queue_.top();
35  int32 i = (int32) pr.second.first, j = (int32) pr.second.second;
36  queue_.pop();
37  // check to make sure clusters have not already been merged
38  if ((active_clusters_.find(i) != active_clusters_.end()) &&
39  (active_clusters_.find(j) != active_clusters_.end()))
40  MergeClusters(i, j);
41  }
42 
43  std::vector<int32> new_assignments(num_points_);
44  int32 label_id = 0;
45  std::set<int32>::iterator it;
46  // Iterate through the clusters and assign all utterances within the cluster
47  // an ID label unique to the cluster. This is the final output and frees up
48  // the cluster memory accordingly.
49  for (it = active_clusters_.begin(); it != active_clusters_.end(); ++it) {
50  ++label_id;
51  AhcCluster *cluster = clusters_map_[*it];
52  std::vector<int32>::iterator utt_it;
53  for (utt_it = cluster->utt_ids.begin();
54  utt_it != cluster->utt_ids.end(); ++utt_it)
55  new_assignments[*utt_it] = label_id;
56  delete cluster;
57  }
58  assignments_->swap(new_assignments);
59 }

68 void AgglomerativeClusterer::Initialize() {
69  KALDI_ASSERT(num_clusters_ != 0);
70  for (int32 i = 0; i < num_points_; i++) {
71  // create an initial cluster of size 1 for each point
72  std::vector<int32> ids;
73  ids.push_back(i);
74  AhcCluster *c = new AhcCluster(++count_, -1, -1, ids);
75  clusters_map_[count_] = c;
76  active_clusters_.insert(count_);
77 
78  // propagate the queue with all pairs from the cost matrix
79  for (int32 j = i+1; j < num_clusters_; j++) {
80  BaseFloat cost = costs_(i,j);
81  cluster_cost_map_[std::make_pair(i+1, j+1)] = cost;
82  if (cost <= thresh_)
83  queue_.push(std::make_pair(cost,
84  std::make_pair(static_cast<uint16>(i+1),
85  static_cast<uint16>(j+1))));
86  }
87  }
88 }

90 void AgglomerativeClusterer::MergeClusters(int32 i, int32 j) {
91  AhcCluster *clust1 = clusters_map_[i];
92  AhcCluster *clust2 = clusters_map_[j];
93  // For memory efficiency, the first cluster is updated to contain the new
94  // merged cluster information, and the second cluster is later deleted.
95  clust1->id = ++count_;
96  clust1->parent1 = i;
97  clust1->parent2 = j;
98  clust1->size += clust2->size;
99  clust1->utt_ids.insert(clust1->utt_ids.end(), clust2->utt_ids.begin(),
100  clust2->utt_ids.end());
101  // Remove the merged clusters from the list of active clusters.
102  active_clusters_.erase(i);
103  active_clusters_.erase(j);
104  // Update the queue with all the new costs involving the new cluster
105  std::set<int32>::iterator it;
106  for (it = active_clusters_.begin(); it != active_clusters_.end(); ++it) {
107  // The new cost is the sum of the costs of the new cluster's parents
108  BaseFloat new_cost = GetCost(*it, i) + GetCost(*it, j);
109  cluster_cost_map_[std::make_pair(*it, count_)] = new_cost;
110  BaseFloat norm = clust1->size * (clusters_map_[*it])->size;
111  if (new_cost / norm <= thresh_)
112  queue_.push(std::make_pair(new_cost / norm,
113  std::make_pair(static_cast<uint16>(*it),
114  static_cast<uint16>(count_))));
115  }
116  active_clusters_.insert(count_);
117  clusters_map_[count_] = clust1;
118  delete clust2;
119  num_clusters_--;
120 }
