30 static void UnitTestAperiodic() {
31  WaveData wave;
32  {
33  std::ifstream is("../feat/test_data/test.wav");
34  wave.Read(is);
35  }
36  KALDI_ASSERT(wave.Data().NumRows() == 1);
37  SubVector<BaseFloat> waveform(wave.Data(), 0);
38 
39  Matrix<BaseFloat> f0_info;
40  {
41  bool binary;
42  Input in("../feat/test_data/test.f0", &binary);
43  f0_info.Read(in.Stream(), binary);
44  }
45 
46  AperiodicEnergyOptions opts;
47  opts.frame_opts.snip_edges = false;
48  opts.frame_opts.frame_length_ms = 30; // That's what was used for get_f0
49  AperiodicEnergy ap_energy(opts);
50  int32 num_frames = NumFrames(waveform.Dim(), opts.frame_opts);
51  int32 num_frames_f0 = f0_info.NumRows();
52  KALDI_LOG << "NF = " << num_frames << "; NR = " << f0_info.NumRows();
53 // KALDI_ASSERT(f0_info.NumRows() == num_frames);
54  Vector<BaseFloat> f0(num_frames_f0);
55  f0.CopyColFromMat(f0_info, 1);
56  Vector<BaseFloat> pov(num_frames_f0);
57  pov.CopyColFromMat(f0_info, 0);
58 
59  Matrix<BaseFloat> m;
60  ap_energy.Compute(waveform, pov, f0, &m);
61 
62  // Very basic test: we check we have the right numbers of rows and columns,
63  // and that the values are within a reasonable range.
64  KALDI_ASSERT(m.NumRows() == num_frames);
65  for (int32 r = 0; r < num_frames; r++) {
66  SubVector<BaseFloat> this_ap_energy(m.Row(r));
67  KALDI_ASSERT(this_ap_energy.Dim() == opts.banks_opts.num_bins);
68  for (int32 c = 0; c < this_ap_energy.Dim(); c++) {
69  KALDI_ASSERT(-1000.0 < this_ap_energy(c) && this_ap_energy(c) <= 0.0);
70  }
71  }
72 }
73 }
