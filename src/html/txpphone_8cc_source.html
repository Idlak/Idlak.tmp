41 void TxpPhone::StartElement(const char* name, const char** atts) {
42  TxpPhoneDescr p_descr;
43  if (!strcmp(name, "phone")) {
44  SetAttribute("name", atts, &phon_name_);
45  } else if (!strcmp(name, "description")) {
46  p_descr.name = phon_name_;

49  SetAttribute("ipa", atts, &(p_descr.ipa));
50  phonemes_.insert(PhoneItem(phon_name_, p_descr));
