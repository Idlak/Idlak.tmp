146 // hold information on a feature function defined in cex-<architecture>.xml
147 struct TxpCexspecFeat {
148  // name of feature
149  std::string name;
150  // htsname of feature (for information only)
151  std::string htsname;
152  // description of function (for information only)
153  std::string desc;
154  // delimiter used before feature in model name
155  std::string delim;
156  // value when no feature value is meaningful
157  std::string nullvalue;
158  // pointer to the extraction function
159  cexfunction func;
160  // whether to allow cross silence context pause
161  bool pause_context;
162  // the type of fuction (string or integer)
163  enum CEXSPEC_TYPE type;
164  // name of the valid set of values if a string type fucntion
165  std::string set;
166  // maximum value if an integer type value
167  int32 max;
168  // minimum value if an integer type value
169  int32 min;
170  // mapping from specific feature extraction values
171  // to architecture specific values
172  LookupMap mapping;
173 };
