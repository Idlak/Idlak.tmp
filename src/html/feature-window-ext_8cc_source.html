27  // offset is the amount at the start that has been extracted.
28  int32 offset = num_frames * frame_shift;
29  KALDI_ASSERT(wave_remainder != NULL);
30  int32 remaining_len = wave.Dim() - offset;

32  KALDI_ASSERT(remaining_len >= 0);
33  if (remaining_len > 0)
34  wave_remainder->CopyFromVec(SubVector<BaseFloat>(wave, offset, remaining_len));
35 }
