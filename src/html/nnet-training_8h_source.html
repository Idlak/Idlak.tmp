119 // This struct is used in multiple nnet training classes for keeping
120 // track of objective function values.
121 // Also see struct AccuracyInfo, in nnet-diagnostics.h.
122 struct ObjectiveFunctionInfo {
123  int32 current_phase;
124  int32 minibatches_this_phase; // The number of minibatches' worth of stats that
125  // we accumulated in the phase numbered
126  // 'current_phase'.
127  double tot_weight;
128  double tot_objf;
129  double tot_aux_objf; // An 'auxiliary' objective function that is optional-
130  // may be used when things like regularization are being
131  // used.
132 
133  double tot_weight_this_phase;
134  double tot_objf_this_phase;
135  double tot_aux_objf_this_phase;
136 
137  ObjectiveFunctionInfo():
138  current_phase(0),
139  minibatches_this_phase(0),
140  tot_weight(0.0), tot_objf(0.0), tot_aux_objf(0.0),
141  tot_weight_this_phase(0.0), tot_objf_this_phase(0.0),
142  tot_aux_objf_this_phase(0.0) { }
143 
144  // This function updates the stats and, if the phase has just changed,
145  // prints a message indicating progress. The phase equals
146  // minibatch_counter / minibatches_per_phase. Its only function is to
147  // control how frequently we print logging messages.
148  void UpdateStats(const std::string &output_name,
149  int32 minibatches_per_phase,
150  int32 minibatch_counter,
151  BaseFloat this_minibatch_weight,
152  BaseFloat this_minibatch_tot_objf,
153  BaseFloat this_minibatch_tot_aux_objf = 0.0);
154 
155  // Prints stats for the current phase.
156  // Note: 'phase' will normally be this->current_phase + 1, but may under
157  // unusual circumstances (e.g. multilingual training, where not all outputs
158  // are seen on all minibatches) be larger than that.
159  void PrintStatsForThisPhase(const std::string &output_name,
160  int32 minibatches_per_phase,
161  int32 phase) const;
162  // Prints total stats, and returns true if total stats' weight was nonzero.
163  bool PrintTotalStats(const std::string &output_name) const;
164 };
