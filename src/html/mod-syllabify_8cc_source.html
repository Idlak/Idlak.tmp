39 bool TxpSyllabify::Process(pugi::xml_document* input) {
40  std::string sylpron;
41  PhoneVector pvector;
42  pugi::xpath_node_set spts = input->document_element().select_nodes("//spt");
43  spts.sort();
44  for (pugi::xpath_node_set::const_iterator it = spts.begin();
45  it != spts.end();
46  ++it) {
47  pugi::xml_node pre_node;
48  pugi::xml_node spt = (*it).node();
49  pugi::xpath_node_set tks = spt.select_nodes("descendant::tk");
50  tks.sort();
51  for (pugi::xpath_node_set::const_iterator it2 = tks.begin();
52  it2 != tks.end();
53  ++it2) {
54  pugi::xml_node node = (*it2).node();
55  if (!node.attribute("pron").empty() && strlen(node.attribute("pron").value()) > 0) {
56  sylmax_.GetPhoneVector(node.attribute("pron").value(), &pvector);
57  if (it2 == tks.end() - 1) pvector[pvector.size() - 1].cross_word = false;
58  sylmax_.Maxonset(&pvector);
59  if (it2 != tks.begin()) {
60  sylmax_.Writespron(&pvector, &sylpron);
61  pre_node.append_attribute("spron").set_value(sylpron.c_str());
62  // add syllabic xml structure
63  _add_sylxml(sylpron.c_str(), &pre_node);
64  }
65  pre_node = node;
66  }
67  else {
68  node.append_attribute("spron").set_value("");
69  }
70  }
71  if (!pre_node.empty()) {
72  sylmax_.Writespron(&pvector, &sylpron);
73  pre_node.append_attribute("spron").set_value(sylpron.c_str());
74  // add syllabic xml structure
75  _add_sylxml(sylpron.c_str(), &pre_node);
76  }
77  }
78  return true;
79 }

81 static void _add_sylxml(const char* spron, pugi::xml_node* node) {
82  // p is an interative pointer to the syllabified string
83  // phone points to the next phone, plen its string length
84  // syl points to the syllable, slen its string length
85  // stress points to the nucleus type (in English 0 - no stress
86  // 2 - secondary stress, 1 - stressed)
87  // type keeps track of syllable subtype 0 - onset, 1 - nucleus
88  // 2 - coda
89  const char *p, *phone = NULL, *syl = NULL, *stress = "";
90  int32 plen = 0, slen = 0, type = 0, sylid = 1, phonid = 1;
91  // buffers used to create null terminated strings
92  std::string s;
93  std::string s2;
94  // used to add xml structure
95  pugi::xml_node sylnode, phonenode;
96  sylnode = node->append_child("syl");
97  for (p = spron; *p; p++) {
98  // Syllable marker
99  if (*p == '|' || *p == '_' || *p == '+') {
100  // output phone and phone syllable type
101  if (phone) {
102  s = std::string(phone, plen);
103  phonenode = sylnode.append_child("phon");
104  phonenode.append_attribute("val").set_value(s.c_str());
105  phonenode.append_attribute("type").set_value(txpsyllabletype[type]);
106  phonenode.append_attribute("phonid").set_value(phonid);
107  phonid++;
108  }
109  // if (phone) std::cout << s.c_str() << ' ' << type << '\n';
110  // reset phone
111  phone = NULL;
112  plen = 0;
113  // update type
114  if (*p == '+') {
115  type++;
116  slen++;
117  // std::cout << *p << " s" << slen << "\n";
118  } else if (*p == '|') {
119  // end of syllable
120  // output syllable
121  if (syl) {
122  s = std::string(syl, slen);
123  s2 = std::string(stress, 1);
124  sylnode.append_attribute("val").set_value(s.c_str());
125  sylnode.append_attribute("stress").set_value(s2.c_str());
126  sylnode.append_attribute("sylid").set_value(sylid);
127  sylnode.append_attribute("nophons").set_value(phonid - 1);
128  }
129  // if (syl) std::cout << "H" << s.c_str() << ' ' << *stress << '\n';
130  // reset syllable, stress and type
131  if (*(p + 1)) sylnode = node->append_child("syl");
132  syl = NULL;
133  slen = 0;
134  sylid++;
135  stress = "";
136  type = 0;
137  phonid = 1;
138  } else {
139  // between phone marker '_'
140  slen++;
141  // std::cout << *p << " s" << slen << "\n";
142  }
143  } else if (*p >= '0' && *p <= '9') {
144  // phone type (stress) marker
145  // set to nucleus type
146  type = 1;
147  // output phone without stress marker
148  if (phone) {
149  s = std::string(phone, plen);
150  phonenode = sylnode.append_child("phon");
151  phonenode.append_attribute("val").set_value(s.c_str());
152  phonenode.append_attribute("type").set_value(txpsyllabletype[type]);
153  phonenode.append_attribute("phonid").set_value(phonid);
154  phonid++;
155  }
156  // if (phone) std::cout << s.c_str() << ' ' << type << '\n';
157  // reset phone
158  phone = NULL;
159  plen = 0;
160  // update stress value
161  stress = p;
162  slen++;
163  // std::cout << *p << " n" << slen << "\n";
164  } else {
165  // other i.e. phone symbol - increment
166  if (!phone) phone = p;
167  plen++;
168  if (!syl) syl = p;
169  slen++;
170  // std::cout << *p << " e" << slen << "\n";
171  }
172  }
173  // output any phone/syllable left over (in case syllable
174  // not properly terminated
175  if (phone) {
176  s = std::string(phone, plen);
177  phonenode = sylnode.append_child("phon");
178  phonenode.append_attribute("val").set_value(s.c_str());
179  phonenode.append_attribute("type").set_value(txpsyllabletype[type]);
180  phonenode.append_attribute("phonid").set_value(phonid);
181  phonid++;
182  }
183  // if (phone) std::cout << s.c_str() << ' ' << type << '\n';
184  if (syl) {
185  s = std::string(syl, slen);
186  s2 = std::string(stress, 1);
187  }
188  // if (syl) std::cout << s.c_str() << ' ' << *stress << '\n';
189  if (syl) {
190  sylnode = node->append_child("syl");
191  sylnode.append_attribute("val").set_value(s.c_str());
192  sylnode.append_attribute("stress").set_value(s2.c_str());
193  sylnode.append_attribute("sylid").set_value(sylid);
194  sylnode.append_attribute("nophons").set_value(phonid - 1);
195  sylid++;
196  }
197  sylnode.parent().append_attribute("nosyl").set_value(sylid - 1);
198  }
