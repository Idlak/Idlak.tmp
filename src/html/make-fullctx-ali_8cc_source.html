19 // Full context alignments have the form
20 // <phone contexts> <full context> <state no> as an integer vector for
21 // each frame.
22 // Full context is output by cex idlak voice build module and assigns a
23 // full context for each phone.
24 // the align idlak voice build module produces a standard quin phone alignment
25 // by transition id.
26 // This program takes this alignment, extracts the phone context and state no
27 // for each frame and merges it with the phone full context information.
28 // The process is complicated by the fact that initial and final pauses may be
29 // missing from the data while assummed in the full context information.
30 // Current policy is to remove context information for leading and final pauses
31 // if they are not in the alignment and set all context information to 0.
32 // Models in any built tree for silence data are then ignored.

62  int maxsilphone = 1;
63  ParseOptions po(usage);
64  po.Register("phone-context", &phonecontext, "Size of the phone context, e.g. 3 for triphone.");
65  po.Register("mid-context", &midcontext, "Position of the middle phone, e.g. 1 for triphone.");
66  po.Register("max-sil-phone", &maxsilphone, "Maximum value of silence phone");
67  po.Read(argc, argv);
68 
69  if (po.NumArgs() != 4) {

72  }

74  std::string model_filename = po.GetArg(1);
75  std::string old_alignments_rspecifier = po.GetArg(2);
76  std::string contexts_rspecifier = po.GetArg(3);
77  std::string new_alignments_wspecifier = po.GetArg(4);
78 
79  TransitionModel trans_model;
80  ReadKaldiObject(model_filename, &trans_model);
81  SequentialInt32VectorReader alignment_reader(old_alignments_rspecifier);
82  SequentialInt32VectorVectorReader contexts_reader(contexts_rspecifier);
83  Int32VectorVectorWriter alignment_writer(new_alignments_wspecifier);
84 
85  int num_success = 0, num_fail = 0;
86 
87  for (; !contexts_reader.Done() && !alignment_reader.Done();
88  contexts_reader.Next(), alignment_reader.Next()) {
89  std::string ctxkey = alignment_reader.Key();
90  const std::vector< std::vector<int32> > &contexts =
91  contexts_reader.Value();
92  std::string key = alignment_reader.Key();
93  const std::vector<int32> &old_alignment = alignment_reader.Value();
94 
95  // Check the keys are the same
96  if (ctxkey != key) {
97  KALDI_WARN << "Could not merge alignment and contexts for key " << key
98  <<" (missing data?)";
99  num_fail++;
100  continue;
101  }
102  /*std::cout << ctxkey << " " << key << "\n";*/
103  std::vector< std::vector<int32> > phone_windows;
104  std::vector<int32> phones;
105  std::vector<int32> states;
106  std::vector< std::vector<int32> > output;
107  std::vector<int32> curphone;
108 
109  for (size_t i = 0; i < old_alignment.size(); i++)
110  states.push_back(trans_model.TransitionIdToHmmState(old_alignment[i]));
111 
112  if (GetPhoneWindows(trans_model,
113  old_alignment,
114  phonecontext,
115  midcontext,
116  true,
117  &phone_windows,
118  &phones)) {
119  last_phone = -1;
120  phone_index = 0;
121  for (v = phone_windows.begin(), j = 0;
122  v != phone_windows.end();
123  v++, j++) {
124  curphone = *v;
125  if ( curphone[midcontext] != last_phone ) {
126  if (last_phone >= 0) phone_index++;
127  }
128  // check we have a silence at start of utt if not
129  // skip silence in context data
130  if (!phone_index && curphone[midcontext] > maxsilphone)
131  phone_index++;
132  if (phone_index >= contexts.size()) {
133  break;
134  }
135  /*std::cout << phone_index << " " << curphone[midcontext] << "\n";*/
136  // first item in context is the mid point phone - ignore it
137  for (k = 1; k < contexts[phone_index].size(); k++)
138  curphone.push_back(contexts[phone_index][k]);
139  curphone.push_back(states[j]);
140  output.push_back(curphone);
141  last_phone = curphone[midcontext];
142  }
143  if (curphone[midcontext] > maxsilphone) phone_index++;
144  if (phone_index + 1 != contexts.size()) {
145  KALDI_WARN << "Merge of alignment and contexts failed for key " << key
146  <<" mismatching number of phones contexts:"
147  << contexts.size()
148  <<" alignment:" << phone_index + 1;
149  num_fail++;
150  continue;
151  }
152 
153  alignment_writer.Write(key, output);
154  num_success++;
155  } else {
156  KALDI_WARN << "Could not convert alignment for key " << key
157  <<" (possibly truncated alignment?)";
158  num_fail++;
159  }
160  }
161  } catch(const std::exception &e) {
162  std::cerr << e.what();
163  return -1;
164  }
165 }
