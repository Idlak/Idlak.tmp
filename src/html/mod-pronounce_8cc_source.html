38 bool TxpPronounce::Process(pugi::xml_document* input) {
39  pugi::xml_node parent, child;
40  pugi::xpath_node_set tks = input->document_element().select_nodes("//tk");
41  const char* lex_entry;
42  const char* lex_pron;
43  const char* word;
44  std::string utfchar, word_str, altprons;
45  TxpLexiconLkp lexlkp;
46  int32 i;
47  tks.sort();
48  for (pugi::xpath_node_set::const_iterator it = tks.begin();
49  it != tks.end();
50  ++it) {
51  pugi::xml_node node = (*it).node();
52  word = node.attribute("norm").value();
53  if (std::string(word).compare("") == 0)
54  word = node.attribute("tknorm").value();
55  lexlkp.Reset();
56  // Check to see if token is first daughter of a lex tag
57  parent = node.parent();
58  lex_entry = NULL;
59  lex_pron = NULL;
60  while (parent) {
61  if (!strcmp(parent.name(), "lex")) {
62  if ((*(parent.select_nodes("descendant::tk[1]").begin())).node()
63  == node) {
64  lex_entry = parent.attribute("entry").value();
65  lex_pron = parent.attribute("pron").value();
66  }
67  }
68  parent = parent.parent();
69  }
70  word_str = std::string(word);
71  // If pron is set use that pron
72  if (lex_pron && *lex_pron) {
73  node.append_attribute("pron").set_value(lex_pron);
74  } else {
75  // standard lookup of word
76  AppendPron(lex_entry, std::string(word), &lexlkp);
77  // in the received pronounciation lookup phonemes and check if they
78  // exist in the phoneme set
79  int32 st_phon_pos = 0, end_phon_pos = 0;
80  std::string sub_word = lexlkp.pron.substr(st_phon_pos);
81  std::string phone_2_chk = "";
82  while (sub_word.length() > 0) {
83  end_phon_pos = sub_word.find(" ");
84  if (end_phon_pos == std::string::npos) {
85  end_phon_pos = sub_word.length()-1;
86  phone_2_chk = sub_word;
87  } else
88  phone_2_chk = sub_word.substr(0, end_phon_pos);
89 
90  // if the phoneme has a stress, remove it
91  if (isdigit(phone_2_chk.substr(phone_2_chk.length()-1)[0])) {
92  phone_2_chk = phone_2_chk.substr(0, phone_2_chk.length()-1);
93  }
94 
95  // if a phoneme turns out to be an empty string, then it is wrong and
96  // the pronounciation is removed.
97  if (phone_2_chk.compare("") == 0) {
98  lexlkp.pron = "";
99  break;
100  }
101 
102  // if no phoneme is found, pronounciation is removed
103  const TxpPhoneDescr* found_phone = phone_.GetPhone(phone_2_chk.c_str());
104  if (found_phone == NULL) {
105  KALDI_ERR << "Phoneme " << phone_2_chk << " from word " << word << " could not be found";
106  lexlkp.pron = "";
107  break;
108  }
109 
110  st_phon_pos = end_phon_pos + 1;
111  sub_word = sub_word.substr(st_phon_pos);
112  }
113 
114  node.append_attribute("pron").set_value(lexlkp.pron.c_str());
115  if (lexlkp.lts) {
116  node.append_attribute("lts").set_value("true");
117  } else if (!lexlkp.lts && lexlkp.altprons.size() > 1) {
118  altprons.clear();
119  for (i = 0; i < lexlkp.altprons.size(); i++) {
120  if (i) altprons = altprons + ", ";
121  altprons = altprons + lexlkp.altprons[i];
122  }
123  node.append_attribute("altprons").set_value(altprons.c_str());
124  }
125  }
126  }
127  return true;
128 }

130 void TxpPronounce::AppendPron(const char* entry,
131  const std::string &word,
132  TxpLexiconLkp* lexlkp) {
133  bool found = false;
134  if (!lexlkp->pron.empty()) lexlkp->pron += " ";
135  if (entry) {
136  found = lex_.GetPron(word, std::string(entry), lexlkp);
137  if (!found) {
138  found = lex_.GetPron(word, std::string(""), lexlkp);
139  }
140  } else {
141  found = lex_.GetPron(word, std::string(""), lexlkp);
142  }
143  if (!found)
144  found = lts_.GetPron(word, lexlkp);
145 }
