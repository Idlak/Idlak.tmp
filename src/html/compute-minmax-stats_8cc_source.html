25 bool AccMinmaxStatsWrapper(std::string utt,
26  const MatrixBase<BaseFloat> &feats,
27  RandomAccessBaseFloatVectorReader *weights_reader,
28  Matrix<double> *cmvn_stats) {
29  if (!weights_reader->IsOpen()) {
30  AccMinmaxStats(feats, NULL, cmvn_stats);
31  return true;
32  } else {
33  if (!weights_reader->HasKey(utt)) {
34  KALDI_WARN << "No weights available for utterance " << utt;
35  return false;
36  }
37  const Vector<BaseFloat> &weights = weights_reader->Value(utt);
38  if (weights.Dim() != feats.NumRows()) {
39  KALDI_WARN << "Weights for utterance " << utt << " have wrong dimension "
40  << weights.Dim() << " vs. " << feats.NumRows();
41  return false;
42  }
43  AccMinmaxStats(feats, &weights, cmvn_stats);
44  return true;
45  }
46 }

51 int main(int argc, char *argv[]) {
52  try {

54  using kaldi::int32;
55 
56  const char *usage =
57  "Compute statistics for having features values between 0.01 and 0.99\n"
58  "If wspecifier provided: per-utterance by default, or per-speaker if\n"
59  "spk2utt option provided; if wxfilename: global\n"

64  bool binary = true;
65  po.Register("spk2utt", &spk2utt_rspecifier, "rspecifier for speaker to utterance-list map");
66  po.Register("binary", &binary, "write in binary mode (applies only to global CMN/CVN)");
67 
68  po.Read(argc, argv);
69 
70  if (po.NumArgs() != 2) {
71  po.PrintUsage();
72  exit(1);
73  }
74 
75  int32 num_done = 0, num_err = 0;
76  std::string rspecifier = po.GetArg(1);
77  std::string wspecifier_or_wxfilename = po.GetArg(2);
78 
79  RandomAccessBaseFloatVectorReader weights_reader(weights_rspecifier);
80 
81  if (ClassifyWspecifier(wspecifier_or_wxfilename, NULL, NULL, NULL)
82  != kNoWspecifier) { // writing to a Table: per-speaker or per-utt CMN/CVN.
83  std::string wspecifier = wspecifier_or_wxfilename;
84 
85  DoubleMatrixWriter writer(wspecifier);
86 
87  if (spk2utt_rspecifier != "") {
88  SequentialTokenVectorReader spk2utt_reader(spk2utt_rspecifier);
89  RandomAccessBaseFloatMatrixReader feat_reader(rspecifier);
90 
91  for (; !spk2utt_reader.Done(); spk2utt_reader.Next()) {
92  std::string spk = spk2utt_reader.Key();
93  const std::vector<std::string> &uttlist = spk2utt_reader.Value();
94  bool is_init = false;
95  Matrix<double> stats;
96  for (size_t i = 0; i < uttlist.size(); i++) {
97  std::string utt = uttlist[i];
98  if (!feat_reader.HasKey(utt)) {
99  KALDI_WARN << "Did not find features for utterance " << utt;
100  num_err++;
101  continue;
102  }
103  const Matrix<BaseFloat> &feats = feat_reader.Value(utt);
104  if (!is_init) {
105  InitMinmaxStats(feats.NumCols(), &stats);
106  is_init = true;
107  }
108  if (!AccMinmaxStatsWrapper(utt, feats, &weights_reader, &stats)) {
109  num_err++;
110  } else {
111  num_done++;
112  }
113  }
114  if (stats.NumRows() == 0) {
115  KALDI_WARN << "No stats accumulated for speaker " << spk;
116  } else {
117  writer.Write(spk, stats);
118  }
119  }
120  } else { // per-utterance normalization
121  SequentialBaseFloatMatrixReader feat_reader(rspecifier);
122 
123  for (; !feat_reader.Done(); feat_reader.Next()) {
124  std::string utt = feat_reader.Key();
125  Matrix<double> stats;
126  const Matrix<BaseFloat> &feats = feat_reader.Value();
127  InitMinmaxStats(feats.NumCols(), &stats);
128 
129  if (!AccMinmaxStatsWrapper(utt, feats, &weights_reader, &stats)) {
130  num_err++;
131  continue;
132  }
133  writer.Write(feat_reader.Key(), stats);
134  num_done++;
135  }
136  }
137  } else { // accumulate global stats
138  if (spk2utt_rspecifier != "")
139  KALDI_ERR << "--spk2utt option not compatible with wxfilename as output "
140  << "(did you forget ark:?)";
141  std::string wxfilename = wspecifier_or_wxfilename;
142  bool is_init = false;
143  Matrix<double> stats;
144  SequentialBaseFloatMatrixReader feat_reader(rspecifier);
145  for (; !feat_reader.Done(); feat_reader.Next()) {
146  std::string utt = feat_reader.Key();
147  const Matrix<BaseFloat> &feats = feat_reader.Value();
148  if (!is_init) {
149  InitMinmaxStats(feats.NumCols(), &stats);
150  is_init = true;
151  }
152  if (!AccMinmaxStatsWrapper(utt, feats, &weights_reader, &stats)) {
153  num_err++;
154  } else {
155  num_done++;
156  }
157  }
158  Matrix<float> fstats(stats);
159  Output ko(wxfilename, binary);
160  fstats.Write(ko.Stream(), binary);
161  num_done++;
162  }
163  KALDI_LOG << "Done accumulating Minmax stats for " << num_done
164  << " utterances; " << num_err << " had errors.";
165  return 0;
166  } catch(const std::exception &e) {
167  std::cerr << e.what();
168  return -1;
169  }
170 }
