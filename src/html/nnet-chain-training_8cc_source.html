28  const fst::StdVectorFst &den_fst,
29  Nnet *nnet):

32  nnet_(nnet),

34  opts_.nnet_config.compiler_config),
35  num_minibatches_processed_(0),

44  const int32 num_updatable = NumUpdatableComponents(*delta_nnet_);

46  num_max_change_global_applied_ = 0;
47

49  bool binary;

51  Input ki(opts.nnet_config.read_cache, &binary);
52  compiler_.ReadCache(ki.Stream(), binary);

62 void NnetChainTrainer::Train(const NnetChainExample &chain_eg) {
63  bool need_model_derivative = true;
64  const NnetTrainerOptions &nnet_config = opts_.nnet_config;
65  bool use_xent_regularization = (opts_.chain_config.xent_regularize != 0.0);
66  ComputationRequest request;
67  GetChainComputationRequest(*nnet_, chain_eg, need_model_derivative,
68  nnet_config.store_component_stats,
69  use_xent_regularization, need_model_derivative,
70  &request);
71  std::shared_ptr<const NnetComputation> computation = compiler_.Compile(request);
72 
73  if (nnet_config.backstitch_training_scale > 0.0 && num_minibatches_processed_
74  % nnet_config.backstitch_training_interval ==
75  srand_seed_ % nnet_config.backstitch_training_interval) {
76  // backstitch training is incompatible with momentum > 0
77  KALDI_ASSERT(nnet_config.momentum == 0.0);
78  FreezeNaturalGradient(true, delta_nnet_);
79  bool is_backstitch_step1 = true;
80  srand(srand_seed_ + num_minibatches_processed_);
81  ResetGenerators(nnet_);
82  TrainInternalBackstitch(chain_eg, *computation, is_backstitch_step1);
83  FreezeNaturalGradient(false, delta_nnet_); // un-freeze natural gradient
84  is_backstitch_step1 = false;
85  srand(srand_seed_ + num_minibatches_processed_);
86  ResetGenerators(nnet_);
87  TrainInternalBackstitch(chain_eg, *computation, is_backstitch_step1);
88  } else { // conventional training
89  TrainInternal(chain_eg, *computation);
90  }
91  if (num_minibatches_processed_ == 0) {
92  ConsolidateMemory(nnet_);
93  ConsolidateMemory(delta_nnet_);
94  }
95  num_minibatches_processed_++;
96 }

114 ChainTrainerMemoryHolder::ChainTrainerMemoryHolder(const Nnet &nnet,
115  int32 den_graph_states,
116  const NnetChainExample &eg) {
117 
118  std::vector<NnetChainSupervision>::const_iterator iter = eg.outputs.begin(),
119  end = eg.outputs.end();
120 
121  int32 max_rows = 0,
122  max_cols = 0;
123 
124  size_t max_frames_per_sequence = 0,
125  max_sequence_size = 0,
126  max_alpha_matrix_size = 0;
127 
128  for (; iter != end; ++iter) {
129  // there will normally be just one of these things; we'll normally loop once.
130  const NnetChainSupervision &sup = *iter;
131 
132  int32 output_rows = sup.supervision.num_sequences * sup.supervision.frames_per_sequence;
133  int32 output_cols = nnet.OutputDim("output");
134 
135  size_t curr_frames_per_sequence = output_rows / sup.supervision.num_sequences + 1;
136  size_t den_graph_size = den_graph_states + 1;
137  size_t curr_sequence_size = den_graph_size * sup.supervision.num_sequences;
138  size_t curr_alpha_matrix_size = curr_frames_per_sequence * curr_sequence_size;
139 
140  if (curr_alpha_matrix_size > max_alpha_matrix_size) {
141  max_alpha_matrix_size = curr_alpha_matrix_size;
142  max_frames_per_sequence = curr_frames_per_sequence;
143  max_sequence_size = curr_sequence_size;
144  }
145 
146  size_t matrix_size = output_rows * output_cols;
147  if (matrix_size > (max_rows * max_cols)) {
148  max_rows = output_rows;
149  max_cols = output_cols;
150  }
151  }
152 
153  // the sequence of resizes is in a specific order (bigger to smaller)
154  // so that the cudaMalloc won't trash the memory it has already
155  // alloc'd in the previous iterations
156  alpha_.Resize(max_frames_per_sequence,
157  max_sequence_size,
158  kUndefined);
159 
160 
161  nnet_output_deriv_.Resize(max_rows, max_cols, kUndefined);
162  // note: the same block of memory can be used for xent_output_deriv_ as is
163  // used for exp_nnet_output_transposed_ in chain-training.cc.
164  xent_output_deriv_.Resize(max_rows, max_cols,
165  kUndefined, kStrideEqualNumCols);
166 
167  beta_.Resize(2, max_sequence_size, kUndefined);
168 }

170 void NnetChainTrainer::TrainInternal(const NnetChainExample &eg,
171  const NnetComputation &computation) {
172  const NnetTrainerOptions &nnet_config = opts_.nnet_config;
173  // note: because we give the 1st arg (nnet_) as a pointer to the
174  // constructor of 'computer', it will use that copy of the nnet to
175  // store stats.
176  NnetComputer computer(nnet_config.compute_config, computation,
177  nnet_, delta_nnet_);
178 
179  // reserve the memory needed in ProcessOutputs (before memory gets fragmented
180  // by the call to computer.Run().
181  ChainTrainerMemoryHolder *memory_holder =
182  new ChainTrainerMemoryHolder(*nnet_, den_graph_.NumStates(), eg);
183 
184  // give the inputs to the computer object.
185  computer.AcceptInputs(*nnet_, eg.inputs);
186  computer.Run();
187 
188  // 'this->ProcessOutputs()' is going to need the same sizes as are stored in
189  // 'memory_holder'.
190  delete memory_holder;
191 
192  // Probably could be merged in a single call PreallocateChainTrainerMemory(*nnet_, eg) ?
193  this->ProcessOutputs(false, eg, &computer);
194  computer.Run();
195 
196  // If relevant, add in the part of the gradient that comes from L2
197  // regularization.
198  ApplyL2Regularization(*nnet_,
199  GetNumNvalues(eg.inputs, false) *
200  nnet_config.l2_regularize_factor,
201  delta_nnet_);
202 
203  // Updates the parameters of nnet
204  bool success = UpdateNnetWithMaxChange(*delta_nnet_,
205  nnet_config.max_param_change, 1.0, 1.0 - nnet_config.momentum, nnet_,
206  &num_max_change_per_component_applied_, &num_max_change_global_applied_);
207 
208  // Scale down the batchnorm stats (keeps them fresh... this affects what
209  // happens when we use the model with batchnorm test-mode set).
210  ScaleBatchnormStats(nnet_config.batchnorm_stats_scale, nnet_);
211 
212  // The following will only do something if we have a LinearComponent
213  // or AffineComponent with orthonormal-constraint set to a nonzero value.
214  ConstrainOrthonormal(nnet_);
215 
216  // Scale delta_nnet
217  if (success)
218  ScaleNnet(nnet_config.momentum, delta_nnet_);
219  else
220  ScaleNnet(0.0, delta_nnet_);
221 }

223 void NnetChainTrainer::TrainInternalBackstitch(const NnetChainExample &eg,
224  const NnetComputation &computation,
225  bool is_backstitch_step1) {
226  const NnetTrainerOptions &nnet_config = opts_.nnet_config;
227  // note: because we give the 1st arg (nnet_) as a pointer to the
228  // constructor of 'computer', it will use that copy of the nnet to
229  // store stats.
230  NnetComputer computer(nnet_config.compute_config, computation,
231  nnet_, delta_nnet_);
232  // give the inputs to the computer object.
233  computer.AcceptInputs(*nnet_, eg.inputs);
234  computer.Run();
235 
236  bool is_backstitch_step2 = !is_backstitch_step1;
237  this->ProcessOutputs(is_backstitch_step2, eg, &computer);
238  computer.Run();
239 
240  BaseFloat max_change_scale, scale_adding;
241  if (is_backstitch_step1) {
242  // max-change is scaled by backstitch_training_scale;
243  // delta_nnet is scaled by -backstitch_training_scale when added to nnet;
244  max_change_scale = nnet_config.backstitch_training_scale;
245  scale_adding = -nnet_config.backstitch_training_scale;
246  } else {
247  // max-change is scaled by 1 + backstitch_training_scale;
248  // delta_nnet is scaled by 1 + backstitch_training_scale when added to nnet;
249  max_change_scale = 1.0 + nnet_config.backstitch_training_scale;
250  scale_adding = 1.0 + nnet_config.backstitch_training_scale;
251  // If relevant, add in the part of the gradient that comes from L2
252  // regularization. It may not be optimally inefficient to do it on both
253  // passes of the backstitch, like we do here, but it probably minimizes
254  // any harmful interactions with the max-change.
255  ApplyL2Regularization(*nnet_,
256  1.0 / scale_adding * GetNumNvalues(eg.inputs, false) *
257  nnet_config.l2_regularize_factor, delta_nnet_);
258  }
259 
260  // Updates the parameters of nnet
261  UpdateNnetWithMaxChange(*delta_nnet_,
262  nnet_config.max_param_change, max_change_scale, scale_adding, nnet_,
263  &num_max_change_per_component_applied_, &num_max_change_global_applied_);
264 
265  if (is_backstitch_step1) {
266  // The following will only do something if we have a LinearComponent or
267  // AffineComponent with orthonormal-constraint set to a nonzero value. We
268  // choose to do this only on the 1st backstitch step, for efficiency.
269  ConstrainOrthonormal(nnet_);
270  }
271 
272  if (!is_backstitch_step1) {
273  // Scale down the batchnorm stats (keeps them fresh... this affects what
274  // happens when we use the model with batchnorm test-mode set). Do this
275  // after backstitch step 2 so that the stats are scaled down before we start
276  // the next minibatch.
277  ScaleBatchnormStats(nnet_config.batchnorm_stats_scale, nnet_);
278  }
279 
280  ScaleNnet(0.0, delta_nnet_);
281 }

283 void NnetChainTrainer::ProcessOutputs(bool is_backstitch_step2,
284  const NnetChainExample &eg,
285  NnetComputer *computer) {
286  // normally the eg will have just one output named 'output', but
287  // we don't assume this.
288  // In backstitch training, the output-name with the "_backstitch" suffix is
289  // the one computed after the first, backward step of backstitch.
290  const std::string suffix = (is_backstitch_step2 ? "_backstitch" : "");
291  std::vector<NnetChainSupervision>::const_iterator iter = eg.outputs.begin(),
292  end = eg.outputs.end();
293  for (; iter != end; ++iter) {
294  const NnetChainSupervision &sup = *iter;
295  int32 node_index = nnet_->GetNodeIndex(sup.name);
296  if (node_index < 0 ||
297  !nnet_->IsOutputNode(node_index))
298  KALDI_ERR << "Network has no output named " << sup.name;
299 
300  const CuMatrixBase<BaseFloat> &nnet_output = computer->GetOutput(sup.name);
301  CuMatrix<BaseFloat> nnet_output_deriv(nnet_output.NumRows(),
302  nnet_output.NumCols(),
303  kUndefined);
304 
305  bool use_xent = (opts_.chain_config.xent_regularize != 0.0);
306  std::string xent_name = sup.name + "-xent"; // typically "output-xent".
307  CuMatrix<BaseFloat> xent_deriv;
308 
309  BaseFloat tot_objf, tot_l2_term, tot_weight;
310 
311  ComputeChainObjfAndDeriv(opts_.chain_config, den_graph_,
312  sup.supervision, nnet_output,
313  &tot_objf, &tot_l2_term, &tot_weight,
314  &nnet_output_deriv,
315  (use_xent ? &xent_deriv : NULL));
316 
317  if (use_xent) {
318  // this block computes the cross-entropy objective.
319  const CuMatrixBase<BaseFloat> &xent_output = computer->GetOutput(
320  xent_name);
321  // at this point, xent_deriv is posteriors derived from the numerator
322  // computation. note, xent_objf has a factor of '.supervision.weight'
323  BaseFloat xent_objf = TraceMatMat(xent_output, xent_deriv, kTrans);
324  objf_info_[xent_name + suffix].UpdateStats(xent_name + suffix,
325  opts_.nnet_config.print_interval,
326  num_minibatches_processed_,
327  tot_weight, xent_objf);
328  }
329 
330  if (opts_.apply_deriv_weights && sup.deriv_weights.Dim() != 0) {
331  CuVector<BaseFloat> cu_deriv_weights(sup.deriv_weights);
332  nnet_output_deriv.MulRowsVec(cu_deriv_weights);
333  if (use_xent)
334  xent_deriv.MulRowsVec(cu_deriv_weights);
335  }
336 
337  computer->AcceptInput(sup.name, &nnet_output_deriv);
338 
339  objf_info_[sup.name + suffix].UpdateStats(sup.name + suffix,
340  opts_.nnet_config.print_interval,
341  num_minibatches_processed_,
342  tot_weight, tot_objf, tot_l2_term);
343 
344  if (use_xent) {
345  xent_deriv.Scale(opts_.chain_config.xent_regularize);
346  computer->AcceptInput(xent_name, &xent_deriv);
347  }
348  }
349 }

365 void NnetChainTrainer::PrintMaxChangeStats() const {
366  KALDI_ASSERT(delta_nnet_ != NULL);
367  const NnetTrainerOptions &nnet_config = opts_.nnet_config;
368  int32 i = 0;
369  for (int32 c = 0; c < delta_nnet_->NumComponents(); c++) {
370  Component *comp = delta_nnet_->GetComponent(c);
371  if (comp->Properties() & kUpdatableComponent) {
372  UpdatableComponent *uc = dynamic_cast<UpdatableComponent*>(comp);
373  if (uc == NULL)
374  KALDI_ERR << "Updatable component does not inherit from class "
375  << "UpdatableComponent; change this code.";
376  if (num_max_change_per_component_applied_[i] > 0)
377  KALDI_LOG << "For " << delta_nnet_->GetComponentName(c)
378  << ", per-component max-change was enforced "
379  << (100.0 * num_max_change_per_component_applied_[i]) /
380  (num_minibatches_processed_ *
381  (nnet_config.backstitch_training_scale == 0.0 ? 1.0 :
382  1.0 + 1.0 / nnet_config.backstitch_training_interval))
383  << " \% of the time.";
384  i++;
385  }
386  }
387  if (num_max_change_global_applied_ > 0)
388  KALDI_LOG << "The global max-change was enforced "
389  << (100.0 * num_max_change_global_applied_) /
390  (num_minibatches_processed_ *
391  (nnet_config.backstitch_training_scale == 0.0 ? 1.0 :
392  1.0 + 1.0 / nnet_config.backstitch_training_interval))
393  << " \% of the time.";
394 }
