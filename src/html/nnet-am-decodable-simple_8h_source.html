56  NnetSimpleComputationOptions():

59  extra_left_context_initial(-1),

68  void Register(OptionsItf *opts) {
69  opts->Register("extra-left-context", &extra_left_context,
70  "Number of frames of additional left-context to add on top "
71  "of the neural net's inherent left context (may be useful in "
72  "recurrent setups");

74  "Number of frames of additional right-context to add on top "

78  "If >= 0, overrides the --extra-left-context value at the "

81  "If >= 0, overrides the --extra-right-context value at the "
82  "end of an utterance.");
83  opts->Register("frame-subsampling-factor", &frame_subsampling_factor,
84  "Required if the frame-rate of the output (e.g. in 'chain' "
85  "models) is less than the frame-rate of the original "

87  opts->Register("acoustic-scale", &acoustic_scale,
88  "Scaling factor for acoustic log-likelihoods (caution: is a no-op "

92  "by the neural net. Measured before any subsampling, if the "

95  opts->Register("debug-computation", &debug_computation, "If true, turn on "

158  // returns the number of frames of likelihoods. The same as feats_.NumRows()
159  // in the normal case (but may be less if opts_.frame_subsampling_factor !=
160  // 1).
161  inline int32 NumFrames() const { return num_subsampled_frames_; }
162 
163  inline int32 OutputDim() const { return output_dim_; }
164 
165  // Gets the output for a particular frame, with 0 <= frame < NumFrames().
166  // 'output' must be correctly sized (with dimension OutputDim()).
167  void GetOutputForFrame(int32 frame, VectorBase<BaseFloat> *output);
168 
169  // Gets the output for a particular frame and pdf_id, with
170  // 0 <= subsampled_frame < NumFrames(),
171  // and 0 <= pdf_id < OutputDim().
172  inline BaseFloat GetOutput(int32 subsampled_frame, int32 pdf_id) {
173  if (subsampled_frame < current_log_post_subsampled_offset_ ||
174  subsampled_frame >= current_log_post_subsampled_offset_ +
175  current_log_post_.NumRows())
176  EnsureFrameIsComputed(subsampled_frame);
177  return current_log_post_(subsampled_frame -
178  current_log_post_subsampled_offset_,
179  pdf_id);
180  }
181  private:
182  KALDI_DISALLOW_COPY_AND_ASSIGN(DecodableNnetSimple);
183 
184  // This call is made to ensure that we have the log-probs for this frame
185  // cached in current_log_post_.
186  void EnsureFrameIsComputed(int32 subsampled_frame);
187 
188  // This function does the actual nnet computation; it is called from
189  // EnsureFrameIsComputed. Any padding at file start/end is done by
190  // the caller of this function (so the input should exceed the output
191  // by a suitable amount of context). It puts its output in current_log_post_.
192  void DoNnetComputation(int32 input_t_start,
193  const MatrixBase<BaseFloat> &input_feats,
194  const VectorBase<BaseFloat> &ivector,
195  int32 output_t_start,
196  int32 num_subsampled_frames);
197 
198  // Gets the iVector that will be used for this chunk of frames, if we are
199  // using iVectors (else does nothing). note: the num_output_frames is
200  // interpreted as the number of t value, which in the subsampled case is not
201  // the same as the number of subsampled frames (it would be larger by
202  // opts_.frame_subsampling_factor).
203  void GetCurrentIvector(int32 output_t_start,
204  int32 num_output_frames,
205  Vector<BaseFloat> *ivector);
206 
207  // called from constructor
208  void CheckAndFixConfigs();
209 
210  // returns dimension of the provided iVectors if supplied, or 0 otherwise.
211  int32 GetIvectorDim() const;
212 
213  NnetSimpleComputationOptions opts_;
214  const Nnet &nnet_;
215  int32 nnet_left_context_;
216  int32 nnet_right_context_;
217  int32 output_dim_;
218  // the log priors (or the empty vector if the priors are not set in the model)
219  CuVector<BaseFloat> log_priors_;
220  const MatrixBase<BaseFloat> &feats_;
221  // note: num_subsampled_frames_ will equal feats_.NumRows() in the normal case
222  // when opts_.frame_subsampling_factor == 1.
223  int32 num_subsampled_frames_;
224 
225  // ivector_ is the iVector if we're using iVectors that are estimated in batch
226  // mode.
227  const VectorBase<BaseFloat> *ivector_;
228 
229  // online_ivector_feats_ is the iVectors if we're using online-estimated ones.
230  const MatrixBase<BaseFloat> *online_ivector_feats_;
231  // online_ivector_period_ helps us interpret online_ivector_feats_; it's the
232  // number of frames the rows of ivector_feats are separated by.
233  int32 online_ivector_period_;
234 
235  // a reference to a compiler passed in via the constructor, which may be
236  // declared at the top level of the program so that we don't have to recompile
237  // computations each time.
238  CachingOptimizingCompiler &compiler_;
239 
240  // The current log-posteriors that we got from the last time we
241  // ran the computation.
242  Matrix<BaseFloat> current_log_post_;
243  // The time-offset of the current log-posteriors. Note: if
244  // opts_.frame_subsampling_factor > 1, this will be measured in subsampled
245  // frames.
246  int32 current_log_post_subsampled_offset_;
247 };
