59 template<typename Real>
60 void CuRand<Real>::RandUniform(CuMatrixBase<Real> *tgt) {
61 #if HAVE_CUDA == 1
62  if (CuDevice::Instantiate().Enabled()) {
63  CuTimer tim;
64  // Better use 'tmp' matrix, 'tgt' can be a window into a larger matrix,
65  // so we should not use it to generate random numbers over whole stride.
66  // Use the option kStrideEqualNumCols to ensure consistency
67  // (because when memory is nearly exhausted, the stride of CudaMallocPitch
68  // may vary).
69  CuMatrix<Real> tmp(tgt->NumRows(), tgt->NumCols(), kUndefined,
70  kStrideEqualNumCols);
71  size_t s = static_cast<size_t>(tmp.NumRows()) * static_cast<size_t>(tmp.Stride());
72  CURAND_SAFE_CALL(curandGenerateUniformWrap(gen_, tmp.Data(), s));
73  tgt->CopyFromMat(tmp);
74  CuDevice::Instantiate().AccuProfile(__func__, tim);
75  } else
76 #endif
77  {
78  tgt->Mat().SetRandUniform();
79  }
80 }

112 template<typename Real>
113 void CuRand<Real>::RandGaussian(CuMatrixBase<Real> *tgt) {
114 #if HAVE_CUDA == 1
115  if (CuDevice::Instantiate().Enabled()) {
116  CuTimer tim;
117  // Better use 'tmp' matrix, 'tgt' can be a window into a larger matrix,
118  // so we should not use it to generate random numbers over whole stride.
119  // Also, we ensure to have 'even' number of elements for calling 'curand'
120  // by possibly adding one column. Even number of elements is required by
121  // curandGenerateUniform(), curandGenerateUniformDouble().
122  // Use the option kStrideEqualNumCols to ensure consistency
123  // (because when memory is nearly exhausted, the stride of CudaMallocPitch
124  // may vary).
125  MatrixIndexT num_cols_even = tgt->NumCols() + (tgt->NumCols() % 2); // + 0 or 1,
126  CuMatrix<Real> tmp(tgt->NumRows(), num_cols_even, kUndefined,
127  kStrideEqualNumCols);
128  CURAND_SAFE_CALL(curandGenerateNormalWrap(gen_, tmp.Data(), tmp.NumRows()*tmp.Stride()));
129  tgt->CopyFromMat(tmp.ColRange(0,tgt->NumCols()));
130  CuDevice::Instantiate().AccuProfile(__func__, tim);
131  } else
132 #endif
133  {
134  tgt->Mat().SetRandn();
135  }
136 }

138 template<typename Real>
139 void CuRand<Real>::RandGaussian(CuMatrix<Real> *tgt) {
140 #if HAVE_CUDA == 1
141  if (CuDevice::Instantiate().Enabled()) {
142  CuTimer tim;
143  // Here we don't need to use 'tmp' matrix, if the number of elements is even,
144  MatrixIndexT num_elements = tgt->NumRows() * tgt->Stride();
145  if (0 == (num_elements % 2)) {
146  CURAND_SAFE_CALL(curandGenerateNormalWrap(gen_, tgt->Data(), num_elements));
147  } else {
148  // We use 'tmp' matrix with one column added, this guarantees an even
149  // number of elements. Use the option kStrideEqualNumCols to ensure
150  // consistency (because when memory is nearly exhausted, the stride of
151  // CudaMallocPitch may vary).
152  MatrixIndexT num_cols_even = tgt->NumCols() + (tgt->NumCols() % 2); // + 0 or 1,
153  CuMatrix<Real> tmp(tgt->NumRows(), num_cols_even, kUndefined,
154  kStrideEqualNumCols);
155  CURAND_SAFE_CALL(curandGenerateNormalWrap(gen_, tmp.Data(),
156  tmp.NumRows() * tmp.Stride()));
157  tgt->CopyFromMat(tmp.ColRange(0,tgt->NumCols()));
158  }
159  CuDevice::Instantiate().AccuProfile(__func__, tim);
160  } else
161 #endif
162  {
163  tgt->Mat().SetRandn();
164  }
165 }

167 template<typename Real>
168 void CuRand<Real>::RandGaussian(CuVectorBase<Real> *tgt) {
169 #if HAVE_CUDA == 1
170  if (CuDevice::Instantiate().Enabled()) {
171  CuTimer tim;
172  // To ensure 'even' number of elements, we use 'tmp' vector of even length.
173  // Even number of elements is required by 'curand' functions:
174  // curandGenerateUniform(), curandGenerateUniformDouble().
175  MatrixIndexT num_elements = tgt->Dim();
176  if (0 == (num_elements % 2)) {
177  CURAND_SAFE_CALL(curandGenerateNormalWrap(gen_, tgt->Data(), tgt->Dim()));
178  } else {
179  MatrixIndexT dim_even = tgt->Dim() + (tgt->Dim() % 2); // + 0 or 1,
180  CuVector<Real> tmp(dim_even, kUndefined);
181  CURAND_SAFE_CALL(curandGenerateNormalWrap(gen_, tmp.Data(), tmp.Dim()));
182  tgt->CopyFromVec(tmp.Range(0,tgt->Dim()));
183  }
184  CuDevice::Instantiate().AccuProfile(__func__, tim);
185  } else
186 #endif
187  {
188  tgt->Vec().SetRandn();
189  }
190 }

193 template<typename Real>
194 void CuRand<Real>::BinarizeProbs(const CuMatrix<Real> &probs, CuMatrix<Real> *states) {
195  CuMatrix<Real> tmp(probs.NumRows(), probs.NumCols());
196  this->RandUniform(&tmp); // [0..1]
197  tmp.Scale(-1.0); // [-1..0]
198  tmp.AddMat(1.0, probs); // [-1..+1]
199  states->Heaviside(tmp); // negative
200 }
