145 // Note on KALDI_ASSERT and KALDI_PARANOID_ASSERT
146 // The original (simple) version of the code was this
147 //
148 // #define KALDI_ASSERT(cond) if (!(cond))
149 // kaldi::KaldiAssertFailure_(__func__, __FILE__, __LINE__, #cond);
150 //
151 // That worked well, but we were concerned that it
152 // could potentially cause a performance issue due to failed branch
153 // prediction (best practice is to have the if branch be the commonly
154 // taken one).
155 // Therefore, we decided to move the call into the else{} branch.
156 // A single block {} around if /else does not work, because it causes
157 // syntax error (unmatched else block) in the following code:
158 //
159 // if (condition)
160 // KALDI_ASSERT(condition2);
161 // else
162 // SomethingElse();
163 //
164 // do {} while(0) -- note there is no semicolon at the end! --- works nicely
165 // and compilers will be able to optimize the loop away (as the condition
166 // is always false).
167 #ifndef NDEBUG
168 #define KALDI_ASSERT(cond) do { if (cond) (void)0; else \
169  ::kaldi::KaldiAssertFailure_(__func__, __FILE__, __LINE__, #cond); } while(0)
170 #else
171 #define KALDI_ASSERT(cond) (void)0
172 #endif
173 // Also see KALDI_COMPILE_TIME_ASSERT, defined in base/kaldi-utils.h,
174 // and KALDI_ASSERT_IS_INTEGER_TYPE and KALDI_ASSERT_IS_FLOATING_TYPE,
175 // also defined there.
176 // some more expensive asserts only checked if this defined
177 #ifdef KALDI_PARANOID
178 #define KALDI_PARANOID_ASSERT(cond) do { if (cond) (void)0; else \
179  ::kaldi::KaldiAssertFailure_(__func__, __FILE__, __LINE__, #cond); } while(0)
180 #else
181 #define KALDI_PARANOID_ASSERT(cond) (void)0
182 #endif
