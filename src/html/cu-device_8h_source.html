84  // We provide functions Malloc(), MallocPitch() and Free() which replace
85  // cudaMalloc(), cudaMallocPitch() and cudaFree(). Their function is to cache
86  // the results of previous allocations to avoid the very large overhead that
87  // CUDA's allocation seems to give for some setups.
88  inline void* Malloc(size_t size) {
89  return multi_threaded_ ? g_cuda_allocator.MallocLocking(size) :
90  g_cuda_allocator.Malloc(size);
91  }
92 
93  inline void* MallocPitch(size_t row_bytes, size_t num_rows, size_t *pitch) {
94  if (multi_threaded_) {
95  return g_cuda_allocator.MallocPitchLocking(row_bytes, num_rows, pitch);
96  } else if (debug_stride_mode_) {
97  // The pitch bucket size is hardware dependent.
98  // It is 512 on K40c with CUDA 7.5
99  // "% 8" ensures that any 8 adjacent allocations have different pitches
100  // if their original pitches are same in the normal mode.
101  return g_cuda_allocator.MallocPitch(
102  row_bytes + 512 * RandInt(0, 4), num_rows,
103  pitch);
104  } else {
105  return g_cuda_allocator.MallocPitch(row_bytes, num_rows, pitch);
106  }
107  }
108 
109  inline void Free(void *ptr) {
110  if (multi_threaded_) g_cuda_allocator.FreeLocking(ptr);
111  else g_cuda_allocator.Free(ptr);
112  }
