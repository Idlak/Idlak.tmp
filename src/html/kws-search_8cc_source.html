34 // encode ilabel, olabel pair as a single 64bit (output) symbol
35 uint64 EncodeLabel(StateId ilabel, StateId olabel) {
36  return (static_cast<int64>(olabel) << 32) + static_cast<int64>(ilabel);

46 // this is a mapper adapter that helps converting
47 // between the StdArc FST (i.e. tropical semiring FST)
48 // to the KwsLexicographic FST. Structure will be kept,
49 // the weights converted/recomputed
50 class VectorFstToKwsLexicographicFstMapper {
51  public:
52  typedef fst::StdArc FromArc;
53  typedef FromArc::Weight FromWeight;
54  typedef KwsLexicographicArc ToArc;
55  typedef KwsLexicographicWeight ToWeight;
56 
57  VectorFstToKwsLexicographicFstMapper() {}
58 
59  ToArc operator()(const FromArc &arc) const {
60  return ToArc(arc.ilabel,
61  arc.olabel,
62  (arc.weight == FromWeight::Zero() ?
63  ToWeight::Zero() :
64  ToWeight(arc.weight.Value(),
65  StdLStdWeight::One())),
66  arc.nextstate);
67  }
68 
69  fst::MapFinalAction FinalAction() const {
70  return fst::MAP_NO_SUPERFINAL;
71  }
72 
73  fst::MapSymbolsAction InputSymbolsAction() const {
74  return fst::MAP_COPY_SYMBOLS;
75  }
76 
77  fst::MapSymbolsAction OutputSymbolsAction() const {
78  return fst::MAP_COPY_SYMBOLS;
79  }

84 struct ActivePath {
85  std::vector<KwsLexicographicArc::Label> path;
86  KwsLexicographicArc::Weight weight;
87  KwsLexicographicArc::Label last;
88 };

90 bool GenerateActivePaths(const KwsLexicographicFst &proxy,
91  std::vector<ActivePath> *paths,
92  KwsLexicographicFst::StateId cur_state,
93  std::vector<KwsLexicographicArc::Label> cur_path,
94  KwsLexicographicArc::Weight cur_weight) {
95  for (fst::ArcIterator<KwsLexicographicFst> aiter(proxy, cur_state);
96  !aiter.Done(); aiter.Next()) {
97  const Arc &arc = aiter.Value();
98  Weight temp_weight = Times(arc.weight, cur_weight);
99 
100  cur_path.push_back(arc.ilabel);
101 
102  if ( arc.olabel != 0 ) {
103  ActivePath path;
104  path.path = cur_path;
105  path.weight = temp_weight;
106  path.last = arc.olabel;
107  paths->push_back(path);
108  } else {
109  GenerateActivePaths(proxy, paths,
110  arc.nextstate, cur_path, temp_weight);
111  }
112  cur_path.pop_back();
113  }
114 
115  return true;
116 }

119 typedef kaldi::TableWriter< kaldi::BasicVectorHolder<double> >
120  VectorOfDoublesWriter;
121 void OutputDetailedStatistics(const std::string &kwid,
122  const kaldi::KwsLexicographicFst &keyword,
123  const unordered_map<uint32, uint64> &label_decoder,
124  VectorOfDoublesWriter *output ) {
125  std::vector<kaldi::ActivePath> paths;
126 
127  if (keyword.Start() == fst::kNoStateId)
128  return;
129 
130  kaldi::GenerateActivePaths(keyword, &paths, keyword.Start(),
131  std::vector<kaldi::KwsLexicographicArc::Label>(),
132  kaldi::KwsLexicographicArc::Weight::One());
133 
134  for (int i = 0; i < paths.size(); ++i) {
135  std::vector<double> out;
136  double score;
137  int32 tbeg, tend, uid;
138 
139  uint64 osymbol = label_decoder.find(paths[i].last)->second;
140  uid = kaldi::DecodeLabelUid(osymbol);
141  tbeg = paths[i].weight.Value2().Value1().Value();
142  tend = paths[i].weight.Value2().Value2().Value();
143  score = paths[i].weight.Value1().Value();
144 
145  out.push_back(uid);
146  out.push_back(tbeg);
147  out.push_back(tend);
148  out.push_back(score);
149 
150  for (int j = 0; j < paths[i].path.size(); ++j) {
151  out.push_back(paths[i].path[j]);
152  }
153  output->Write(kwid, out);
154  }
155 }

158 int main(int argc, char *argv[]) {
159  try {
160  using namespace kaldi;
161  using namespace fst;
162  typedef kaldi::int32 int32;
163  typedef kaldi::uint32 uint32;
164  typedef kaldi::uint64 uint64;

166  const char *usage =
167  "Search the keywords over the index. This program can be executed\n"
168  "in parallel, either on the index side or the keywords side; we use\n"
169  "a script to combine the final search results. Note that the index\n"
170  "archive has a single key \"global\".\n\n"
171  "Search has one or two outputs. The first one is mandatory and will\n"
172  "contain the seach output, i.e. list of all found keyword instances\n"
173  "The file is in the following format:\n"
174  "kw_id utt_id beg_frame end_frame neg_logprob\n"
175  " e.g.: \n"
176  "KW105-0198 7 335 376 1.91254\n\n"
177  "The second parameter is optional and allows the user to gather more\n"
178  "statistics about the individual instances from the posting list.\n"
179  "Remember \"keyword\" is an FST and as such, there can be multiple\n"
180  "paths matching in the keyword and in the lattice index in that given\n"
181  "time period. The stats output will provide all matching paths\n"
182  "each with the appropriate score. \n"
183  "The format is as follows:\n"
184  "kw_id utt_id beg_frame end_frame neg_logprob 0 w_id1 w_id2 ... 0\n"
185  " e.g.: \n"
186  "KW105-0198 7 335 376 16.01254 0 5766 5659 0\n"
187  "\n"
188  "Usage: kws-search [options] <index-rspecifier> <keywords-rspecifier> "
189  "<results-wspecifier> [<stats_wspecifier>]\n"
190  " e.g.: kws-search ark:index.idx ark:keywords.fsts "
191  "ark:results ark:stats\n";
192 
193  ParseOptions po(usage);
194 
195  int32 n_best = -1;
196  int32 keyword_nbest = -1;
197  bool strict = true;
198  double negative_tolerance = -0.1;
199  double keyword_beam = -1;
200  int32 frame_subsampling_factor = 1;
201 
202  po.Register("frame-subsampling-factor", &frame_subsampling_factor,
203  "Frame subsampling factor. (Default value 1)");
204  po.Register("nbest", &n_best, "Return the best n hypotheses.");
205  po.Register("keyword-nbest", &keyword_nbest,
206  "Pick the best n keywords if the FST contains "
207  "multiple keywords.");
208  po.Register("strict", &strict, "Affects the return status of the program.");
209  po.Register("negative-tolerance", &negative_tolerance,
210  "The program will print a warning if we get negative score "
211  "smaller than this tolerance.");
212  po.Register("keyword-beam", &keyword_beam,
213  "Prune the FST with the given beam if the FST contains "
214  "multiple keywords.");
215 
216  if (n_best < 0 && n_best != -1) {
217  KALDI_ERR << "Bad number for nbest";
218  exit(1);
219  }
220  if (keyword_nbest < 0 && keyword_nbest != -1) {
221  KALDI_ERR << "Bad number for keyword-nbest";
222  exit(1);
223  }
224  if (keyword_beam < 0 && keyword_beam != -1) {
225  KALDI_ERR << "Bad number for keyword-beam";
226  exit(1);
227  }
228 
229  po.Read(argc, argv);
230 
231  if (po.NumArgs() < 3 || po.NumArgs() > 4) {
232  po.PrintUsage();
233  exit(1);
234  }
235 
236  std::string index_rspecifier = po.GetArg(1),
237  keyword_rspecifier = po.GetArg(2),
238  result_wspecifier = po.GetArg(3),
239  stats_wspecifier = po.GetOptArg(4);
240 
241  RandomAccessTableReader< VectorFstTplHolder<KwsLexicographicArc> >
242  index_reader(index_rspecifier);
243  SequentialTableReader<VectorFstHolder> keyword_reader(keyword_rspecifier);
244  VectorOfDoublesWriter result_writer(result_wspecifier);
245  VectorOfDoublesWriter stats_writer(stats_wspecifier);
246 
247 
248  // Index has key "global"
249  KwsLexicographicFst index = index_reader.Value("global");
250 
251  // First we have to remove the disambiguation symbols. But rather than
252  // removing them totally, we actually move them from input side to output
253  // side, making the output symbol a "combined" symbol of the disambiguation
254  // symbols and the utterance id's.
255  // Note that in Dogan and Murat's original paper, they simply remove the
256  // disambiguation symbol on the input symbol side, which will not allow us
257  // to do epsilon removal after composition with the keyword FST. They have
258  // to traverse the resulting FST.
259  int32 label_count = 1;
260  unordered_map<uint64, uint32> label_encoder;
261  unordered_map<uint32, uint64> label_decoder;
262  for (StateIterator<KwsLexicographicFst> siter(index);
263  !siter.Done(); siter.Next()) {
264  StateId state_id = siter.Value();
265  for (MutableArcIterator<KwsLexicographicFst>
266  aiter(&index, state_id); !aiter.Done(); aiter.Next()) {
267  KwsLexicographicArc arc = aiter.Value();
268  // Skip the non-final arcs
269  if (index.Final(arc.nextstate) == Weight::Zero())
270  continue;
271  // Encode the input and output label of the final arc, and this is the
272  // new output label for this arc; set the input label to <epsilon>
273  uint64 osymbol = EncodeLabel(arc.ilabel, arc.olabel);
274  arc.ilabel = 0;
275  if (label_encoder.find(osymbol) == label_encoder.end()) {
276  arc.olabel = label_count;
277  label_encoder[osymbol] = label_count;
278  label_decoder[label_count] = osymbol;
279  label_count++;
280  } else {
281  arc.olabel = label_encoder[osymbol];
282  }
283  aiter.SetValue(arc);
284  }
285  }
286  ArcSort(&index, fst::ILabelCompare<KwsLexicographicArc>());
287 
288  int32 n_done = 0;
289  int32 n_fail = 0;
290  for (; !keyword_reader.Done(); keyword_reader.Next()) {
291  std::string key = keyword_reader.Key();
292  VectorFst<StdArc> keyword = keyword_reader.Value();
293  keyword_reader.FreeCurrent();
294 
295  // Process the case where we have confusion for keywords
296  if (keyword_beam != -1) {
297  Prune(&keyword, keyword_beam);
298  }
299  if (keyword_nbest != -1) {
300  VectorFst<StdArc> tmp;
301  ShortestPath(keyword, &tmp, keyword_nbest, true, true);
302  keyword = tmp;
303  }
304 
305  KwsLexicographicFst keyword_fst;
306  KwsLexicographicFst result_fst;
307  Map(keyword, &keyword_fst, VectorFstToKwsLexicographicFstMapper());
308  Compose(keyword_fst, index, &result_fst);
309 
310  if (stats_wspecifier != "") {
311  KwsLexicographicFst matched_seq(result_fst);
312  OutputDetailedStatistics(key,
313  matched_seq,
314  label_decoder,
315  &stats_writer);
316  }
317 
318  Project(&result_fst, PROJECT_OUTPUT);
319  Minimize(&result_fst, (KwsLexicographicFst *) nullptr, kDelta, true);
320  ShortestPath(result_fst, &result_fst, n_best);
321  RmEpsilon(&result_fst);
322 
323  // No result found
324  if (result_fst.Start() == kNoStateId)
325  continue;
326 
327  // Got something here
328  double score;
329  int32 tbeg, tend, uid;
330  for (ArcIterator<KwsLexicographicFst>
331  aiter(result_fst, result_fst.Start()); !aiter.Done(); aiter.Next()) {
332  const KwsLexicographicArc &arc = aiter.Value();
333 
334  // We're expecting a two-state FST
335  if (result_fst.Final(arc.nextstate) != Weight::One()) {
336  KALDI_WARN << "The resulting FST does not have "
337  << "the expected structure for key " << key;
338  n_fail++;
339  continue;
340  }
341 
342  uint64 osymbol = label_decoder[arc.olabel];
343  uid = static_cast<int32>(DecodeLabelUid(osymbol));
344  tbeg = arc.weight.Value2().Value1().Value();
345  tend = arc.weight.Value2().Value2().Value();
346  score = arc.weight.Value1().Value();
347 
348  if (score < 0) {
349  if (score < negative_tolerance) {
350  KALDI_WARN << "Score out of expected range: " << score;
351  }
352  score = 0.0;
353  }
354  vector<double> result;
355  result.push_back(uid);
356  result.push_back(tbeg * frame_subsampling_factor);
357  result.push_back(tend * frame_subsampling_factor);
358  result.push_back(score);
359  result_writer.Write(key, result);
360  }
361 
362  n_done++;
363  }
364 
365  KALDI_LOG << "Done " << n_done << " keywords";
366  if (strict == true)
367  return (n_done != 0 ? 0 : 1);
368  else
369  return 0;
370  } catch(const std::exception &e) {
371  std::cerr << e.what();
372  return -1;
373  }
