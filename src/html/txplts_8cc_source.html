30 void TxpLts::StartElement(const char* name, const char** atts) {
31  std::string terminal;
32  std::string nonterminal;
33  LtsMap::iterator it;
34  if (!strcmp(name, "tree")) {
35  SetAttribute("ltr", atts, &ltr_);
36  SetAttribute("terminal", atts, &terminal);
37  SetAttribute("nonterminal", atts, &nonterminal);
38  TxpLtsTree tree;
39  if (atoi(nonterminal.c_str()) == 0) tree.root = 0;
40  else tree.root = atoi(terminal.c_str());
41  ltslkp_.insert(LtsItem(ltr_, tree));
42  } else if (!strcmp(name, "node")) {
43  TxpLtsNode node;
44  std::string pos;
45  std::string yes;
46  std::string no;
47  SetAttribute("pos", atts, &pos);
48  SetAttribute("posval", atts, &(node.posval));
49  SetAttribute("yes", atts, &yes);
50  SetAttribute("no", atts, &no);
51  SetAttribute("val", atts, &(node.val));
52  if (!pos.empty()) {
53  // non terminal node
54  node.yes = atoi(yes.c_str());
55  node.no = atoi(no.c_str());
56  node.pos = pos2int(pos);
57  }
58  it = ltslkp_.find(ltr_);
59  (it->second).nodes.push_back(node);
60  }
61 }

63 int TxpLts::GetPron(const std::string &word, TxpLexiconLkp* lkp) {
64  TxpUtf8 utf8;
65  int32 clen, pos = 0;
66  const char *p, *phone, *stress;
67  std::string ltr;
68  LtsMap::iterator it;
69  TxpLtsTree stress_tree;
70  bool stress_tree_present = true;
71 
72  // get stress lookup
73  it = ltslkp_.find(std::string("0"));
74  if (it == ltslkp_.end()) {
75  stress_tree_present = false;
76  KALDI_WARN << "No stress lookup tree: name='0'";
77  } else {
78  stress_tree = it->second;
79  }
80  // process each letter
81  p = word.c_str();
82  while (*p) {
83  clen = utf8.Clen(p);
84  ltr = std::string(p, clen);
85  it = ltslkp_.find(ltr);
86  if (it != ltslkp_.end()) {
87  phone = ApplyTree(it->second, word.c_str(), pos);
88  const char * orig = phone;
89  // If not a null result
90  if (strcmp("0", phone)) {
91  const char *sep = strchr(phone, '_');
92  if (!lkp->pron.empty()) lkp->pron += " ";
93  // Split the phone on "_" as LTS can return multiphone
94  while (sep) {
95  lkp->pron += std::string(phone, sep - phone);
96  lkp->pron += " ";
97  phone = sep + 1;
98  sep = strchr(phone, '_');
99  }
100 
101  // Check is syllabic and if so get stress
102  if (phone[strlen(phone) - 1] == '0' && stress_tree_present) {
103  stress = ApplyTree(stress_tree, word.c_str(), pos);
104  lkp->pron += std::string(phone, strlen(phone) - 1);
105  lkp->pron += stress;
106  free((void *)stress);
107  } else {
108  lkp->pron += phone;
109  }
110  }
111  free((void *)orig);
112  } else {
113  // HACKY: Remove annoying warning for English possessive form
114  if (ltr.compare("'"))
115  KALDI_WARN << "Letter not in LTS tree: " << ltr;
116  }
117  p += clen;
118  pos++;
119  }
120  lkp->lts = true;
121  return true;
122 }

124 static int32 pos2int(const std::string &pos) {
125  int32 loc = 0, idx = 0;
126  while ((loc = pos.find("n.", loc)) != std::string::npos) {
127  idx += 1;
128  loc += 2;
129  }
130  loc = 0;
131  while ((loc = pos.find("p.", loc)) != std::string::npos) {
132  idx -= 1;
133  loc += 2;
134  }
135  return idx;
136 }

138 static bool ApplyQuestion(const char* word, const int32 pos, const char* utfchar) {
139  TxpUtf8 utf8;
140  int32 clen = 0, idx = 0;
141  const char* p = word;
142  // position before word start
143  if (pos < 0 && !strcmp(utfchar, "#")) return true;
144  while (*p) {
145  clen = utf8.Clen(p);
146  if (idx == pos) break;
147  p += clen;
148  idx++;
149  }
150  // position after word end
151  if (*p && !strcmp(utfchar, "#")) return true;
152  // matches utf8char at pos
153  if (!strncmp(p, utfchar, clen)) return true;
154  return false;
155 }

157 static const char* ApplyTree(const TxpLtsTree &tree,
158  const char* word,
159  const int32 pos) {
160  int32 currentnode = tree.root;
161  TxpLtsNode node;
162  while (1) {
163  node = tree.nodes[currentnode];
164  // check for non terminal node
165  if (!node.val.empty()) {
166  break;
167  } else {
168  if (ApplyQuestion(word, pos + node.pos, node.posval.c_str())) {
169  currentnode = node.yes;
170  } else {
171  currentnode = node.no;
172  }
173  }
174  }
175  return strdup(node.val.c_str());
176 }
