266 // read kaldi context setup xml to allow on th fly question generation from
267 // new questions added by kaldi i.e. by merging questions or for built in
268 // kaldi phone questions.
269 void _read_context_setup(std::string kaldicontextsetup_filename,
270  int32 N,
271  const fst::SymbolTable &phones_symtab,
272  ConvertCexVector * cvncex_array) {
273  int32 i, j, pos;
274  bool kphonefound = false, binary;
275  // Read XMl context setup information
276  Input ki(kaldicontextsetup_filename, &binary);
277  pugi::xml_node kphone;
278  pugi::xml_node cexfunction;
279  pugi::xml_document doc;
280  pugi::xml_parse_result r = doc.load(ki.Stream(), pugi::encoding_utf8);
281  ConvertCex cnvcex;
282 
283  // Add this look up for kaldi phones
284  pugi::xpath_node_set kphones =
285  doc.document_element().select_nodes("//kphone");
286  kphones.sort();
287 
288  for(i = 0; i < N; i++) {
289  pos = i - (N/2);
290  std::string name = "";
291  // construct a context name
292  if (pos < 0) {
293  for(j = 0; j > pos; j--)
294  name = name + "Backward";
295  }
296  else if (pos > 0) {
297  for(j = 0; j < pos; j++)
298  name = name + "Forward";
299  }
300  name = name + "PhoneKaldi";
301  kphonefound = false;
302  for (pugi::xpath_node_set::const_iterator it = kphones.begin();
303  it != kphones.end();
304  ++it) {
305  kphone = (*it).node();
306  if (!strcmp(kphone.attribute("name").as_string(), name.c_str())) {
307  kphonefound = true;
308  break;
309  }
310  }
311  if (!kphonefound)
312  KALDI_ERR << "No appropriate kphone context in HTS model for: "<< name;
313 
314  cnvcex.Init(name,
315  std::string(
316  kphone.attribute("htsprv").as_string()),
317  std::string(
318  kphone.attribute("htspst").as_string()),
319  false);
320  // Take symbol names from kaldi phone set and add to the
321  // conversion table
322  // kaldi phones are by default the first N fields in the kaldi full
323  // context integer array
324  for (j = 0; j < phones_symtab.NumSymbols(); j++)
325  cnvcex.AddVal(phones_symtab.Find(j));
326  cvncex_array->push_back(cnvcex);
327  }
328 
329  // Add the other lookups from the context setup
330  pugi::xpath_node_set cexfunctions = doc.document_element().select_nodes("//cexfunction");
331  cexfunctions.sort();
332  for (pugi::xpath_node_set::const_iterator it = cexfunctions.begin();
333  it != cexfunctions.end();
334  ++it) {
335  int32 isinteger;
336  cexfunction = (*it).node();
337  isinteger = cexfunction.attribute("isinteger").as_int();
338  //std::cout << cexfunction.attribute("name").as_string() << isinteger << "\n";
339  cnvcex.Init(std::string(
340  cexfunction.attribute("name").as_string()),
341  std::string(
342  cexfunction.attribute("htsprv").as_string()),
343  std::string(
344  cexfunction.attribute("htspst").as_string()),
345  isinteger);
346  if (isinteger == 0) {
347  pugi::xpath_node_set vals = cexfunction.select_nodes("descendant::val");
348  for (pugi::xpath_node_set::const_iterator it = vals.begin();
349  it != vals.end();
350  ++it) {
351  pugi::xml_node val = (*it).node();
352  cnvcex.SetVal(val.attribute("mapping").as_int(),
353  std::string(val.attribute("name").as_string()));
354  }
355  }
356  cvncex_array->push_back(cnvcex);
357  }
358 }

361 // Only questions true for the state value are output
362 void _convert_tree(const std::string &tab, const EventMap * emap,
363  bool yes,
364  const ConvertCexVector &cvncex_array,
365  TreeNodeVector *treenode_array,
366  LookupInt * contextname2index,
367  int32 state, TreeNode * parent) {
368  // to get indentation on recursion
369  std::string newtab = tab + " ";
370  // string to hold full kaldi style question
371  std::string kaldiq;
372  // string to hold hts question name
373  std::string htsqname;
374  // string to hold full hts context style question
375  std::string htsq;
376  // tree node
377  std::vector<EventMap*> out;
378  // set of correct values for yes branch
379  const ConstIntegerSet<EventValueType> * yesset;
380  // integer to string info to map kaldi question to hts question
381  const ConvertCex * cnvcex;
382  LookupInt::iterator lkp;
383  // if a state question is the answer yes or no
384  bool stateqY = false;
385  // Get the node information
386  emap->GetChildren(&out);
387  // It is a non terminal node
388  if (out.size()) {
389  yesset = emap->YesSet();
390  kaldiq.clear();
391  std::ostringstream kaldiqstream(kaldiq);
392  std::ostringstream htsqstream(kaldiq);
393  kaldiqstream << emap->first << " ?";
394  // if not a state question retrieve the context info
395  if (emap->first != -1) {
396  cnvcex = &cvncex_array.at(emap->first);
397  htsqstream << cnvcex->GetName();
398  lkp = contextname2index->find(cnvcex->GetName());
399  if (lkp == contextname2index->end()) {
400  contextname2index->insert(LookupIntItem(cnvcex->GetName(), 1));
401  htsqstream << "1";
402  }
403  else {
404  contextname2index->at(cnvcex->GetName()) = lkp->second + 1;
405  htsqstream << (lkp->second) + 1;
406  }
407  }
408  else
409  cnvcex = NULL;
410 
411  htsqname = htsqstream.str();
412  htsqstream.clear();
413  htsqstream << " { ";
414  for (ConstIntegerSet<EventValueType>::iterator iter = yesset->begin();
415  iter != yesset->end();
416  ++iter) {
417  kaldiqstream << " " << *iter;
418  // we do not need to build HTS questiosn for state questions
419  // as in HTS these are different trees
420  if (emap->first != -1) {
421  if (iter != yesset->begin()) htsqstream << ",";
422  //std::cout << htsqstream.str() << "!!\n";
423  htsqstream << "\"*" << cnvcex->GetPrvdelim();
424  if (cnvcex->GetIsinteger()) htsqstream << *iter;
425  else htsqstream << cnvcex->LookupVal(*iter);
426  htsqstream << cnvcex->GetPstdelim() << "*\"";
427  }
428  else {
429  if (!stateqY && *iter == state) stateqY = true;
430  }
431  // std::cout << *iter << " ";
432  }
433  htsqstream << " }";
434  //std::cout << "]\n";
435  if (emap->first == -1) {
436  //std::cout << stateqY << " " << state << " " << kaldiqstream.str() << "\n";
437  if (stateqY) _convert_tree(tab, out[0], yes, cvncex_array, treenode_array,
438  contextname2index, state, parent);
439  else _convert_tree(tab, out[1], yes, cvncex_array, treenode_array,
440  contextname2index, state, parent);
441  }
442  else {
443  TreeNode * node = new TreeNode(htsqname, htsqstream.str());
444  treenode_array->push_back(node);
445  if (parent) parent->SetParent(yes, treenode_array->size() - 1);
446  //std::cout << newtab << htsqname << "\n";
447  //std::cout << newtab << "[" << kaldiqstream.str() << "]\n";
448 
449  _convert_tree(newtab, out[0], true, cvncex_array, treenode_array,
450  contextname2index, state, node);
451  _convert_tree(newtab, out[1], false, cvncex_array, treenode_array,
452  contextname2index, state, node);
453  }
454  }
455  // it is a terminal node
456  else {
457  int32 val = static_cast<const ConstantEventMap * >(emap)->GetAnswer();
458  parent->SetVal(yes, val);
459  //std::cout << newtab << "A:" << static_cast<const ConstantEventMap * >(emap)->GetAnswer() << "\n";
460  }
461 }
