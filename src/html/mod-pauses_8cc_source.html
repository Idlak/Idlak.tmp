44 // If fileids are present in the document it breaks the
45 // document up by them
46 bool TxpPauses::Process(pugi::xml_document* input) {
47  pugi::xpath_node_set files;
48  files = input->document_element().select_nodes("//fileid");
49  if (files.size()) {
50  for (pugi::xpath_node_set::const_iterator it = files.begin();
51  it != files.end();
52  ++it) {
53  pugi::xml_node file = (*it).node();
54  ProcessFile(&file);
55  }
56  } else {
57  pugi::xml_node document_element = input->document_element();
58  ProcessFile(&document_element);
59  }
60  return true;
61 }

63 // Adds breaks as required and ensures all documents/fileids
64 // have a break initial and final
65 bool TxpPauses::ProcessFile(pugi::xml_node* file) {
66  pugi::xml_node *breakitem = NULL, *ptk = NULL, pretk;
67  const TxpPbreakInfo* pbreak;
68  TxpPbreakInfo pbreakpunc;
69  bool prebreak, pstbreak, newline = false, newline2 = false;
70  pugi::xpath_node_set files;
71  pugi::xpath_node_set tks =
72  file->select_nodes(".//tk|.//break|.//ws");
73  tks.sort();
74  for (pugi::xpath_node_set::const_iterator it = tks.begin();
75  it != tks.end();
76  ++it) {
77  pugi::xml_node tk = (*it).node();
78  if (!strcmp(tk.name(), "break")) {
79  if (!tk.attribute("type")) tk.append_attribute("type");
80  if (!tk.attribute("time")) tk.append_attribute("time");
81  if (!tk.attribute("strength").empty()) {
82  pbreak = pbreak_.GetPbreakPst(tk.attribute("strength").value());
83  if (pbreak) {
84  tk.attribute("type").set_value(pbreak->type);
85  tk.attribute("time").set_value(pbreak->time);
86  }
87  }
88  // break tags override punctuation so keep a record of what happened
89  // before tk
90  if (!*tk.attribute("type").value()) {
91  tk.attribute("type").set_value(pbreak_.get_default_type());
92  }
93  if (!*tk.attribute("time").value()) {
94  tk.attribute("time").set_value(pbreak_.get_default_time());
95  }
96  breakitem = &tk;
97  // tk.print(std::cout);
98  } else if (!strcmp(tk.name(), "ws")) {
99  pbreak_.GetWhitespaceBreaks(tk.first_child().value(),
100  tk.attribute("col").as_int(0),
101  hzone_, hzone_start_, hzone_end_,
102  &newline, &newline2);
103  } else if (!strcmp(tk.name(), "tk")) {
104  // std::cout << "!" << tk.attribute("norm").value() << "\n";
105  // if (ptk) std::cout << "P:" << ptk->attribute("norm").value() << " - ";
106  // std::cout << "L:" << tk.attribute("norm").value() << std::endl;
107  pbreakpunc.Clear();
108  // Only add breaks if no break is present
109  if (!breakitem) {
110  // first token
111  if (!ptk) {
112  if (pbreak_.GetPbreak(tk.attribute("prepunc").value(),
113  TXPPBREAK_POS_PRE, pbreakpunc)) {
114  pbreak = &pbreakpunc;
115  } else {
116  // Insert default document start break
117  pbreak = pbreak_.GetPbreakPre("DEFAULT");
118  }
119  _insert_break_before(&tk, pbreak);
120  } else {
121  // Between tokens
122  // break caused by punctuation
123  // std::cout << "X:" << tk.attribute("prepunc").value() << std::endl;
124  pstbreak = pbreak_.GetPbreak(ptk->attribute("pstpunc").value(),
125  TXPPBREAK_POS_PST, pbreakpunc);
126  prebreak = pbreak_.GetPbreak(tk.attribute("prepunc").value(),
127  TXPPBREAK_POS_PRE, pbreakpunc);
128  pbreak = &pbreakpunc;
129  if (!prebreak && !pstbreak) {
130  // Break caused by whitespace
131  if (newline2) {
132  pbreak = pbreak_.GetPbreakPst("newlineX2");
133  } else if (newline) {
134  pbreak = pbreak_.GetPbreakPst("newlineX2");
135  } else {
136  pbreak = NULL;
137  }
138  }
139  _insert_break_after(ptk, pbreak);
140  }
141  }
142  pretk = (*it).node();
143  ptk = &pretk;
144  breakitem = NULL;
145  }
146  }
147  // last item
148  if (!breakitem && ptk) {
149  if (pbreak_.GetPbreak(ptk->attribute("pstpunc").value(),
150  TXPPBREAK_POS_PST, pbreakpunc)) {
151  pbreak = &pbreakpunc;
152  } else {
153  // Insert default document end break
154  pbreak = pbreak_.GetPbreakPst("DEFAULT");
155  }
156  _insert_break_after(ptk, pbreak);
157  }
158  return true;
159 }

161 static void _insert_break_after(pugi::xml_node* tk,
162  const TxpPbreakInfo* pbreak) {
163  pugi::xml_node breakitem;
164  if (!pbreak) return;
165  // std::cout << tk.attribute("norm").value() << std::endl;
166  breakitem = tk->parent().insert_child_after("break", *tk);
167  breakitem.append_attribute("type").set_value(pbreak->type);
168  breakitem.append_attribute("time").set_value(pbreak->time);
169 }
