37  std::string cexfunctions;
38  pugi::xml_node header = GetHeader(input);
39  cexspec_.GetFunctionSpec(&header);
40  cexspec_.AddPauseNodes(input);
41  pugi::xpath_node_set tks =
42  input->document_element().select_nodes("//phon");
43  tks.sort();
44  TxpCexspecContext context(*input, cexspec_.GetPauseHandling());
45  kaldi::int32 i = 0;
46  for (pugi::xpath_node_set::const_iterator it = tks.begin();
47  it != tks.end();
48  ++it, i++, context.Next()) {
49  pugi::xml_node phon = (*it).node();
50  model->clear();
51  cexspec_.ExtractFeatures(context, model);
52  phon.text() = model->c_str();
53  }
54  delete model;
55  // should set to error status
56  return true;
57 }
