47 void SpectrogramComputer::Compute(BaseFloat signal_log_energy,
48  BaseFloat vtln_warp,
49  VectorBase<BaseFloat> *signal_frame,
50  VectorBase<BaseFloat> *feature) {
51  KALDI_ASSERT(signal_frame->Dim() == opts_.frame_opts.PaddedWindowSize() &&
52  feature->Dim() == this->Dim());
53 
54 
55  // Compute energy after window function (not the raw one)
56  if (!opts_.raw_energy)
57  signal_log_energy = Log(std::max(VecVec(*signal_frame, *signal_frame),
58  std::numeric_limits<float>::epsilon()));
59 
60  if (srfft_ != NULL) // Compute FFT using split-radix algorithm.
61  srfft_->Compute(signal_frame->Data(), true);
62  else // An alternative algorithm that works for non-powers-of-two
63  RealFft(signal_frame, true);
64 
65  // Convert the FFT into a power spectrum.
66  ComputePowerSpectrum(signal_frame);
67  SubVector<BaseFloat> power_spectrum(*signal_frame,
68  0, signal_frame->Dim() / 2 + 1);
69 
70  power_spectrum.ApplyFloor(std::numeric_limits<float>::epsilon());
71  power_spectrum.ApplyLog();
72 
73  feature->CopyFromVec(power_spectrum);
74 
75  if (opts_.energy_floor > 0.0 && signal_log_energy < log_energy_floor_)
76  signal_log_energy = log_energy_floor_;
77  // The zeroth spectrogram component is always set to the signal energy,
78  // instead of the square of the constant component of the signal.
79  (*feature)(0) = signal_log_energy;
80 }
