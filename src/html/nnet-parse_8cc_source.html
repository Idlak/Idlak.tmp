31 bool ConfigLine::ParseLine(const std::string &line) {
32  data_.clear();
33  whole_line_ = line;
34  if (line.size() == 0) return false; // Empty line
35  size_t pos = 0, size = line.size();
36  while (isspace(line[pos]) && pos < size) pos++;
37  if (pos == size)
38  return false; // whitespace-only line
39  size_t first_token_start_pos = pos;
40  // first get first_token_.
41  while (!isspace(line[pos]) && pos < size) {
42  if (line[pos] == '=') {
43  // If the first block of non-whitespace looks like "foo-bar=...",
44  // then we ignore it: there is no initial token, and FirstToken()
45  // is empty.
46  pos = first_token_start_pos;
47  break;
48  }
49  pos++;
50  }
51  first_token_ = std::string(line, first_token_start_pos, pos - first_token_start_pos);
52  // first_token_ is expected to be either empty or something like
53  // "component-node", which actually is a slightly more restrictive set of
54  // strings than IsValidName() checks for this is a convenient way to check it.
55  if (!first_token_.empty() && !IsValidName(first_token_))
56  return false;
57 
58  while (pos < size) {
59  if (isspace(line[pos])) {
60  pos++;
61  continue;
62  }
63 
64  // OK, at this point we know that we are pointing at nonspace.
65  size_t next_equals_sign = line.find_first_of("=", pos);
66  if (next_equals_sign == pos || next_equals_sign == std::string::npos) {
67  // we're looking for something like 'key=value'. If there is no equals sign,
68  // or it's not preceded by something, it's a parsing failure.
69  return false;
70  }
71  std::string key(line, pos, next_equals_sign - pos);
72  if (!IsValidName(key)) return false;
73 
74  // handle any quotes. we support key='blah blah' or key="foo bar".
75  // no escaping is supported.
76  if (line[next_equals_sign+1] == '\'' || line[next_equals_sign+1] == '"') {
77  char my_quote = line[next_equals_sign+1];
78  size_t next_quote = line.find_first_of(my_quote, next_equals_sign + 2);
79  if (next_quote == std::string::npos) { // no matching quote was found.
80  KALDI_WARN << "No matching quote for " << my_quote << " in config line '"
81  << line << "'";
82  return false;
83  } else {
84  std::string value(line, next_equals_sign + 2,
85  next_quote - next_equals_sign - 2);
86  data_.insert(std::make_pair(key, std::make_pair(value, false)));
87  pos = next_quote + 1;
88  continue;
89  }
90  } else {
91  // we want to be able to parse something like "... input=Offset(a, -1) foo=bar":
92  // in general, config values with spaces in them, even without quoting.
93 
94  size_t next_next_equals_sign = line.find_first_of("=", next_equals_sign + 1),
95  terminating_space = size;
96 
97  if (next_next_equals_sign != std::string::npos) { // found a later equals sign.
98  size_t preceding_space = line.find_last_of(" \t", next_next_equals_sign);
99  if (preceding_space != std::string::npos &&
100  preceding_space > next_equals_sign)
101  terminating_space = preceding_space;
102  }
103  while (isspace(line[terminating_space - 1]) && terminating_space > 0)
104  terminating_space--;
105 
106  std::string value(line, next_equals_sign + 1,
107  terminating_space - (next_equals_sign + 1));
108  data_.insert(std::make_pair(key, std::make_pair(value, false)));
109  pos = terminating_space;
110  }
111  }
112  return true;
113 }

220 // This is like ExpectToken but for two tokens, and it
221 // will either accept token1 and then token2, or just token2.
222 // This is useful in Read functions where the first token
223 // may already have been consumed.
224 void ExpectOneOrTwoTokens(std::istream &is, bool binary,
225  const std::string &token1,
226  const std::string &token2) {
227  KALDI_ASSERT(token1 != token2);
228  std::string temp;
229  ReadToken(is, binary, &temp);
230  if (temp == token1) {
231  ExpectToken(is, binary, token2);
232  } else {
233  if (temp != token2) {
234  KALDI_ERR << "Expecting token " << token1 << " or " << token2
235  << " but got " << temp;
236  }
237  }
238 }

241 bool ParseFromString(const std::string &name, std::string *string,
242  int32 *param) {
243  std::vector<std::string> split_string;
244  SplitStringToVector(*string, " \t", true,
245  &split_string);
246  std::string name_equals = name + "="; // the name and then the equals sign.
247  size_t len = name_equals.length();
248 
249  for (size_t i = 0; i < split_string.size(); i++) {
250  if (split_string[i].compare(0, len, name_equals) == 0) {
251  if (!ConvertStringToInteger(split_string[i].substr(len), param))
252  KALDI_ERR << "Bad option " << split_string[i];
253  *string = "";
254  // Set "string" to all the pieces but the one we used.
255  for (size_t j = 0; j < split_string.size(); j++) {
256  if (j != i) {
257  if (!string->empty()) *string += " ";
258  *string += split_string[j];
259  }
260  }
261  return true;
262  }
263  }
264  return false;
265 }

377 bool DescriptorTokenize(const std::string &input,
378  std::vector<std::string> *tokens) {
379  KALDI_ASSERT(tokens != NULL);
380  size_t start = input.find_first_not_of(" \t"), size = input.size();
381  tokens->clear();
382  while (start < size) {
383  KALDI_ASSERT(!isspace(input[start]));
384  if (input[start] == '(' || input[start] == ')' || input[start] == ',') {
385  tokens->push_back(std::string(input, start, 1));
386  start = input.find_first_not_of(" \t", start + 1);
387  } else {
388  size_t found = input.find_first_of(" \t(),", start);
389  KALDI_ASSERT(found != start);
390  if (found == std::string::npos) {
391  std::string str(input, start, input.size() - start);
392  BaseFloat tmp;
393  if (!IsValidName(str) && !ConvertStringToReal(str, &tmp)) {
394  KALDI_WARN << "Could not tokenize line " << ErrorContext(std::string(input, start));
395  return false;
396  }
397  tokens->push_back(str);
398  break;
399  } else {
400  if (input[found] == '(' || input[found] == ')' || input[found] == ',') {
401  std::string str(input, start, found - start);
402  BaseFloat tmp;
403  if (!IsValidName(str) && !ConvertStringToReal(str, &tmp)) {
404  KALDI_WARN << "Could not tokenize line " << ErrorContext(std::string(input, start));
405  return false;
406  }
407  tokens->push_back(str);
408  start = found;
409  } else {
410  std::string str(input, start, found - start);
411  BaseFloat tmp;
412  if (!IsValidName(str) && !ConvertStringToReal(str, &tmp)) {
413  KALDI_WARN << "Could not tokenize line " << ErrorContext(std::string(input, start));
414  return false;
415  }
416  tokens->push_back(str);
417  start = input.find_first_not_of(" \t", found);
418  }
419  }
420  }
421  }
422  return true;
423 }

425 bool IsValidName(const std::string &name) {
426  if (name.size() == 0) return false;
427  for (size_t i = 0; i < name.size(); i++) {
428  if (i == 0 && !isalpha(name[i]) && name[i] != '_')
429  return false;
430  if (!isalnum(name[i]) && name[i] != '_' && name[i] != '-' && name[i] != '.')
431  return false;
432  }
433  return true;
434 }

436 void ReadConfigLines(std::istream &is,
437  std::vector<std::string> *lines) {
438  KALDI_ASSERT(lines != NULL);
439  std::string line;
440  while (std::getline(is, line)) {
441  if (line.size() == 0) continue;
442  size_t start = line.find_first_not_of(" \t");
443  size_t end = line.find_first_of('#');
444  if (start == std::string::npos || start == end) continue;
445  end = line.find_last_not_of(" \t", end - 1);
446  KALDI_ASSERT(end >= start);
447  lines->push_back(line.substr(start, end - start + 1));
448  }
449 }

467 static void PrintFloatSuccinctly(std::ostream &os, BaseFloat f) {
468  if (fabs(f) < 10000.0 && fabs(f) >= 10.0) {
469  os << std::fixed << std::setprecision(0) << f;
470  } else if (fabs(f) >= 0.995) {
471  os << std::fixed << std::setprecision(1) << f;
472  } else if (fabs(f) >= 0.01) {
473  os << std::fixed << std::setprecision(2) << f;
474  } else {
475  os << std::setprecision(1) << f;
476  }
477  os.unsetf(std::ios_base::floatfield);
478  os << std::setprecision(6); // Restore the default.
479 }

482 // Returns a string that summarizes a vector fairly succintly, for
483 // printing stats in info lines.
484 std::string SummarizeVector(const VectorBase<float> &vec) {
485  std::ostringstream os;
486  if (vec.Dim() < 10) {
487  os << "[ ";
488  for (int32 i = 0; i < vec.Dim(); i++) {
489  PrintFloatSuccinctly(os, vec(i));
490  os << ' ';
491  }
492  os << "]";
493  } else {
494  // print out mean and standard deviation, and some selected values.
495  BaseFloat mean = vec.Sum() / vec.Dim(),
496  stddev = sqrt(VecVec(vec, vec) / vec.Dim() - mean * mean);
497 
498  std::string percentiles_str = "0,1,2,5 10,20,50,80,90 95,98,99,100";
499  std::vector<int32> percentiles;
500  bool ans = SplitStringToIntegers(percentiles_str, ", ", false,
501  &percentiles);
502  KALDI_ASSERT(ans);
503  os << "[percentiles(" << percentiles_str << ")=(";
504  Vector<BaseFloat> vec_sorted(vec);
505  std::sort(vec_sorted.Data(), vec_sorted.Data() + vec_sorted.Dim());
506  int32 n = vec.Dim() - 1;
507  for (size_t i = 0; i < percentiles.size(); i++) {
508  int32 percentile = percentiles[i];
509  BaseFloat value = vec_sorted((n * percentile) / 100);
510  PrintFloatSuccinctly(os, value);
511  if (i + 1 < percentiles.size())
512  os << (i == 3 || i == 8 ? ' ' : ',');
513  }
514  os << std::setprecision(3);
515  os << "), mean=" << mean << ", stddev=" << stddev << "]";
516  }
517  return os.str();
518 }

531  const std::string &name,

539  os << "{mean,stddev}=" << mean << ',' << stddev;

547 void PrintParameterStats(std::ostringstream &os,
548  const std::string &name,
549  const CuMatrix<BaseFloat> &params,
550  bool include_mean,
551  bool include_row_norms,
552  bool include_column_norms,
553  bool include_singular_values) {
554  os << std::setprecision(4);
555  os << ", " << name << '-';
556  int32 dim = params.NumRows() * params.NumCols();
557  if (include_mean) {
558  BaseFloat mean = params.Sum() / dim,
559  stddev = std::sqrt(TraceMatMat(params, params, kTrans) / dim -
560  mean * mean);
561  os << "{mean,stddev}=" << mean << ',' << stddev;
562  } else {
563  BaseFloat rms = std::sqrt(TraceMatMat(params, params, kTrans) / dim);
564  os << "rms=" << rms;
565  }
566  os << std::setprecision(6); // restore the default precision.
567 
568  if (include_row_norms) {
569  CuVector<BaseFloat> row_norms(params.NumRows());
570  row_norms.AddDiagMat2(1.0, params, kNoTrans, 0.0);
571  row_norms.ApplyPow(0.5);
572  Vector<BaseFloat> row_norms_cpu;
573  row_norms.Swap(&row_norms_cpu);
574  os << ", " << name << "-row-norms="
575  << SummarizeVector(row_norms_cpu);
576  }
577  if (include_column_norms) {
578  CuVector<BaseFloat> col_norms(params.NumCols());
579  col_norms.AddDiagMat2(1.0, params, kTrans, 0.0);
580  col_norms.ApplyPow(0.5);
581  Vector<BaseFloat> col_norms_cpu;
582  col_norms.Swap(&col_norms_cpu);
583  os << ", " << name << "-col-norms="
584  << SummarizeVector(col_norms_cpu);
585  }
586  if (include_singular_values) {
587  Matrix<BaseFloat> params_cpu(params);
588  Vector<BaseFloat> s(std::min(params.NumRows(), params.NumCols()));
589  params_cpu.Svd(&s);
590  std::string singular_values_str = SummarizeVector(s);
591  os << ", " << name << "-singular-values=" << singular_values_str;
592  std::ostringstream name_os;
593  }
594 }
