50 void GetBlockSizesForSimpleMatrixOperation(int32 num_rows,
51  int32 num_cols,
52  dim3 *dimGrid,
53  dim3 *dimBlock) {
54  KALDI_ASSERT(num_rows > 0 && num_cols > 0);
55  int32 col_blocksize = 64, row_blocksize = 4;
56  while (col_blocksize > 1 &&
57  (num_cols + (num_cols / 2) <= col_blocksize ||
58  num_rows > 65535 * row_blocksize)) {
59  col_blocksize /= 2;
60  row_blocksize *= 2;
61  }
62 
63  dimBlock->x = col_blocksize;
64  dimBlock->y = row_blocksize;
65  dimBlock->z = 1;
66  dimGrid->x = n_blocks(num_cols, col_blocksize);
67  dimGrid->y = n_blocks(num_rows, row_blocksize);
68  KALDI_ASSERT(dimGrid->y <= 65535 &&
69  "Matrix has too many rows to process");
70  dimGrid->z = 1;
71 }
