67 //static
68 BaseFloat NnetComputer::MatrixStddev(const CuMatrixBase<BaseFloat> &m) {
69  if (m.NumRows() == 0)
70  return 0.0;
71  return std::sqrt(TraceMatMat(m, m, kTrans) / (m.NumRows() * m.NumCols()));
72 }

75 BaseFloat NnetComputer::ParameterStddev(const Component &c) {
76  const UpdatableComponent *uc = dynamic_cast<const UpdatableComponent*>(&c);
77  KALDI_ASSERT(uc != NULL &&
78  "Attempting to get parameter stddev of non-updatable component");
79  return std::sqrt(uc->DotProduct(*uc) / uc->NumParameters());
80 }

82 void NnetComputer::DebugBeforeExecute(int32 command,
83  CommandDebugInfo *info) {
84  {
85  const std::vector<int32> &matrices_written =
86  command_attributes_[command].matrices_written;
87  size_t size = matrices_written.size();
88  info->matrices_written_stddevs.resize(size);
89  for (size_t i = 0; i < size; i++) {
90  int32 m = matrices_written[i];
91  info->matrices_written_stddevs[i] = MatrixStddev(matrices_[m]);
92  }
93  }
94  {
95  const std::vector<int32> &submatrices_written =
96  command_attributes_[command].submatrices_written;
97  size_t size = submatrices_written.size();
98  info->submatrices_written_stddevs.resize(size);
99  for (size_t i = 0; i < size; i++) {
100  int32 s = submatrices_written[i];
101  if (!computation_.IsWholeMatrix(s)) {
102  const CuSubMatrix<BaseFloat> submat(GetSubMatrix(s));
103  info->submatrices_written_stddevs[i] = MatrixStddev(submat);
104  }
105  }
106  }
107  const NnetComputation::Command &c = computation_.commands[command];
108  if (c.command_type == kBackprop) {
109  const Component *component = nnet_.GetComponent(c.arg1);
110  if (component->Properties() & kUpdatableComponent)
111  info->components_parameter_stddev = ParameterStddev(*component);
112  }
113 }

116 void NnetComputer::DebugAfterExecute(int32 command,
117  const CommandDebugInfo &info,
118  double command_exec_time) {
119  std::ostringstream os;
120  os << command_strings_[command] << "\t|\t";
121  {
122  const std::vector<int32> &matrices_written =
123  command_attributes_[command].matrices_written;
124  size_t size = matrices_written.size();
125  KALDI_ASSERT(info.matrices_written_stddevs.size() == size);
126  for (size_t i = 0; i < size; i++) {
127  int32 m = matrices_written[i];
128  BaseFloat old_stddev = info.matrices_written_stddevs[i],
129  stddev = MatrixStddev(matrices_[m]);
130  os << 'm' << m << ": " << old_stddev << "->" << stddev << " ";
131  }
132  }
133  {
134  const std::vector<int32> &submatrices_written =
135  command_attributes_[command].submatrices_written;
136  size_t size = submatrices_written.size();
137  KALDI_ASSERT(info.submatrices_written_stddevs.size() == size);
138  for (size_t i = 0; i < size; i++) {
139  int32 s = submatrices_written[i];
140  if (!computation_.IsWholeMatrix(s)) {
141  const CuSubMatrix<BaseFloat> submat(GetSubMatrix(s));
142  BaseFloat old_stddev = info.submatrices_written_stddevs[i],
143  stddev = MatrixStddev(submat);
144  os << submatrix_strings_[s] << ": " << old_stddev << "->"
145  << stddev << " ";
146  }
147  }
148  }
149  const NnetComputation::Command &c = computation_.commands[command];
150  if (c.command_type == kBackprop) {
151  const Component *component = nnet_.GetComponent(c.arg1);
152  if (component->Properties() & kUpdatableComponent) {
153  const std::string &component_name = nnet_.GetComponentName(c.arg1);
154  os << component_name << ": " << info.components_parameter_stddev
155  << "->" << ParameterStddev(*component) << " ";
156  }
157  }
158  os << "\t|\t time: " << command_exec_time << " secs";
159  KALDI_LOG << os.str();
160 }

210 void NnetComputer::ExecuteCommand() {
211  const NnetComputation::Command &c = computation_.commands[program_counter_];
212  int32 m1, m2;
213  try {
214  switch (c.command_type) {
215  case kAllocMatrix:
216  m1 = computation_.submatrices[c.arg1].matrix_index;
217  matrices_[m1].Resize(computation_.matrices[m1].num_rows,
218  computation_.matrices[m1].num_cols,
219  kUndefined,
220  computation_.matrices[m1].stride_type);
221  break;
222  case kDeallocMatrix:
223  m1 = computation_.submatrices[c.arg1].matrix_index;
224  matrices_[m1].Resize(0, 0);
225  break;
226  case kSwapMatrix:
227  m1 = computation_.submatrices[c.arg1].matrix_index;
228  m2 = computation_.submatrices[c.arg2].matrix_index;
229  matrices_[m1].Swap(&(matrices_[m2]));
230  break;
231  case kSetConst: {
232  CuSubMatrix<BaseFloat> s(GetSubMatrix(c.arg1));
233  if (c.alpha == 0.0) s.SetZero();
234  else s.Set(c.alpha);
235  break;
236  }
237  case kPropagate: {
238  const Component *component = nnet_.GetComponent(c.arg1);
239  ComponentPrecomputedIndexes *indexes =
240  computation_.component_precomputed_indexes[c.arg2].data;
241  const CuSubMatrix<BaseFloat> input(GetSubMatrix(c.arg3));
242  CuSubMatrix<BaseFloat> output(GetSubMatrix(c.arg4));
243  void *memo = component->Propagate(indexes, input, &output);
244  if (c.arg6) { // need to store stats.
245  KALDI_ASSERT(nnet_to_store_stats_ != NULL);
246  Component *stats_component = nnet_to_store_stats_->GetComponent(c.arg1);
247  bool was_in_place = (c.arg3 == c.arg4);
248  // if propagate was in-place, provide empty matrix and not 'input', as
249  // input is no longer valid.
250  const CuSubMatrix<BaseFloat> maybe_input(
251  GetSubMatrix(was_in_place ? 0 : c.arg3));
252  stats_component->StoreStats(maybe_input, output, memo);
253  }
254  SaveMemo(c.arg5, *component, memo);
255  break;
256  }
257  case kBackprop:
258  case kBackpropNoModelUpdate: {
259  std::ostringstream debug_str;
260  KALDI_ASSERT(nnet_to_update_ != NULL);
261  debug_str << nnet_.GetComponentName(c.arg1);
262  const Component *component = nnet_.GetComponent(c.arg1);
263  KALDI_ASSERT(!(computation_.need_model_derivative && !nnet_to_update_));
264  Component *upd_component = NULL;
265  if (c.command_type == kBackprop) { // this block sets 'upd_component'
266  Nnet *nnet_to_update;
267  if (component->Properties()&kUpdatableComponent) {
268  nnet_to_update = (computation_.need_model_derivative ?
269  nnet_to_update_ : NULL);
270  } else {
271  // Some non-updatable components, such as CompositeComponent, store
272  // stats in the backprop. For other types of non-updatable
273  // component, this arg won't matter.
274  nnet_to_update = nnet_to_store_stats_;
275  }
276  if (nnet_to_update)
277  upd_component = nnet_to_update->GetComponent(c.arg1);
278  }
279  ComponentPrecomputedIndexes *indexes =
280  computation_.component_precomputed_indexes[c.arg2].data;
281  const CuSubMatrix<BaseFloat> in_value(GetSubMatrix(c.arg3));
282  const CuSubMatrix<BaseFloat> out_value(GetSubMatrix(c.arg4));
283  const CuSubMatrix<BaseFloat> out_deriv(GetSubMatrix(c.arg5));
284  CuSubMatrix<BaseFloat> in_deriv(GetSubMatrix(c.arg6));
285  void *memo = GetMemo(c.arg7);
286  component->Backprop(debug_str.str(), indexes,
287  in_value, out_value, out_deriv,
288  memo, upd_component,
289  c.arg6 == 0 ? NULL : &in_deriv);
290  if (memo != NULL)
291  component->DeleteMemo(memo);
292  break;
293  }
294  case kMatrixCopy: {
295  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
296  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
297  dest.CopyFromMat(src);
298  if (c.alpha != 1.0)
299  dest.Scale(c.alpha); // note: in principle in future we could write a
300  // kernel which would do this in one operation.
301  break;
302  }
303  case kMatrixAdd: {
304  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
305  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
306  dest.AddMat(c.alpha, src);
307  break;
308  }
309  case kAddRows: {
310  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
311  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
312  const CuArray<int32> &indexes = computation_.indexes_cuda[c.arg3];
313  dest.AddRows(c.alpha, src, indexes);
314  break;
315  }
316  case kCopyRows: {
317  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
318  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
319  const CuArray<int32> &indexes = computation_.indexes_cuda[c.arg3];
320  BaseFloat alpha = c.alpha;
321  if (alpha != 1.0) { // for now we're faking the 'alpha' thing because the CopyRows
322  if (alpha == 0.0) break; // command doesn't take that argument.
323  dest.Scale(1.0 / alpha);
324  dest.CopyRows(src, indexes);
325  dest.Scale(c.alpha);
326  } else {
327  dest.CopyRows(src, indexes);
328  }
329  break;
330  }
331  case kCopyRowsMulti: {
332  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
333  CuArray<const BaseFloat*> pointers;
334  GetPointers(c.arg2, dest.NumCols(), &pointers);
335  BaseFloat alpha = c.alpha;
336  if (alpha != 1.0) { // for now we're faking the 'alpha' thing because the CopyRows
337  if (alpha == 0.0) break; // command doesn't take that argument.
338  dest.Scale(1.0 / alpha);
339  dest.CopyRows(pointers);
340  dest.Scale(c.alpha);
341  } else {
342  dest.CopyRows(pointers);
343  }
344  break;
345  }
346  case kCopyToRowsMulti: {
347  // If c.alpha is not 1.0, this command is not supported.
348  KALDI_ASSERT(c.alpha == 1.0);
349  CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg1));
350  CuArray<BaseFloat*> pointers;
351  GetPointers(c.arg2, src.NumCols(), &pointers);
352  src.CopyToRows(pointers);
353  break;
354  }
355  case kAddRowsMulti: {
356  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
357  CuArray<const BaseFloat*> pointers;
358  GetPointers(c.arg2, dest.NumCols(), &pointers);
359  dest.AddRows(c.alpha, pointers);
360  break;
361  }
362  case kAddToRowsMulti: {
363  CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg1));
364  CuArray<BaseFloat*> pointers;
365  GetPointers(c.arg2, src.NumCols(), &pointers);
366  src.AddToRows(c.alpha, pointers);
367  break;
368  }
369  case kAddRowRanges: {
370  CuSubMatrix<BaseFloat> dest(GetSubMatrix(c.arg1));
371  const CuSubMatrix<BaseFloat> src(GetSubMatrix(c.arg2));
372  const CuArray<Int32Pair> &pairs = computation_.indexes_ranges_cuda[c.arg3];
373  BaseFloat alpha = c.alpha;
374  if (alpha != 1.0) { // for now we're faking the 'alpha' thing
375  // because the AddRowRanges
376  if (alpha == 0.0) break; // command doesn't take that argument.
377  dest.Scale(1.0 / alpha);
378  dest.AddRowRanges(src, pairs);
379  dest.Scale(c.alpha);
380  } else {
381  dest.AddRowRanges(src, pairs);
382  }
383  break;
384  }
385  case kCompressMatrix:
386  // This does nothing if CUDA is not in use.
387 #if HAVE_CUDA == 1
388  if (CuDevice::Instantiate().Enabled()) {
389  if (compressed_matrices_.empty())
390  compressed_matrices_.resize(matrices_.size(), NULL);
391  int32 m = computation_.submatrices[c.arg1].matrix_index;
392  KALDI_ASSERT(compressed_matrices_[m] == NULL &&
393  matrices_[m].NumRows() != 0);
394  BaseFloat range = c.alpha;
395  bool truncate = (c.arg3 != 0);
396  compressed_matrices_[m] = NewCuCompressedMatrix(
397  static_cast<CuCompressedMatrixType>(c.arg2),
398  range, truncate);
399  compressed_matrices_[m]->CopyFromMat(matrices_[m]);
400  matrices_[m].Resize(0, 0);
401  }
402 #endif
403  break;
404  case kDecompressMatrix:
405 #if HAVE_CUDA == 1
406  if (CuDevice::Instantiate().Enabled()) {
407  int32 m = computation_.submatrices[c.arg1].matrix_index;
408  CuCompressedMatrixBase *compressed_matrix =
409  compressed_matrices_[m];
410  KALDI_ASSERT(compressed_matrix != NULL &&
411  matrices_[m].NumRows() == 0);
412  matrices_[m].Resize(compressed_matrix->NumRows(),
413  compressed_matrix->NumCols(),
414  kUndefined,
415  computation_.matrices[m].stride_type);
416  compressed_matrix->CopyToMat(&(matrices_[m]));
417  delete compressed_matrix;
418  compressed_matrices_[m] = NULL;
419  }
420 #endif
421  break;
422  case kNoOperation: case kNoOperationPermanent: case kNoOperationMarker:
423  case kNoOperationLabel:
424  break;
425  case kGotoLabel:
426  KALDI_ASSERT(computation_.commands[c.arg1].command_type == kNoOperationLabel);
427  program_counter_ = c.arg1;
428  break;
429  default:
430  KALDI_ERR << "Invalid command in computation";
431  }
432  } catch (...) {
433  if (!debug_) {
434  std::string preamble;
435  computation_.GetCommandStrings(nnet_, &preamble, &command_strings_);
436  KALDI_WARN << "Printing some background info since error was detected";
437  KALDI_LOG << preamble;
438  for (int32 prev_c = 0; prev_c < program_counter_; prev_c++)
439  KALDI_LOG << command_strings_[prev_c];
440  }
441  // the following will re-throw the error, but now we've printed more info
442  // about what went wrong.
443  KALDI_ERR << "Error running command " << command_strings_[program_counter_];
444  }
445 }

457 void NnetComputer::GetPointers(int32 indexes_multi_index,
458  int32 num_cols,
459  CuArray<BaseFloat*> *pointers) {
460  KALDI_ASSERT(static_cast<size_t>(indexes_multi_index)
461  < computation_.indexes_multi.size());
462  const std::vector<std::pair<int32,int32> > &pairs =
463  computation_.indexes_multi[indexes_multi_index];
464  int32 size = pairs.size();
465  std::vector<BaseFloat*> vec(size);
466 
467  // the map "lookup" maps from submatrix index to the Data()
468  // pointer of that submatrix, and the corresponding Stride().
469  unordered_map<int32, std::pair<BaseFloat*, int32> > lookup;
470 
471  for (int32 i = 0; i < size; i++) {
472  int32 submatrix_index = pairs[i].first, row = pairs[i].second;
473  if (submatrix_index != -1) {
474  unordered_map<int32, std::pair<BaseFloat*, int32> >::iterator
475  iter = lookup.find(submatrix_index);
476  if (iter == lookup.end()) {
477  CuSubMatrix<BaseFloat> m = GetSubMatrix(submatrix_index);
478  lookup[submatrix_index] = std::pair<BaseFloat*, int32>(m.Data(),
479  m.Stride());
480  iter = lookup.find(submatrix_index);
481  }
482  BaseFloat *data = iter->second.first;
483  int32 stride = iter->second.second;
484  vec[i] = data + (row * stride);
485  } else {
486  // -1 is a marker that will be translated to NULL.
487  vec[i] = NULL;
488  }
489  }
490 #ifdef KALDI_PARANOID
491  for (int32 i = 0; i < size; i += 30 + RandInt(0, 9)) {
492  // Do a pseudo-random spot check that the row-indexes are not out of range.
493  int32 submatrix_index = pairs[i].first, row = pairs[i].second;
494  CuSubMatrix<BaseFloat> m = GetSubMatrix(submatrix_index);
495  KALDI_ASSERT(row >= 0 && row < m.NumRows() && num_cols == m.NumCols());
496  }
497 #endif
498  pointers->CopyFromVec(vec);
499 }

508 void NnetComputer::Run() {
509  const std::vector<NnetComputation::Command> &c = computation_.commands;
510  int32 num_commands = c.size();
511 
512  if (program_counter_ >= num_commands) {
513  computation_.Print(std::cerr, nnet_);
514  KALDI_ERR << "Running computation that has finished: program-counter="
515  << program_counter_;
516  }
517  CheckNoPendingIo();
518 
519  CommandDebugInfo info;
520  Timer timer;
521  double total_elapsed_previous = 0.0;
522 
523  for (; program_counter_ < num_commands; program_counter_++) {
524  if (c[program_counter_].command_type == kAcceptInput ||
525  c[program_counter_].command_type == kProvideOutput) {
526  // We have hit a part of the computation that requires user
527  // interaction, e.g. the end of the forward or backward phase.
528  break;
529  }
530  if (debug_)
531  DebugBeforeExecute(program_counter_, &info);
532  ExecuteCommand();
533  if (debug_) {
534  double total_elapsed_now = timer.Elapsed();
535  DebugAfterExecute(program_counter_, info,
536  total_elapsed_now - total_elapsed_previous);
537  total_elapsed_previous = total_elapsed_now;
538  }
539  }
540 }

542 void NnetComputer::AcceptInput(const std::string &node_name,
543  CuMatrix<BaseFloat> *input) {
544  bool is_output = false;
545  int32 matrix_index = GetIoMatrixIndex(node_name, is_output);
546 
547  const NnetComputation::MatrixInfo &matrix_info =
548  computation_.matrices[matrix_index];
549  if (input->NumRows() != matrix_info.num_rows) {
550  KALDI_ERR << "Num-rows mismatch for input '" << node_name
551  << "': " << matrix_info.num_rows
552  << " in computation-request, " << input->NumRows()
553  << " provided.";
554  }
555  if (input->NumCols() != matrix_info.num_cols) {
556  KALDI_ERR << "Num-cols mismatch for input '" << node_name
557  << "': " << matrix_info.num_cols
558  << " in computation-request, " << input->NumCols()
559  << " provided.";
560  }
561  if (matrix_info.stride_type == kDefaultStride ||
562  input->Stride() == input->NumCols()) {
563  matrices_[matrix_index].Swap(input);
564  } else {
565  matrices_[matrix_index].Resize(matrix_info.num_rows,
566  matrix_info.num_cols,
567  kUndefined, kStrideEqualNumCols);
568  matrices_[matrix_index].CopyFromMat(*input);
569  input->Resize(0, 0);
570  }
571 }

573 const CuMatrixBase<BaseFloat> &NnetComputer::GetOutput(
574  const std::string &node_name) {

592 void NnetComputer::CheckNoPendingIo() {
593  const std::vector<NnetComputation::Command> &c = computation_.commands;
594  while (program_counter_ < static_cast<int32>(c.size()) &&
595  (c[program_counter_].command_type == kAcceptInput ||
596  c[program_counter_].command_type == kProvideOutput)) {
597  pending_commands_.push_back(program_counter_);
598  program_counter_++;
599  }
600  for (size_t i = 0; i < pending_commands_.size(); i++) {
601  // the order here doesn't really matter; we go from back to front
602  // as it's more efficient, not that efficiency really matters here.
603  int32 command = pending_commands_[i];
604  if (c[command].command_type == kAcceptInput) {
605  // we can't ignore if we needed input from the user that hasn't been
606  // provided.
607  int32 node = c[command].arg2;
608  KALDI_ERR << "Cannot run computation-- we did not get input for node '"
609  << nnet_.GetNodeName(node) << "'";
610  }
611  }
612  pending_commands_.clear();
613 }

615 int32 NnetComputer::GetIoMatrixIndex(const std::string &node_name, bool is_output) {
616  const std::vector<NnetComputation::Command> &c = computation_.commands;
617  int32 node_index = nnet_.GetNodeIndex(node_name);
618  if (node_index == -1)
619  KALDI_ERR << "No node named '" << node_name << "'in network.";
620  // first make sure all the I/O commands that we immediately expect, are listed
621  // in 'pending_commands_'.
622  while (program_counter_ < static_cast<int32>(computation_.commands.size()) &&
623  ((c[program_counter_].command_type == kAcceptInput ||
624  c[program_counter_].command_type == kProvideOutput ||
625  c[program_counter_].command_type == kNoOperationMarker))) {
626  if (c[program_counter_].command_type != kNoOperationMarker)
627  pending_commands_.push_back(program_counter_);
628  program_counter_++;
629  }
630  for (size_t i = 0; i < pending_commands_.size(); i++) {
631  int32 command = pending_commands_[i];
632  bool this_command_is_output =
633  (c[command].command_type == kProvideOutput);
634  int32 this_submatrix_index = c[command].arg1,
635  this_node_index = c[command].arg2;
636  if (this_command_is_output == is_output && node_index == this_node_index) {
637  if (!is_output) {
638  pending_commands_.erase(pending_commands_.begin() + i);
639  // don't erase the command for outputs, as that would prevent things
640  // from being output twice, which is an unnecessary restriction.
641  }
642  if (!(computation_.IsWholeMatrix(this_submatrix_index)))
643  KALDI_ERR << "Getting input or output that is not a whole matrix "
644  << "(probably some optimization code needs to be changed)";
645  return computation_.submatrices[this_submatrix_index].matrix_index;
646  }
647  }
648  // if you get the following error it will likely be a bug in the calling code,
649  // or possibly due to giving the wrong egs.
650  KALDI_ERR << "Could not "
651  << (is_output ? "provide output " : "accept input ")
652  << "for network node " << node_name
653  << " (it is not expected at this point in the computation)";
654  return 0; // Suppress compiler warnings; this line will never be reached.
655 }
