310 class CuSubVector: public CuVectorBase<Real> {
311  public:
312  CuSubVector(const CuVectorBase<Real> &t, const MatrixIndexT origin,
313  const MatrixIndexT length) : CuVectorBase<Real>() {
314  KALDI_ASSERT(static_cast<UnsignedMatrixIndexT>(origin)+
315  static_cast<UnsignedMatrixIndexT>(length) <=
316  static_cast<UnsignedMatrixIndexT>(t.Dim()));
317  CuVectorBase<Real>::data_ = const_cast<Real*>(t.Data()+origin);
318  CuVectorBase<Real>::dim_ = length;
319  }
322  CuSubVector(const CuSubVector &other) : CuVectorBase<Real> () {
323  CuVectorBase<Real>::data_ = other.data_;
324  CuVectorBase<Real>::dim_ = other.dim_;
325  }
326 
327  CuSubVector(const Real* data, MatrixIndexT length) : CuVectorBase<Real> () {
328  // Yes, we're evading C's restrictions on const here, and yes, it can be used
329  // to do wrong stuff; unfortunately the workaround would be very difficult.
330  CuVectorBase<Real>::data_ = const_cast<Real*>(data);
331  CuVectorBase<Real>::dim_ = length;
332  }
333 
335  CuSubVector(const CuMatrixBase<Real> &matrix, MatrixIndexT row) {
336  CuVectorBase<Real>::data_ = const_cast<Real*>(matrix.RowData(row));
337  CuVectorBase<Real>::dim_ = matrix.NumCols();
338  }
