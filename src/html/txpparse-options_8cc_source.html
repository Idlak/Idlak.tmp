54 TxpParseOptions::TxpParseOptions(const char *usage)
55  : ParseOptions(usage) {
56  std::istringstream is(txpconfigdefault);
57  std::string line, key, value, *stringptr;
58  RegisterStandard("tpdb", &tpdb_,
59  "Text processing database (directory XML language/speaker files)"); //NOLINT
60  while (std::getline(is, line)) {
61  // trim out the comments
62  size_t pos;
63  if ((pos = line.find_first_of('#')) != std::string::npos) {
64  line.erase(pos);
65  }
66  // skip empty lines
67  Trim(&line);
68  if (line.length() == 0) continue;
69 
70  // parse option
71  bool has_equal_sign;
72  SplitLongArg(line, &key, &value, &has_equal_sign);
73  NormalizeArgName(&key);
74  Trim(&value);
75  stringptr = new std::string(value);
76  txpoptions_.insert(LookupItemPtr(key, stringptr));
77  Register(key.c_str(), stringptr,
78  "Idlak Text Processing Option");
79  }
80 }

89 int TxpParseOptions::Read(int argc, const char *const *argv) {
90  std::string key, value;
91  int i;
92  // first pass: look for tpdb parameter
93  for (i = 1; i < argc; i++) {
94  if (std::strncmp(argv[i], "--", 2) == 0) {
95  bool has_equal_sign;
96  SplitLongArg(argv[i], &key, &value, &has_equal_sign);
97  NormalizeArgName(&key);
98  Trim(&value);
99  if (key.compare("tpdb") == 0) {
100  // check for tpdb configuration file
101  std::ifstream is((value + "/default.conf").c_str(), std::ifstream::in);
102  if (is.good()) {
103  is.close();
104  ReadConfigFile(value + "/default.conf");
105  }
106  }
107  if (key.compare("help") == 0) {
108  PrintUsage();
109  exit(0);
110  }
111  }
112  }
113  return ParseOptions::Read(argc, argv);
114 }

117 const char* TxpParseOptions::GetValue(const char* module, const char* key) const {
118  LookupMapPtr::const_iterator lookup;
119  std::string optkey(module);
120  optkey = optkey + "-" + key;
121  lookup = txpoptions_.find(optkey);
122  if (lookup == txpoptions_.end()) {
123  // if general section also look for field without section
124  // description
125  if (!strcmp(module, "general")) {
126  optkey = key;
127  lookup = txpoptions_.find(optkey);
128  if (lookup == txpoptions_.end()) return NULL;
129  else return (lookup->second)->c_str();
130  }
131  }
132  return (lookup->second)->c_str();
133 }
