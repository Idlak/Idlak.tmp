39  // Initialize the generator,
40  CURAND_SAFE_CALL(curandCreateGenerator(&gen_, CURAND_RNG_PSEUDO_DEFAULT));
41  // To get same random sequence, call srand() before the constructor is invoked,
42  CURAND_SAFE_CALL(curandSetGeneratorOrdering(gen_, CURAND_ORDERING_PSEUDO_DEFAULT));
43  CURAND_SAFE_CALL(curandSetPseudoRandomGeneratorSeed(gen_, RandInt(128, RAND_MAX)));
44  CURAND_SAFE_CALL(curandSetGeneratorOffset(gen_, 0));
