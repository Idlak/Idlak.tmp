60 bool TxpPhrasing::ProcessFile(pugi::xml_node* rootnode) {
61  bool final = false;
62  pugi::xml_node uttnode, phrasenode, lastbreak, firstbreak;
63  int32 phraseid = 1;
64  int32 uttid = 1;
65  int32 wordid = 1;
66  pugi::xpath_node_set breaks;
67  // Handle empty spurt
68  breaks = rootnode->select_nodes(".//break");
69  if (breaks.size() == 0) return true;
70 
71  uttnode = rootnode->append_child("utt");
72  uttnode.append_attribute("uttid").set_value(uttid);
73 
74  lastbreak = rootnode->select_nodes("(.//break)[last()]").first().node();
75  firstbreak = rootnode->select_nodes("(.//break)[1]").first().node();
76  while (!final) {
77  phrasenode = uttnode.append_child("spt");
78  phrasenode.append_attribute("phraseid").set_value(phraseid);
79  wordid = 1;
80  final = _copy_until_break(rootnode, &phrasenode,
81  &firstbreak, &lastbreak, &wordid);
82  firstbreak = rootnode->select_nodes(".//break[1]").first().node();
83  phrasenode.append_attribute("no_wrds").set_value(wordid - 1);
84  if (_is_utt_final(phrasenode)) {
85  uttnode.append_attribute("no_phrases").set_value(phraseid);
86  phraseid = 0;
87  if (!final) {
88  uttnode = rootnode->append_child("utt");
89  uttid++;
90  uttnode.append_attribute("uttid").set_value(uttid);
91  }
92  }
93  phraseid++;
94  }
95  return true;
96 }

98 static bool _copy_until_break(pugi::xml_node* parent,
99  pugi::xml_node* phrasenode,
100  pugi::xml_node* firstbreak,
101  pugi::xml_node* lastbreak,
102  int32* wordid) {
103  pugi::xml_node child, nextchild, childcopy;
104  for (child = parent->first_child(); child; child = nextchild) {
105  // get next child before current child is deleted
106  nextchild = child.next_sibling();
107  if (child.type() != pugi::node_element) {
108  phrasenode->append_copy(child);
109  parent->remove_child(child);
110  } else {
111  if (!strcmp(child.name(), "utt")) {
112  continue;
113  } else if (!strcmp(child.name(), "break")) {
114  if (child == *firstbreak) {
115  childcopy = phrasenode->append_copy(child);
116  parent->remove_child(child);
117  } else {
118  childcopy = phrasenode->append_copy(child);
119  // Split the break
120  if (child != *lastbreak) {
121  childcopy.attribute("time").set_value(
122  child.attribute("time").as_float() / 2.0f);
123  child.attribute("time").set_value(
124  childcopy.attribute("time").as_float());
125  return false;
126  } else {
127  parent->remove_child(child);
128  }
129  }
130  } else if (!strcmp(child.name(), "tk") || !strcmp(child.name(), "ws")) {
131  if (!strcmp(child.name(), "tk")) {
132  child.append_attribute("wordid").set_value(*wordid);
133  *wordid = *wordid + 1;
134  }
135  // std::cout << child.attribute("norm").value() << "\n";
136  childcopy = phrasenode->append_copy(child);
137  parent->remove_child(child);
138  } else {
139  childcopy = phrasenode->append_child(child.name());
140  for (pugi::xml_attribute a = child.first_attribute();
141  a;
142  a = a.next_attribute()) {
143  childcopy.append_attribute(a.name()).set_value(a.value());
144  }
145  if (_copy_until_break(&child, &childcopy,
146  firstbreak, lastbreak, wordid)) {
147  parent->remove_child(child);
148  } else {
149  return false;
150  }
151  }
152  }
153  }
154  return true;
155 }

157 static bool _is_utt_final(const pugi::xml_node &spt) {
158  pugi::xml_node lastbreak;
159  pugi::xpath_node_set breaks;
160  breaks = spt.select_nodes(".//break");
161  breaks.sort();
162  lastbreak = breaks[breaks.size() - 1].node();
163  // lastbreak.print(std::cout);
164  // std::cout << "!" << lastbreak.attribute("type").value() << "\n";
165  if (lastbreak.attribute("type").as_int(0) == 4) return true;
166  return false;
167 }
