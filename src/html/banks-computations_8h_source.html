87 class FrequencyBanks {
88  public:
89  static inline BaseFloat Scale(BaseFloat freq, std::string scale_type) {
90  if (scale_type == "bark") {
91  return 26.81f / (1960.0f / freq + 1.0) - 0.53;
92  } else if (scale_type == "mel") {
93  return 1127.0f * logf (1.0f + freq / 700.0f);
94  } else if (scale_type == "linear") {
95  return freq;
96  } else {
97  KALDI_ERR << "Invalid scale type " << scale_type;
98  return 0.0;
99  }
100  }
101 
102  static inline BaseFloat InverseScale(BaseFloat freq, std::string scale_type) {
103  if (scale_type == "bark") {
104  return 1960.0f / (26.81f / (freq + 0.53f) - 1.0f);
105  } else if (scale_type == "mel") {
106  return 700.0f * (expf (freq / 1127.0f) - 1.0f);
107  } else if (scale_type == "linear") {
108  return freq;
109  } else {
110  KALDI_ERR << "Invalid scale type " << scale_type;
111  return 0.0;
112  }
113  }
