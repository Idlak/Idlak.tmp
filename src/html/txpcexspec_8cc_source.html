24 // parse file into Cexspec class adding feature specification and
25 // feature functions to architecture
26 void TxpCexspec::StartElement(const char* name, const char** atts) {
27  std::string att, att2;
28  StringSet* set;
29  TxpCexspecFeat feat;
30  int32 featidx;
31 
32  // add features and other definitions
33  // top level behaviour
34  if (!strcmp(name, "cex")) {
35  SetAttribute("maxfieldlen", atts, &att);
36  if (!att.empty()) {
37  cexspec_maxfieldlen_ = atoi(att.c_str());
38  }
39  SetAttribute("pausehandling", atts, &att);
40  if (!att.empty()) {
41  if (!strcmp(att.c_str(), "UTT")) {
42  pauhand_ = CEXSPECPAU_HANDLER_UTTERANCE;
43  }
44  }
45  // sets for string based feature functions
46  } else if (!strcmp(name, "set")) {
47  SetAttribute("name", atts, &att);
48  if (att.empty()) {
49  // throw an error
50  KALDI_WARN << "Badly formed cex set: "
51  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
52  << " Col:" << XML_GetCurrentColumnNumber(parser_);
53  return;
54  }
55  curset_ = att;
56  SetAttribute("null", atts, &att);
57  if (!att2.empty()) {
58  att = CEXSPEC_NULL;
59  }
60  sets_.insert(LookupMapSetItem(curset_, StringSet()));
61  setnull_.insert(LookupItem(curset_, att));
62  // features
63  } else if (!strcmp(name, "item")) {
64  SetAttribute("name", atts, &att);
65  if (att.empty()) {
66  // throw an error
67  KALDI_WARN << "Badly formed cex set item: "
68  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
69  << " Col:" << XML_GetCurrentColumnNumber(parser_);
70  return;
71  }
72  set = &(sets_.find(curset_)->second);
73  set->insert(att);
74  } else if (!strcmp(name, "feat")) {
75  SetAttribute("name", atts, &att);
76  feat.name = att;
77  if (att.empty()) {
78  // throw an error
79  KALDI_WARN << "Badly formed cex set item: "
80  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
81  << " Col:" << XML_GetCurrentColumnNumber(parser_);
82  return;
83  }
84  featidx = GetFeatureIndex(att);
85  if (featidx == NO_INDEX) {
86  curfunc_ = "";
87  // throw an error
88  KALDI_WARN << "Missing feature in architecture: " << att
89  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
90  << " Col:" << XML_GetCurrentColumnNumber(parser_);
91  return;
92  }
93  // valid function try to add rest of the features specification
94  curfunc_ = att;
95  SetAttribute("htsname", atts, &att);
96  feat.htsname = att;
97  SetAttribute("desc", atts, &att);
98  feat.desc = att;
99  SetAttribute("delim", atts, &att);
100  if (att.empty() && cexspecfeats_.size()) {
101  // throw an error
102  KALDI_WARN << "Missing delimiter for feature architecture: " << curfunc_
103  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
104  << " Col:" << XML_GetCurrentColumnNumber(parser_);
105  return;
106  }
107  feat.delim = att;
108  feat.func = CEXFUNC[featidx];
109  feat.pause_context = false;
110  SetAttribute("pauctx", atts, &att);
111  if (att == "true" || att == "True" || att == "TRUE") {
112  feat.pause_context = true;
113  }
114  SetAttribute("pauctx", atts, &att2);
115  feat.type = CEXFUNCTYPE[featidx];
116  if (feat.type == CEXSPEC_TYPE_STR) {
117  SetAttribute("set", atts, &att);
118  if (att.empty()) {
119  // throw an error
120  KALDI_WARN << "Missing set name for string feature architecture: "
121  << curfunc_
122  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
123  << " Col:" << XML_GetCurrentColumnNumber(parser_);
124  return;
125  }
126  // check set has been added
127  if (sets_.find(att) == sets_.end()) {
128  // throw an error
129  KALDI_WARN << "Missing set for string feature architecture: "
130  << curfunc_
131  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
132  << " Col:" << XML_GetCurrentColumnNumber(parser_)
133  << " (must define before function)";
134  return;
135  }
136  feat.set = att;
137  feat.nullvalue = setnull_.find(att)->second;
138  } else if (feat.type == CEXSPEC_TYPE_INT) {
139  SetAttribute("min", atts, &att);
140  if (!att.empty()) feat.min = atoi(att.c_str());
141  SetAttribute("max", atts, &att);
142  if (att.empty()) {
143  // throw an error
144  KALDI_WARN << "Missing maximum value for integer feature architecture: "
145  << curfunc_
146  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
147  << " Col:" << XML_GetCurrentColumnNumber(parser_);
148  return;
149  }
150  feat.max = atoi(att.c_str());
151  }
152  // if we have got to here the feature is valid add it to the architecture
153  cexspecfeatlkp_.insert(LookupIntItem(curfunc_, cexspecfeats_.size()));
154  cexspecfeats_.push_back(feat);
155  } else if (!strcmp(name, "mapping")) {
156  SetAttribute("fromstr", atts, &att);
157  SetAttribute("tostr", atts, &att2);
158  if (att.empty() || att2.empty()) {
159  // throw an error
160  KALDI_WARN << "bad mapping item: " << curfunc_
161  << " Line: " << ::XML_GetCurrentLineNumber(parser_)
162  << " Col:" << XML_GetCurrentColumnNumber(parser_);
163  return;
164  }
165  if (!curfunc_.empty()) {
166  LookupInt::iterator iter;
167  iter = cexspecfeatlkp_.find(curfunc_);
168  if (iter != cexspecfeatlkp_.end()) {
169  cexspecfeats_[iter->second].mapping.insert(LookupItem(att, att2));
170  }
171  }
172  }
173 }

199 // Add pause tk, syl, phon to break tags and information
200 // on pause type
201 // This makes interation for feature extraction more
202 // homogeneous
203 int32 TxpCexspec::AddPauseNodes(pugi::xml_document* doc) {
204  bool uttbreak, endbreak, internalbreak;
205  pugi::xml_node node, childnode;
206  pugi::xpath_node_set nodes =
207  doc->document_element().select_nodes("//phon|//break");
208  nodes.sort();
209  for (pugi::xpath_node_set::const_iterator it = nodes.begin();
210  it != nodes.end();
211  ++it) {
212  node = it->node();
213  if (!strcmp(node.name(), "break")) {
214  // determine break type
215  if (!strcmp(node.attribute("type").value(), "4")) {
216  uttbreak = true;
217  } else {
218  uttbreak = false;
219  }
220  if (it == nodes.begin()) {
221  endbreak = false;
222  internalbreak = false;
223  } else if (it + 1 == nodes.end()) {
224  endbreak = true;
225  internalbreak = false;
226  } else if (!strcmp((it - 1)->node().name(), "break")) {
227  endbreak = false;
228  internalbreak = true;
229  } else {
230  endbreak = true;
231  internalbreak = true;
232  }
233  if (pauhand_ == CEXSPECPAU_HANDLER_UTTERANCE && !uttbreak &&
234  internalbreak && !endbreak) continue;
235  childnode = node.append_child("tk");
236  childnode.append_attribute("pron").set_value("pau");
237  childnode = childnode.append_child("syl");
238  childnode.append_attribute("val").set_value("pau");
239  childnode = childnode.append_child("phon");
240  childnode.append_attribute("val").set_value("pau");
241  }
242  }
243  return true;
244 }

283  int32 i, std::string* buf) const {
284  std::stringstream stream;

290  if (i > feat.max) i = feat.max;
291  stream << i;
292  buf->append(feat.delim);
293  buf->append(Mapping(feat, stream.str()));

365 bool TxpCexspecContext::Next() {
366  pugi::xml_node node, phon, empty;
367 
368  // std::cout << (current_phone_->node()).attribute("val").value() << ":"
369  // << (current_syllable_->node()).attribute("val").value() << ":"
370  // << (current_word_->node()).attribute("norm").value() << ":"

372  // << (current_utterance_->node()).attribute("uttid").value() << " "
373  // << (current_phone_->node()).attribute("type").value() << "\n";
374  // iterate over phone/break items
375  current_phone_++;
376  if (current_phone_ == phones_.end()) return false;
377  phon = current_phone_->node();
378  // update other iterators as required
379  // dummy pau items already added for tk and syl levels
380  node = GetContextUp(phon, "syl");
381  while (node != current_syllable_->node()) current_syllable_++;
382  node = GetContextUp(phon, "tk");
383  while (node != current_word_->node()) current_word_++;
384  node = GetContextUp(phon, "spt");
385  while (node != current_spurt_->node()) current_spurt_++;
386  node = GetContextUp(phon, "utt");
387  while (node != current_utterance_->node()) current_utterance_++;
388 
389  return true;
390 }

421  pau_found++;
422  }
423  }
424  if ((current_phone_ + i) == phones_.end()) return empty;
425  if (pau_found == 2 && !pause_context) return empty;
426  return (current_phone_ + i)->node();
