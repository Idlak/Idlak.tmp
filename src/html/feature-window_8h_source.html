35 struct FrameExtractionOptions {
36  BaseFloat samp_freq;
37  BaseFloat frame_shift_ms; // in milliseconds.
38  BaseFloat frame_length_ms; // in milliseconds.
39  BaseFloat dither; // Amount of dithering, 0.0 means no dither.
40  BaseFloat preemph_coeff; // Preemphasis coefficient.
41  bool remove_dc_offset; // Subtract mean of wave before FFT.
42  std::string window_type; // e.g. Hamming window
43  bool round_to_power_of_two;
44  BaseFloat blackman_coeff;
45  bool snip_edges;
46  bool allow_downsample;
47  // May be "hamming", "rectangular", "povey", "hanning", "blackman"
48  // "povey" is a window I made to be similar to Hamming but to go to zero at the
49  // edges, it's pow((0.5 - 0.5*cos(n/N*2*pi)), 0.85)
50  // I just don't think the Hamming window makes sense as a windowing function.
51  FrameExtractionOptions():
52  samp_freq(16000),
53  frame_shift_ms(10.0),
54  frame_length_ms(25.0),
55  dither(1.0),
56  preemph_coeff(0.97),
57  remove_dc_offset(true),
58  window_type("povey"),
59  round_to_power_of_two(true),
60  blackman_coeff(0.42),
61  snip_edges(true),
62  allow_downsample(false) { }

91  "If true, allow the input waveform to have a higher frequency than "
92  "the specified --sample-frequency (and we'll downsample).");
93  }
94  int32 WindowShift() const {
95  return static_cast<int32>(samp_freq * 0.001 * frame_shift_ms);
96  }
97  int32 WindowSize() const {
98  return static_cast<int32>(samp_freq * 0.001 * frame_length_ms);
99  }
100  int32 PaddedWindowSize() const {
101  return (round_to_power_of_two ? RoundUpToNearestPowerOfTwo(WindowSize()) :
102  WindowSize());
103  }

174 /*
175  ExtractWindow() extracts a windowed frame of waveform (possibly with a
176  power-of-two, padded size, depending on the config), including all the
177  proessing done by ProcessWindow().
178 
179  @param [in] sample_offset If 'wave' is not the entire waveform, but
180  part of it to the left has been discarded, then the
181  number of samples prior to 'wave' that we have
182  already discarded. Set this to zero if you are
183  processing the entire waveform in one piece, or
184  if you get 'no matching function' compilation
185  errors when updating the code.
186  @param [in] wave The waveform
187  @param [in] f The frame index to be extracted, with
188  0 <= f < NumFrames(sample_offset + wave.Dim(), opts, true)
189  @param [in] opts The options class to be used
190  @param [in] window_function The windowing function, as derived from the
191  options class.
192  @param [out] window The windowed, possibly-padded waveform to be
193  extracted. Will be resized as needed.
194  @param [out] log_energy_pre_window If non-NULL, the log-energy of
195  the signal prior to pre-emphasis and multiplying by
196  the windowing function will be written to here.
197 */
