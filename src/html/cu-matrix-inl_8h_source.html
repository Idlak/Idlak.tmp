27 template<typename Real>
28 inline CuSubMatrix<Real>::CuSubMatrix(const CuMatrixBase<Real> &mat,
29  const MatrixIndexT row_offset,
30  const MatrixIndexT num_rows,
31  const MatrixIndexT col_offset,
32  const MatrixIndexT num_cols) {
33  if (num_rows == 0 || num_cols == 0) {
34  KALDI_ASSERT(num_rows == 0 && num_cols == 0);
35  // Everything will have been set to zero in CuMastrixBase's default
36  // initializer, so nothing to do.
37  } else {
38  KALDI_ASSERT(row_offset >= 0 && col_offset >= 0 &&
39  row_offset + num_rows <= mat.num_rows_ &&
40  col_offset + num_cols <= mat.num_cols_);
41  this->data_ = mat.data_ + static_cast<size_t>(col_offset) +
42  static_cast<size_t>(row_offset) * static_cast<size_t>(mat.stride_);
43  this->num_cols_ = num_cols;
44  this->num_rows_ = num_rows;
45  this->stride_ = mat.stride_;
46  }
47 }

50 inline CuSubMatrix<Real>::CuSubMatrix(const Real *data,
51  const MatrixIndexT num_rows,
52  const MatrixIndexT num_cols,
53  const MatrixIndexT stride):
54  CuMatrixBase<Real>(const_cast<Real*>(data), num_rows, num_cols, stride) {
55  // in general if you use SubMatrix or CuSubMatrix, const-correctness is not
56  // preserved (preserving it would require us duplicating the class and it
57  // would have been a hassle).
58 
59  // Note: we used to check that stride >= num_cols. We no longer check for
60  // this as there are some situations where having stride < num_cols is useful,
61  // but beware because most if not all CUBLAS calls will crash when given
62  // such an input, even in a situation where it makes sense.
63  KALDI_ASSERT((num_rows != 0) == (num_cols != 0) && stride >= 0 &&
64  num_rows >= 0 && num_cols >= 0 && stride >= 0);
65 }
