122  // Does the initial computation (populating the graph and computing whether
123  // each required cindex_id is computable), without the pruning. In the normal
124  // case you call this just once with one 'request', but in the 'online' case
125  // you call Compute() [then maybe check AllOutputsAreComputable()] then

140 
141  // This function outputs to "computable" information about whether each
142  // requested element of each output was computable. "computable" will have
143  // the same size as request_->outputs, and each element will have the same
144  // size as request_->outputs[i].indexes.size(). May only be called after
145  // Compute() but before Prune(). If you have already called Prune(), you can
146  // just assume everything was computable, or else Prune() would have crashed.
147  void GetComputableInfo(std::vector<std::vector<bool> > *computable) const;
148 
149  // to be called after Compute(), this prunes away unused cindex_ids.
150  // If not all the outputs are computable, this will die;
151  // you can check the return status of AllOutputsAreComputable() first if
152  // you want to avoid this.
153  void Prune();
154 
155  // This enum says for each cindex_id, whether we can compute it from the given
156  // inputs or not. Note that there may be situations where before adding
157  // dependencies of a particular cindex_id we realize that we won't be able to
158  // use this cindex_id (i.e. it may be computable but it's not used) because
159  // its usable_count is zero, and in those cases we change the status to
160  // kWillNotCompute even though the cindex-id may be computable- for most
161  // purposes this status is treated the same as kNotComputable.
162  enum ComputableInfo {
163  kUnknown = 0,
164  kComputable = 1,
165  kNotComputable = 2,
166  kWillNotCompute = 3
167  };
168  private:
169  // This function, called from ExplainWhyNotComputable(), prints to "os"
170  // a human-readable form of a given cindex_id, that looks like
171  // some_network_node(n, t, x), e.g. "final_logsoftmax(0, -4, 0)".
172  void PrintCindexId(std::ostream &os, int32 cindex_id) const;

174  // This function, typically to be called just before dying, prints logging

176  void ExplainWhyNotComputable(int32 cindex_id) const;

209  // To be called when this cindex_id has just been newly added to graph_, this
210  // function adds various initial variables associated with it, to *this.
211  // is_input should be set to true if this cindex-id is being added as an input
212  // (from request_.inputs), and is_output should be set to true if this

220  void IncrementUsableCount(int32 cindex_id);

223  void DecrementUsableCount(int32 cindex_id);

225  // This function, called from Prune(), modifies the members of
226  // graph_->dependencies-- it removes those cindexes that are not used in the
227  // computation for the current cindex_id. This will only do something
228  // interesting in cases where there are optional dependencies.
229  // It also clears the dependencies of those cindexes that are not computable.
230  void PruneDependencies(int32 cindex_id);
231 
232  // This function, called from Prune(), computes an array "required", with an
233  // element for each cindex_id that says whether it is required to compute the
234  // requested outputs. This is similar in function to the "usable_count_"
235  // array, but it's more exact because it's computed after we have done
236  // PruneDependencies() to remove unused dependencies, so it will only say
237  // something is required if it is really accessed in the computation.
238  // We'll later use this to remove unnecessary cindexes.
239  // 'start_cindex_id' is the cindex_id from which the 'required' array is
240  // to start (normally zero, but may be nonzero in multi-segment computations);
241  // so 'required' is indexed by cindex_id - start_cindex_id.
242  void ComputeRequiredArray(int32 start_cindex_id,
243  std::vector<bool> *required) const;
244 
245  // this function, to be called from Compute(), does some sanity checks to
246  // verify that the internal state is consistent. It only does this for the
247  // current 'segment' of the computation, starting from 'start_cindex_id' (this
248  // will be 0 in normal, single-segment computations).
249  void Check(int32 start_cindex_id) const;

251  const Nnet &nnet_;
252  const ComputationRequest *request_;
253  ComputationGraph *graph_;
254 
255  // this is the transpose of graph_->dependencies; it tells us
256  // for each cindex_id, which other cindex_ids depend on it.
257  std::vector<std::vector<int32> > depend_on_this_;
258 
259  // this vector, indexed by cindex_id, contains our information about whether
260  // each cindex_id is computable; it's ComputableInfo, cast to char.
261  std::vector<char> computable_info_;
262 
263  // this is a queue of cindex_ids that we need to re-compute whether they are
264  // computable or not (because either they are new and haven't had dependencies
265  // added, or their dependencies' computable status has changed since we last
266  // computed their computable_ value).
267  std::deque<int32> computable_queue_;
268  // this vector tells us whether a cindex_id is in computable_queued_; it
269  // stops us from adding things twice.
270  std::vector<bool> computable_queued_;
271 
272  // usable_count_[i] for a cindex_id i is defined as 1 if i is a requested
273  // output, and otherwise as the number of other cindex_ids j such that
274  // computable_info_[j] is not kNotComputable AND usable_count_[j] > 0 AND i is
275  // a member of graph->dependencies[j]. A cindex_id is termed "usable"
276  // (meaning it could potentially participate in the computation of the output)
277  // if its usable_count_ is > 0. This quantity is designed to be easy to keep
278  // updated as we add cindex_ids.
279  std::vector<int32> usable_count_;
280 
281  // current_distance_ >= 0 is the distance to the output, of the cindex_ids in
282  // current_queue_.
283  int32 current_distance_;
284  // the cindex_ids in current_queue_ are at distance "current_distance" to the
285  // output and have not yet had their dependencies processed.
286  std::vector<int32> current_queue_;
287  // the cindex_ids in next_queue_ are at distance current_distance + 1 to the
288  // output and have not yet had their dependencies processed.
289  std::vector<int32> next_queue_;

461  // Adds step(s) for one "sub-phase". A sub-phase is the set of cindex_ids from
462  // one phase that have the same node index. Note: for nodes that are
463  // component-input descriptors, we don't actually create the step here, we
464  // create it just before creating the step for its component, and we recreate
465  // the list of cindexes from those from the component. The reason is that
466  // there are situations where doing it directly from the raw_step would not do
467  // the right thing (especially with non-simple components, it's possible that
468  // the cindexes component-input descriptors could be used twice by two
469  // different components)..

471  const std::vector<Cindex> &sub_phase);

477  void ProcessInputOrOutputStep(const ComputationRequest &request,
478  bool is_output,

480 
481  // Called from ProcessSubPhase- for the case where it's a component node.
482  void ProcessComponentStep(const std::vector<Cindex> &step);
483 
484 
485  // Splits a phase up into multiple "sub-phases", which are just the cindexes
486  // from a phase that are from a single node, sorted. At this point we
487  // represent them as Cindexes, not cindex_ids. For efficiency and because it
488  // would be discarded anyway, it discards any raw steps that correspond to
489  // component-input descriptors because these are not processed inside
490  // ProcessSubPhase().

492  std::vector<std::vector<Cindex> > *sub_phase) const;
493 
494  // This low-level function used by functions like ProcessComponentStep,
