45 // For now we don't give the user a way to modify these from the command line.
46 // or the code, it just documents what the default options are. To change
47 // the options, you have to do it in the code.
48 struct CuAllocatorOptions {
49  // True if we are going to actually cache memory allocations on this device.
50  // You'd normally set it to false only if you wanted to debug a possible
51  // memory problem using cuda-memcheck or cuda-gdb. It will be slower, but
52  // using CUDA's native allocator allows those tools to detect out-of-region
53  // memory accesses.
54  bool cache_memory;
55 
56  // The proportion of the device's memory that the CuAllocator allocates to
57  // start with; by default this is 0.8, although if you want to share the
58  // device (not recommended!) you should set this lower.
59  BaseFloat memory_proportion;
60 
61  // The target number of subregions of the entire CUDA device memory (we'll
62  // start with a smaller number of memory_proportion is << 1). Kind of
63  // a tuning knob.. more regions will make it more aggressively consolidate
64  // memory low addresses.
65  int32 num_subregions;
66 
67  CuAllocatorOptions():
68  cache_memory(true), memory_proportion(0.5), num_subregions(20) { }
69 
70  void Register(OptionsItf *po) {
71  po->Register("cuda-cache-memory", &cache_memory, "True if you want "
72  "to use the caching allocator. Set this to false only if you "
73  "want to use cuda-memcheck or cuda-gdb; it will be slower.");
74  po->Register("cuda-memory-proportion", &memory_proportion,
75  "Proportion of the GPU device memory that the allocator "
76  "should allocate at the start");
77  }
78 
79  void Check() {
80  // don't let it get too close to 1;
81  KALDI_ASSERT(memory_proportion >= 0.05 && memory_proportion < 0.99);
82  }
83 };

226  // a MemoryRegion is a large piece of memory that we allocated via CudaMalloc.
227  // there normally won't be more than about 3 or 4 of these.
228  // We'll identify MemoryRegions by a size_t (e.g 0, 1, 2, 3... ) which is an
229  // index into the memory_regions_ vector.
230  struct MemoryRegion {
231  char *begin; // 'begin' is the start of the memory region.
232  char *end; // 'end' is the end of the memory region.
233  SubRegion *subregion_begin; // The first SubRegion that belongs to this
234  // MemoryRegion.
235  MemoryBlock *block_begin; // The first MemoryBlock that belongs to this
236  // MemoryRegion.
237  };
238 
239  // a SubRegion is a smaller zone of memory within a MemoryRegion. For
240  // example, we divide the first MemoryRegion we allocate into 10 blocks, and
241  // if we allocate blocks of memory later on, we'll sub-divide them into blocks
242  // of about the same size. A SubRegion is just a largish bin into which we
243  // put any blocks of memory that happen to start within that SubRegion;
244  // actually, memory blocks may cross over the boundaries of SubRegions. The
245  // motivation for dividing up MemoryRegions into SubRegions is that it allos
246  // us an efficient mechanism to segregate smaller memory blocks into higher
247  // memory and larger ones into lower memory: for each allocation, we allocate
248  // it from the highest-numbered SubRegion that is able to allocate something of
249  // that size. Over time, this will lead to smaller memory blocks being
250  // concentrated in higher-numbered SubRegions.
251  struct SubRegion {
252  size_t memory_region; // This is an index into the memory_regions_ vector
253  // which identifies which MemoryRegion this SubRegion
254  // is a part of.
255  size_t subregion_index; // The index of this SubRegion within the
256  // subregions_ vector; this can change when we
257  // allocate more MemoryRegions.
258  char *begin; // 'begin' is the start of the memory in this SubRegion.
259  char *end; // 'end' is the end of the memory in this SubRegion.
260 
261  // Contains the free MemoryBlocks starting within this SubRegion.
262  std::set<std::pair<size_t, MemoryBlock*> > free_blocks;
263 
264  // Pointer to the next SubRegion within this MemoryRegion (i.e. the SubRegion
265  // whose begin equals this one's end), or NULL if this is the last one.
266  SubRegion *next;
267  };
