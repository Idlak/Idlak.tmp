31 template<class Arc> void TestFactor() {
32  typedef typename Arc::Label Label;
33  typedef typename Arc::StateId StateId;
34  typedef typename Arc::Weight Weight;
35 
36  VectorFst<Arc> *fst = new VectorFst<Arc>();
37  int n_syms = 2 + kaldi::Rand() % 5, n_states = 3 + kaldi::Rand() % 10, n_arcs = 5 + kaldi::Rand() % 30, n_final = 1 + kaldi::Rand()%3; // Up to 2 unique symbols.
38  cout << "Testing pre-determinize with "<<n_syms<<" symbols, "<<n_states<<" states and "<<n_arcs<<" arcs and "<<n_final<<" final states.\n";
39  SymbolTable *sptr = NULL;
40 
41  vector<Label> all_syms; // including epsilon.
42  // Put symbols in the symbol table from 1..n_syms-1.
43  for (size_t i = 0;i < (size_t)n_syms;i++)
44  all_syms.push_back(i);
45 
46  // Create states.
47  vector<StateId> all_states;
48  for (size_t i = 0;i < (size_t)n_states;i++) {
49  StateId this_state = fst->AddState();
50  if (i == 0) fst->SetStart(i);
51  all_states.push_back(this_state);
52  }
53  // Set final states.
54  for (size_t j = 0;j < (size_t)n_final;j++) {
55  StateId id = all_states[kaldi::Rand() % n_states];
56  Weight weight = (Weight)(0.33*(kaldi::Rand() % 5) );
57  printf("calling SetFinal with %d and %f\n", id, weight.Value());
58  fst->SetFinal(id, weight);
59  }
60  // Create arcs.
61  for (size_t i = 0;i < (size_t)n_arcs;i++) {
62  Arc a;
63  a.nextstate = all_states[kaldi::Rand() % n_states];
64  a.ilabel = all_syms[kaldi::Rand() % n_syms];
65  a.olabel = all_syms[kaldi::Rand() % n_syms]; // same input+output vocab.
66  a.weight = (Weight) (0.33*(kaldi::Rand() % 2));
67  StateId start_state = all_states[kaldi::Rand() % n_states];
68  fst->AddArc(start_state, a);
69  }
70 
71  std::cout <<" printing before trimming\n";
72  {
73  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");
74  fstprinter.Print(&std::cout, "standard output");
75  }
76  // Trim resulting FST.
77  Connect(fst);
78 
79  std::cout <<" printing after trimming\n";
80  {
81  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");
82  fstprinter.Print(&std::cout, "standard output");
83  }
84 
85  vector<Label> extra_syms;
86  if (fst->Start() != kNoStateId) { // "Connect" did not make it empty....
87  PreDeterminize(fst, 1000, &extra_syms);
88  }
89 
90  std::cout <<" printing after predeterminization\n";
91  {
92  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");
93  fstprinter.Print(&std::cout, "standard output");
94  }
95 
96 
97  { // Remove epsilon. All default args.
98  bool connect = true;
99  Weight weight_threshold = Weight::Zero();
100  int64 nstate = -1; // Relates to pruning.
101  double delta = kDelta; // I think a small weight value. Relates to some kind of pruning,
102  // I guess. But with no epsilon cycles, probably doensn't matter.
103  RmEpsilon(fst, connect, weight_threshold, nstate, delta);
104  }

106  std::cout <<" printing after double-epsilon removal\n";

108  FstPrinter<Arc> fstprinter(*fst, sptr, sptr, NULL, false, true, "\t");

163 template<class Arc, class inttype> void TestStringRepository() {
164  typedef typename Arc::Label Label;
165 
166  StringRepository<Label, inttype> sr;
167 
168  int N = 1000;
169  if (sizeof(inttype) == 1) N = 64;
170  vector<vector<Label> > strings(N);
171  vector<inttype> ids(N);
172 
173  for (size_t i = 0;i < N;i++) {
174  size_t len = kaldi::Rand() % 4;
175  vector<Label> vec;
176  for (size_t j = 0;j < len;j++) vec.push_back( (kaldi::Rand()%10) + 150*(kaldi::Rand()%2)); // make it have reasonable range.
177  if (i < 500 && vec.size() == 0) ids[i] = sr.IdOfEmpty();
178  else if (i < 500 && vec.size() == 1) ids[i] = sr.IdOfLabel(vec[0]);
179  else ids[i] = sr.IdOfSeq(vec);
180 
181  strings[i] = vec;
182  }
183 
184  for (size_t i = 0;i < N;i++) {
185  vector<Label> tmpv;
186  tmpv.push_back(10); // just put in garbage.
187  sr.SeqOfId(ids[i], &tmpv);
188  assert(tmpv == strings[i]);
189  assert(sr.IdOfSeq(strings[i]) == ids[i]);
190  if (strings[i].size() == 0) assert(ids[i] == sr.IdOfEmpty());
191  if (strings[i].size() == 1) assert(ids[i] == sr.IdOfLabel(strings[i][0]));
192 
193  if (sizeof(inttype) != 1) {
194  size_t prefix_len = kaldi::Rand() % (strings[i].size() + 1);
195  inttype s2 = sr.RemovePrefix(ids[i], prefix_len);
196  vector<Label> vec2;
197  sr.SeqOfId(s2, &vec2);
198  for (size_t j = 0;j < strings[i].size()-prefix_len;j++) {
199  assert(vec2[j] == strings[i][j+prefix_len]);
200  }
201  }
202 
203  }
204 }
