45 //inline
46 void DistributeComponent::ComputeInputIndexAndBlock(const Index &output_index,
47  Index *input_index,
48  int32 *block) const {
49  int32 num_blocks = input_dim_ / output_dim_;
50  *input_index = output_index;
51  int32 output_x = output_index.x, input_x;
52  if (output_x >= 0) {
53  input_x = output_x / num_blocks;
54  } else {
55  input_x = (output_x - num_blocks + 1) / num_blocks;
56  }
57  input_index->x = input_x;
58  if (block)
59  *block = output_x - (input_x * num_blocks);
60 }

101 // virtual
102 ComponentPrecomputedIndexes* DistributeComponent::PrecomputeIndexes(
103  const MiscComputationInfo &, // misc_info
104  const std::vector<Index> &input_indexes,
105  const std::vector<Index> &output_indexes,
106  bool) const { // the bool is 'need_backprop'- unused.
107  unordered_map<Index, int32, IndexHasher> index_to_input_dim;
108  int32 num_input_indexes = input_indexes.size(),
109  num_output_indexes = output_indexes.size();
110  for (int32 i = 0; i < num_input_indexes; i++)
111  index_to_input_dim[input_indexes[i]] = i;
112  DistributeComponentPrecomputedIndexes *ans = new
113  DistributeComponentPrecomputedIndexes;
114  ans->pairs.resize(output_indexes.size());
115 
116  int32 num_blocks = input_dim_ / output_dim_,
117  block_size = input_dim_ / num_blocks;
118 
119  for (int32 i = 0; i < num_output_indexes; i++) {
120  Index input_index;
121  int32 block_index;
122  ComputeInputIndexAndBlock(output_indexes[i], &input_index, &block_index);
123  unordered_map<Index, int32, IndexHasher>::iterator iter =
124  index_to_input_dim.find(input_index);
125  if (iter == index_to_input_dim.end())
126  KALDI_ERR << "Input index not found (code error)";
127  int32 input_row = iter->second;
128  ans->pairs[i] = std::pair<int32,int32>(input_row, block_index * block_size);
129  }
130  return ans;
131 }

193 // virtual
194 void DistributeComponent::Backprop(const std::string &debug_info,
195  const ComponentPrecomputedIndexes *indexes,
196  const CuMatrixBase<BaseFloat> &, // in_value,
197  const CuMatrixBase<BaseFloat> &, // out_value
198  const CuMatrixBase<BaseFloat> &out_deriv,
199  void *memo,
200  Component *, // to_update,
201  CuMatrixBase<BaseFloat> *in_deriv) const {
202  if (in_deriv == NULL) return;
203 
204  int32 num_blocks = input_dim_ / output_dim_,
205  num_output_rows = out_deriv.NumRows();
206  if (num_output_rows != in_deriv->NumRows() * num_blocks) {
207  // there could be some 'gaps', i.e. some input values that are not ever
208  // referred to. So we need to zero the input. This would't happen in the
209  // setups I plan to use this for.
210  in_deriv->SetZero();
211  }
212 
213  std::vector<BaseFloat*> input_pointers;
214  ComputeInputPointers(indexes, num_output_rows, in_deriv, &input_pointers);
215  CuArray<BaseFloat*> input_pointers_cuda(input_pointers);
216  out_deriv.CopyToRows(input_pointers_cuda);
217 }

287 ComponentPrecomputedIndexes*
288 StatisticsExtractionComponent::PrecomputeIndexes(
289  const MiscComputationInfo &misc_info,
290  const std::vector<Index> &input_indexes,
291  const std::vector<Index> &output_indexes,
292  bool need_backprop) const {
293  int32 num_input_indexes = input_indexes.size(),
294  num_output_indexes = output_indexes.size();
295  StatisticsExtractionComponentPrecomputedIndexes *ans = new
296  StatisticsExtractionComponentPrecomputedIndexes();
297  // both input and output indexes are assumed sorted first on
298  // n and x, then on t.
299  Int32Pair invalid_pair;
300  invalid_pair.first = -1;
301  invalid_pair.second = -1;
302  std::vector<Int32Pair> forward_indexes_cpu(output_indexes.size(),
303  invalid_pair);
304  std::vector<int32> backward_indexes_cpu(input_indexes.size(), -1);
305  Vector<BaseFloat> counts_cpu(output_indexes.size());
306 
307  // this map maps from Index to the position in 'input_indexes'.
308  unordered_map<Index, int32, IndexHasher> index_to_input_pos;
309  for (int32 i = 0; i < num_input_indexes; i++)
310  index_to_input_pos[input_indexes[i]] = i;
311 
312  for (int32 i = 0; i < num_output_indexes; i++) {
313  Index output_index = output_indexes[i];
314  Index input_index(output_index);
315  int32 t = output_index.t,
316  t_start = output_period_ * (t / output_period_);
317  if (t_start > t) // could happen for negative t_start due to
318  t_start -= output_period_; // the way modulus works in C.
319  int32 t_end = t_start + output_period_;
320  for (int32 t = t_start; t < t_end; t += input_period_) {
321  input_index.t = t;
322  unordered_map<Index, int32, IndexHasher>::iterator iter =
323  index_to_input_pos.find(input_index);
324  if (iter != index_to_input_pos.end()) {
325  int32 input_pos = iter->second;
326  if (forward_indexes_cpu[i].first == -1) {
327  forward_indexes_cpu[i].first = input_pos;
328  forward_indexes_cpu[i].second = input_pos + 1;
329  counts_cpu(i) = 1.0;
330  } else {
331  // the following might fail, for instance, if the sorting
332  // of the input or output indexes was not as expected.
333  KALDI_ASSERT(forward_indexes_cpu[i].second == input_pos);
334  forward_indexes_cpu[i].second++;
335  counts_cpu(i) += 1.0;
336  }
337  KALDI_ASSERT(backward_indexes_cpu[input_pos] == -1);
338  backward_indexes_cpu[input_pos] = i;
339  }
340  }
341  KALDI_ASSERT(counts_cpu(i) != 0.0);
342  }
343  for (int32 i = 0; i < num_input_indexes; i++) {
344  KALDI_ASSERT(backward_indexes_cpu[i] != -1);
345  }
346  ans->forward_indexes = forward_indexes_cpu;
347  ans->counts = counts_cpu;
348  if (need_backprop)
349  ans->backward_indexes = backward_indexes_cpu;
350  return ans;
351 }

448  const ComponentPrecomputedIndexes *indexes_in,
449  const CuMatrixBase<BaseFloat> &in,
450  CuMatrixBase<BaseFloat> *out) const {
451  KALDI_ASSERT(indexes_in != NULL);

454  indexes_in);
455  int32 num_rows_out = out->NumRows();

465  if (include_variance_) {

467  CuMatrix<BaseFloat> in_squared(in);
468  in_squared.ApplyPow(2.0);
469  out->ColRange(input_dim_ + 1,
470  input_dim_).AddRowRanges(in_squared,
471  indexes->forward_indexes);
472  }
473  return NULL;
474 }

701 ComponentPrecomputedIndexes*
702 StatisticsPoolingComponent::PrecomputeIndexes(
703  const MiscComputationInfo &misc_info,
704  const std::vector<Index> &input_indexes,
705  const std::vector<Index> &output_indexes,
706  bool need_backprop) const {
707  int32 num_input_indexes = input_indexes.size(),
708  num_output_indexes = output_indexes.size();
709  StatisticsPoolingComponentPrecomputedIndexes *ans = new
710  StatisticsPoolingComponentPrecomputedIndexes();
711 
712  Int32Pair invalid_pair;
713  invalid_pair.first = -1;
714  invalid_pair.second = -1;
715  // forward_indexes_cpu[i] will be the (begin, end) of input indexes
716  // included in the sum for the i'th output index.
717  std::vector<Int32Pair> forward_indexes_cpu(num_output_indexes,
718  invalid_pair);
719  // backward_indexes_cpu[i] will be the (begin, end) of output indexes
720  // for which the i'th input index participates in the sum.
721  // because of the way the indexes are sorted (and the fact that only
722  // required indexes are present at the input), it naturally has this
723  // structure [i.e. no gaps in the sets of indexes].
724  std::vector<Int32Pair> backward_indexes_cpu(num_input_indexes,
725  invalid_pair);
726 
727  // this map maps from Index to the position in 'input_indexes'.
728  unordered_map<Index, int32, IndexHasher> index_to_input_pos;
729  for (int32 i = 0; i < num_input_indexes; i++)
730  index_to_input_pos[input_indexes[i]] = i;
731 
732  for (int32 i = 0; i < num_output_indexes; i++) {
733  Index input_index(output_indexes[i]);
734  int32 middle_t = input_index.t,
735  t_start = middle_t - left_context_,
736  t_last = middle_t + right_context_;
737  for (int32 t = t_start; t <= t_last; t += input_period_) {
738  input_index.t = t;
739  unordered_map<Index, int32, IndexHasher>::iterator iter =
740  index_to_input_pos.find(input_index);
741  if (iter != index_to_input_pos.end()) {
742  int32 input_pos = iter->second;
743  if (forward_indexes_cpu[i].first == -1) {
744  forward_indexes_cpu[i].first = input_pos;
745  forward_indexes_cpu[i].second = input_pos + 1;
746  } else {
747  KALDI_ASSERT(forward_indexes_cpu[i].second == input_pos);
748  forward_indexes_cpu[i].second++;
749  }
750  if (backward_indexes_cpu[input_pos].first == -1) {
751  backward_indexes_cpu[input_pos].first = i;
752  backward_indexes_cpu[input_pos].second = i + 1;
753  } else {
754  KALDI_ASSERT(backward_indexes_cpu[input_pos].second == i);
755  backward_indexes_cpu[input_pos].second++;
756  }
757  }
758  }
759  KALDI_ASSERT(forward_indexes_cpu[i].first != -1);
760  }
761  for (int32 i = 0; i < num_input_indexes; i++) {
762  KALDI_ASSERT(backward_indexes_cpu[i].first != -1);
763  }
764 
765  ans->forward_indexes = forward_indexes_cpu;
766  if (need_backprop)
767  ans->backward_indexes = backward_indexes_cpu;
768  return ans;
769 }

771 void* StatisticsPoolingComponent::Propagate(
772  const ComponentPrecomputedIndexes *indexes_in,
773  const CuMatrixBase<BaseFloat> &in,
774  CuMatrixBase<BaseFloat> *out) const {
775  out->SetZero();
776  KALDI_ASSERT(indexes_in != NULL);
777  const StatisticsPoolingComponentPrecomputedIndexes *indexes =
778  dynamic_cast<const StatisticsPoolingComponentPrecomputedIndexes*>(indexes_in);
779  int32 num_rows_out = out->NumRows();
780  KALDI_ASSERT(indexes != NULL &&
781  indexes->forward_indexes.Dim() == num_rows_out &&
782  in.NumCols() == input_dim_ &&
783  out->NumCols() == OutputDim());
784  CuVector<BaseFloat> counts(num_rows_out);
785  // counts_mat is a fake matrix with one column, containing the counts.
786  CuSubMatrix<BaseFloat> counts_mat(counts.Data(), num_rows_out, 1, 1);
787  counts_mat.AddRowRanges(in.ColRange(0, 1), indexes->forward_indexes);
788 
789  CuSubMatrix<BaseFloat> out_non_count(*out, 0, num_rows_out,
790  num_log_count_features_, input_dim_ - 1);
791  out_non_count.AddRowRanges(in.ColRange(1, input_dim_ - 1),
792  indexes->forward_indexes);
793  out_non_count.DivRowsVec(counts);
794 
795  if (num_log_count_features_ > 0) {
796  counts.ApplyLog();
797  CuVector<BaseFloat> ones(num_log_count_features_, kUndefined);
798  ones.Set(1.0);
799  out->ColRange(0, num_log_count_features_).AddVecVec(1.0, counts, ones);
800  }
801 
802  if (output_stddevs_) {
803  // if this is true, then we assume the input contains x^2 stats as well as x
804  // stats, and we want to process them into a standard deviation.
805  KALDI_ASSERT((input_dim_ - 1) % 2 == 0);
806  int32 feature_dim = (input_dim_ - 1) / 2;
807  CuSubMatrix<BaseFloat> mean(*out, 0, num_rows_out,
808  num_log_count_features_, feature_dim),
809  variance(*out, 0, num_rows_out,
810  num_log_count_features_ + feature_dim, feature_dim);
811  // subtract mean-squared from average of x^2 to get the variance.
812  variance.AddMatMatElements(-1.0, mean, mean, 1.0);
813  variance.ApplyFloor(variance_floor_);
814  // compute the standard deviation via square root.
815  variance.ApplyPow(0.5);
816  }
817  return NULL;
818 }

820 void StatisticsPoolingComponent::Backprop(
821  const std::string &debug_info,
822  const ComponentPrecomputedIndexes *indexes_in,
823  const CuMatrixBase<BaseFloat> &in_value,
824  const CuMatrixBase<BaseFloat> &out_value,
825  const CuMatrixBase<BaseFloat> &out_deriv_in,
826  void *memo,
827  Component *, // to_update,
828  CuMatrixBase<BaseFloat> *in_deriv) const {
829  KALDI_ASSERT(indexes_in != NULL);
830  const StatisticsPoolingComponentPrecomputedIndexes *indexes =
831  dynamic_cast<const StatisticsPoolingComponentPrecomputedIndexes*>(
832  indexes_in);
833  int32 num_rows_out = out_deriv_in.NumRows();
834  CuMatrix<BaseFloat> out_deriv(out_deriv_in);
835  if (output_stddevs_) {
836  // for now we actually ignore the covariance flooring in the backprop- this
837  // is an approximation. Typically the derivatives computed will be quite
838  // tiny for floored variances (they should be zero), so it won't affect the
839  // derivatives much.
840  int32 feature_dim = (input_dim_ - 1) / 2;
841  CuSubMatrix<BaseFloat> mean_deriv(out_deriv, 0, num_rows_out,
842  num_log_count_features_, feature_dim),
843  variance_deriv(out_deriv, 0, num_rows_out,
844  num_log_count_features_ + feature_dim, feature_dim),
845  mean_value(out_value, 0, num_rows_out,
846  num_log_count_features_, feature_dim),
847  stddev_value(out_value, 0, num_rows_out,
848  num_log_count_features_ + feature_dim, feature_dim);
849  // we currently have the deriv w.r.t. the stddev. step 1 is to get it
850  // w.r.t. the centered variance. If the centered variance is s,
851  // and the stddev is sqrt(s), then d/ds sqrt(s) = 0.5 / sqrt(s),
852  // so we need to multiply variance_deriv by 0.5 / the stddev.
853  variance_deriv.DivElements(stddev_value);
854  variance_deriv.Scale(0.5);
855 
856  // the deriv w.r.t. the uncentered variance is the same as w.r.t. the
857  // uncentered variance (since they difer by a constant term of -(mean *
858  // mean), but we need to add to dF/dmean, the value -2.0 * mean *
859  // dF/dvariance.
860  mean_deriv.AddMatMatElements(-2.0, mean_value, variance_deriv, 1.0);
861  }
862  // now we have to account for the effect of division by the count, on
863  // the derivative.
864  CuVector<BaseFloat> counts(num_rows_out, kUndefined);
865  if (num_log_count_features_ > 0) {
866  counts.CopyColFromMat(out_value, 0);
867  counts.ApplyExp();
868  } else {
869  counts.SetZero();
870  // we need to recompute the counts from the input since they are not in the
871  // output. The submatrix initializer below takes num-rows, num-cols,
872  // stride; num-cols and stride are 1.
873  CuSubMatrix<BaseFloat> counts_mat(counts.Data(), num_rows_out, 1, 1);
874  counts_mat.AddRowRanges(in_value.ColRange(0, 1), indexes->forward_indexes);
875  }
876  // Divide the output derivative by the counts. This is what we want as it
877  // concerns the mean and x^2 stats. As for the counts themselves, the
878  // derivative will end up being discarded when we backprop to the
879  // StatisticsExtractionComponent (as the count is not differentiable) so it
880  // doesn't really matter.
881  out_deriv.DivRowsVec(counts);
882 
883  // Now propagate the derivative back to the input. we don't propagate it
884  // back for the count's row since it's non-differentiable.
885  in_deriv->ColRange(1, input_dim_ - 1).
886  AddRowRanges(out_deriv.ColRange(num_log_count_features_, input_dim_ - 1),
887  indexes->backward_indexes);
888 }

1050 // virtual
1051 ComponentPrecomputedIndexes*
1052 BackpropTruncationComponent::PrecomputeIndexes(
1053  const MiscComputationInfo &misc_info,
1054  const std::vector<Index> &input_indexes,
1055  const std::vector<Index> &output_indexes,
1056  bool need_backprop) const {
1057  int32 num_input_indexes = input_indexes.size(),
1058  num_output_indexes = output_indexes.size();
1059  KALDI_ASSERT(num_input_indexes == num_output_indexes);
1060  Vector<BaseFloat> zeroing_cpu(num_output_indexes);
1061 
1062  for (int32 i = 0; i < num_output_indexes; i++) {
1063  const int32 output_n = output_indexes[i].n;
1064  const int32 output_t = output_indexes[i].t;
1065  // checks if output_t crosses a boundary that is a multiple of
1066  // zeroing_interval_. Note that frame (output_t - recurrence_interval_) is
1067  // right before frame output_t in RNNs. If the range
1068  // [output_t - recurrence_interval_, output_t] contains a multiple of
1069  // zeroing_interval_, then frame output_t crosses the boundary.
1070  // output_n is used to shift where we put the boundary, so that
1071  // we don't always zero out gradients on frame 0. It will help avoid
1072  // learning utterance-boundary effects.
1073  if (DivideRoundingDown(output_t - output_n, zeroing_interval_) !=
1074  DivideRoundingDown(output_t - recurrence_interval_ - output_n,
1075  zeroing_interval_))
1076  zeroing_cpu(i) = -1.0;
1077  }
1078 
1079  BackpropTruncationComponentPrecomputedIndexes *ans = new
1080  BackpropTruncationComponentPrecomputedIndexes();
1081  ans->zeroing = zeroing_cpu;
1082  ans->zeroing_sum = -zeroing_cpu.Sum();
1083  return ans;
1084 }

1097 // virtual
1098 void BackpropTruncationComponent::Backprop(const std::string &debug_info,
1099  const ComponentPrecomputedIndexes *indexes_in,
1100  const CuMatrixBase<BaseFloat> &, //in_value
1101  const CuMatrixBase<BaseFloat> &,
1102  const CuMatrixBase<BaseFloat> &out_deriv,
1103  void *memo,
1104  Component *to_update_in, // may be NULL; may be
1105  // identical to "this" or different.
1106  CuMatrixBase<BaseFloat> *in_deriv) const {
1107  const BackpropTruncationComponentPrecomputedIndexes *indexes =
1108  dynamic_cast<const BackpropTruncationComponentPrecomputedIndexes*>(
1109  indexes_in);
1110  KALDI_ASSERT(indexes->zeroing.Dim() == out_deriv.NumRows());
1111  // the following statement will do nothing if in_deriv and out_deriv have same
1112  // memory.
1113  in_deriv->CopyFromMat(out_deriv);
1114  if (scale_ != 1.0)
1115  in_deriv->Scale(scale_);
1116 
1117  BackpropTruncationComponent *to_update =
1118  dynamic_cast<BackpropTruncationComponent*>(to_update_in);
1119 
1120  // computes clipping_scales
1121  BaseFloat clipping_threshold =
1122  (clipping_threshold_ <= 0.0 ? 1.0e+10 : clipping_threshold_);
1123  // each row in the derivative matrix, which corresponds to one sample in
1124  // the mini-batch, is scaled to have a max-norm of clipping_threshold_
1125  CuVector<BaseFloat> clipping_scales(in_deriv->NumRows());
1126  clipping_scales.AddDiagMat2(pow(clipping_threshold, -2), *in_deriv,
1127  kNoTrans, 0.0);
1128  // now clipping_scales contains the squared (norm of each row divided by
1129  // clipping_threshold)
1130  int32 num_not_scaled;
1131  clipping_scales.ApplyFloor(1.0, &num_not_scaled);
1132  // now clipping_scales contains min(1, squared-(norm/clipping_threshold))
1133  clipping_scales.ApplyPow(-0.5);
1134  // now clipping_scales contains max(1, clipping_threshold/vector_norm)
1135  if (to_update != NULL) {
1136  to_update->num_clipped_ += (clipping_scales.Dim() - num_not_scaled);
1137  to_update->count_ += clipping_scales.Dim();
1138  }
1139 
1140  // computes zeroing_scales
1141  BaseFloat zeroing_threshold =
1142  (zeroing_threshold_ <= 0.0 ? 1.0e+10 : zeroing_threshold_);
1143  // zeroing_scales_vec is actually a 1-row matrix. (the ApplyHeaviside
1144  // function isn't defined for vectors).
1145  CuMatrix<BaseFloat> zeroing_scales(1, in_deriv->NumRows());
1146  CuSubVector<BaseFloat> zeroing_scales_vec(zeroing_scales, 0);
1147  zeroing_scales_vec.Set(-pow(zeroing_threshold, 2));
1148  // now zeroing_scales_vec contains -(squared zeroing_threshold)
1149  zeroing_scales_vec.AddDiagMat2(1.0, *in_deriv, kNoTrans, 1.0);
1150  // now zeroing_scales_vec contains squared norm of each row -
1151  // squared zeroing_threshold
1152  zeroing_scales.ApplyHeaviside();
1153  // now the element of zeroing_scales_vec is 1.0 if its corresponding
1154  // sample's norm exceeds zero_threshold, and 0.0 otherwise
1155  zeroing_scales_vec.MulElements(indexes->zeroing);
1156  // now the element of zeroing_scales_vec is -1.0 if we want to zero its
1157  // corresponding sample's gradient, and 0.0 otherwise
1158  if (to_update != NULL) {
1159  to_update->num_zeroed_ -= zeroing_scales_vec.Sum(); // since it is negative
1160  to_update->count_zeroing_boundaries_ += indexes->zeroing_sum;
1161  }
1162  zeroing_scales_vec.Add(1.0);
1163  // now the element of zeroing_scales_vec is 0.0 if we want to zero its
1164  // corresponding sample's gradient, and 1.0 otherwise
1165 
1166  // combines clipping_scales and zeroing_scales and applies combined_scales
1167  // to in_deriv all at once
1168  CuVector<BaseFloat> combined_scales(clipping_scales);
1169  combined_scales.MulElements(zeroing_scales_vec);
1170  in_deriv->MulRowsVec(combined_scales);
1171 }

1413  const ComponentPrecomputedIndexes *indexes,
1414  const CuMatrixBase<BaseFloat> &in,
1415  CuMatrixBase<BaseFloat> *out) const {
1416  KALDI_ASSERT(in.NumRows() == 0 && out->NumCols() == output_dim_);
1417  BaseFloat dropout_proportion = dropout_proportion_;
1418  KALDI_ASSERT(dropout_proportion >= 0.0 && dropout_proportion <= 1.0);
1419 
1420  if (dropout_proportion == 0) {
1421  out->Set(1.0);
1422  return NULL;
1423  }
1424 
1425  if (continuous_) {
1426  if (test_mode_) {
1427  out->Set(1.0);
1428  } else {
1429  const_cast<CuRand<BaseFloat>&>(random_generator_).RandUniform(out);
1430  out->Scale(dropout_proportion * 4.0);
1431  // make the expected value 1.0.
1432  out->Add(1.0 - (2.0 * dropout_proportion));
1433  }
1434  return NULL;
1435  }
1436 
1437  if (test_mode_) {
1438  out->Set(1.0 - dropout_proportion);
1439  return NULL;
1440  }
1441 
1442  const_cast<CuRand<BaseFloat>&>(random_generator_).RandUniform(out);
1443  out->Add(-dropout_proportion);
1444  out->ApplyHeaviside();
1445 
1446  if (out->NumCols() == 2 || out->NumCols() == 3) {
1447  // This is a kind of special case relevant to LSTms.
1448  // To generate data where it's never the case that both of the dimensions
1449  // for a row are zero, we generate uniformly distributed data (call this u_i),
1450  // and for row i, set (*out)(i, 0) = (0 if u_i < dropout_proportion else 1)
1451  // and (*out)(i, 1) = (0 if u_i > 1-dropout_proportion else 1)
1452  int32 num_rows = out->NumRows();
1453  // later we may make this a bit more efficient.
1454  CuVector<BaseFloat> temp(num_rows, kUndefined);
1455  const_cast<CuRand<BaseFloat>&>(random_generator_).RandUniform(&temp);
1456  temp.Add(-dropout_proportion);
1457  out->CopyColFromVec(temp, 0);
1458  temp.Add(-1.0 + (2.0 * dropout_proportion));
1459  // Now, 'temp' contains the original uniformly-distributed data plus
1460  // -(1 - dropout_proportion).
1461  temp.Scale(-1.0);
1462  out->CopyColFromVec(temp, 1);
1463  out->ApplyHeaviside();
1464  }
1465  return NULL;
1466 }

1708 ComponentPrecomputedIndexes* GeneralDropoutComponent::PrecomputeIndexes(
1709  const MiscComputationInfo &misc_info,
1710  const std::vector<Index> &input_indexes,
1711  const std::vector<Index> &output_indexes,
1712  bool need_backprop) const {
1713  KALDI_ASSERT(input_indexes == output_indexes);
1714 
1715  GeneralDropoutComponentPrecomputedIndexes *ans = new
1716  GeneralDropoutComponentPrecomputedIndexes;
1717  int32 size = input_indexes.size(), time_period = time_period_,
1718  cur_row = 0;
1719  std::vector<int32> indexes(size);
1720  // the map 'm' will map from a pair from (n, t) value to the row-index of the
1721  // dropout-mask matrix*. However, the 't' isn't a real 't' value;
1722  // if time_period_ == 0, the 't' value will just be zero; otherwise,
1723  // it will be t divided by time_period_ (rounding towards negative infinity).
1724 
1725  // *before considering effects related to when block_dim_ != dim_.
1726 
1727  std::unordered_map<std::pair<int32,int32>, int32, PairHasher<int32> > m;
1728  for (int32 i = 0; i < size; i++) {
1729  int32 n = input_indexes[i].n,
1730  t = (time_period == 0 ? 0 : DivideRoundingDown(input_indexes[i].t,
1731  time_period));
1732  std::pair<int32, int32> p(n, t);
1733 
1734  std::unordered_map<std::pair<int32,int32>, int32,
1735  PairHasher<int32> >::const_iterator
1736  iter = m.find(p);
1737  if (iter != m.end()) {
1738  indexes[i] = iter->second;
1739  } else {
1740  m[p] = cur_row;
1741  indexes[i] = cur_row;
1742  cur_row++;
1743  }
1744  }
1745  int32 multiple = dim_ / block_dim_;
1746  ans->num_mask_rows = cur_row;
1747  if (multiple == 1) {
1748  ans->indexes.CopyFromVec(indexes);
1749  } else {
1750  ans->num_mask_rows = cur_row * multiple;
1751  std::vector<int32> repeated_indexes;
1752  repeated_indexes.reserve(size * multiple);
1753  for (int32 i = 0; i < size; i++) {
1754  int32 row = indexes[i];
1755  for (int32 j = 0; j < multiple; j++)
1756  repeated_indexes.push_back(row);
1757  }
1758  ans->indexes.CopyFromVec(repeated_indexes);
1759  }
1760  return ans;
1761 }
