37 // The Decodable objects that we define in this header do the neural net
38 // computation in a way that's compatible with online feature extraction. It
39 // differs from the one declared in online-nnet3-decodable-simple.h because it
40 // uses the 'looped' network evaluation, which is more efficient because it
41 // re-uses hidden activations (and therefore doesn't have to pad chunks of data
42 // with extra left-context); it is applicable to TDNNs and to forwards-recurrent
43 // topologies like LSTMs, but not tobackwards-recurrent topologies such as
44 // BLSTMs.
45 
46 // The options are passed in the same way as in decodable-simple-looped.h,
47 // we use the same options and info class.
48

50 // This object is used as a base class for DecodableNnetLoopedOnline
51 // and DecodableAmNnetLoopedOnline.
52 // It takes care of the neural net computation and computations related to how
53 // many frames are ready (etc.), but it does not override the LogLikelihood() or
54 // NumIndices() functions so it is not usable as an object of type
55 // DecodableInterface.
56 class DecodableNnetLoopedOnlineBase: public DecodableInterface {
57  public:

62  OnlineFeatureInterface *input_features,
63  OnlineFeatureInterface *ivector_features);

92  inline void EnsureFrameIsComputed(int32 subsampled_frame) {
93  KALDI_ASSERT(subsampled_frame >= current_log_post_subsampled_offset_ &&
94  "Frames must be accessed in order.");
95  while (subsampled_frame >= current_log_post_subsampled_offset_ +
96  current_log_post_.NumRows())
97  AdvanceChunk();
98  }
99 
100  // The current log-posteriors that we got from the last time we
101  // ran the computation.
102  Matrix<BaseFloat> current_log_post_;
103 
104  // The number of chunks we have computed so far.
105  int32 num_chunks_computed_;
106 
107  // The time-offset of the current log-posteriors, equals
108  // (num_chunks_computed_ - 1) *
109  // (info_.frames_per_chunk_ / info_.opts_.frame_subsampling_factor).
110  int32 current_log_post_subsampled_offset_;
111 
112  const DecodableNnetSimpleLoopedInfo &info_;
113 
114  private:
115 
116  // This function does the computation for the next chunk. It will change
117  // current_log_post_ and current_log_post_subsampled_offset_, and
118  // increment num_chunks_computed_.
119  void AdvanceChunk();
120 
121  OnlineFeatureInterface *input_features_;
122  OnlineFeatureInterface *ivector_features_;
123 
124  NnetComputer computer_;
125 
126  KALDI_DISALLOW_COPY_AND_ASSIGN(DecodableNnetLoopedOnlineBase);
127 };

129 // This decodable object takes indexes of the form (pdf_id + 1),
130 // or whatever the output-dimension of the neural network represents,
131 // plus one.
132 // It fully implements DecodableInterface.
133 // Note: whether or not division by the prior takes place depends on
134 // whether you supplied class AmNnetSimple (or just Nnet), to the constructor
135 // of the DecodableNnetSimpleLoopedInfo that you initailized this
136 // with.

138  public:

143  DecodableNnetLoopedOnlineBase(info, input_features, ivector_features) { }

148
