34 int main(int argc, char *argv[]) {
35  using namespace kaldi;
36  using namespace kaldi::nnet1;
37  typedef kaldi::int32 int32;
38 
39  try {
40  const char *usage =
41  "Perform one iteration of Multi-stream training, per-utterance BPTT for (B)LSTMs.\n"
42  "The updates are done per-utterance, while several utterances are \n"
43  "processed at the same time.\n"
44  "\n"
45  "Usage: nnet-train-multistream-perutt [options] <feature-rspecifier> <labels-rspecifier> <model-in> [<model-out>]\n"
46  "e.g.: nnet-train-blstm-streams scp:feats.scp ark:targets.ark nnet.init nnet.iter1\n";
47 
48  ParseOptions po(usage);
49 
50  // training options,
51  NnetTrainOptions trn_opts;
52  trn_opts.Register(&po);
53  LossOptions loss_opts;
54  loss_opts.Register(&po);
55 
56  bool binary = true;
57  po.Register("binary", &binary, "Write model in binary mode");
58 
59  bool crossvalidate = false;
60  po.Register("cross-validate", &crossvalidate,
61  "Perform cross-validation (no backpropagation)");
62 
63  std::string feature_transform;
64  po.Register("feature-transform", &feature_transform,
65  "Feature transform in Nnet format");
66 
67  std::string output_feature_transform;
68  po.Register("output-feature-transform", &output_feature_transform,
69  "Output feature transform in Nnet format");
70 
71  std::string objective_function = "xent";
72  po.Register("objective-function", &objective_function,
73  "Objective function : xent|mse");
74 
75  int32 length_tolerance = 5;
76  po.Register("length-tolerance", &length_tolerance,
77  "Allowed length difference of features/targets (frames)");
78 
79  std::string frame_weights;
80  po.Register("frame-weights", &frame_weights,
81  "Per-frame weights to scale gradients (frame selection/weighting).");
82 
83  int32 num_streams = 20;
84  po.Register("num-streams", &num_streams,
85  "Number of sentences processed in parallel (can be lower if sentences are long)");
86 
87  double max_frames = 8000;
88  po.Register("max-frames", &max_frames,
89  "Max number of frames to be processed");
90 
91  bool dummy = false;
92  po.Register("randomize", &dummy, "Dummy option.");
93 
94  std::string use_gpu = "yes";
95  po.Register("use-gpu", &use_gpu,
96  "yes|no|optional, only has effect if compiled with CUDA");
97 
98  po.Read(argc, argv);
99 
100  if (po.NumArgs() != 3 + (crossvalidate ? 0 : 1)) {
101  po.PrintUsage();
102  exit(1);
103  }
104 
105  std::string feature_rspecifier = po.GetArg(1),
106  targets_rspecifier = po.GetArg(2),
107  model_filename = po.GetArg(3);
108 
109  std::string target_model_filename;
110  if (!crossvalidate) {
111  target_model_filename = po.GetArg(4);
112  }
113 
114  using namespace kaldi;
115  using namespace kaldi::nnet1;
116  typedef kaldi::int32 int32;
117 
118 #if HAVE_CUDA == 1
119  CuDevice::Instantiate().SelectGpuId(use_gpu);
120 #endif
121 
122  Nnet nnet_transf, out_nnet_transf;
123  if ( feature_transform != "" ) {
124  nnet_transf.Read(feature_transform);
125  }
126  if (output_feature_transform != "") {
127  out_nnet_transf.Read(output_feature_transform);
128  }
129 
130  Nnet nnet;
131  nnet.Read(model_filename);
132  nnet.SetTrainOptions(trn_opts);
133 
134  if (crossvalidate) {
135  nnet_transf.SetDropoutRate(0.0);
136  nnet.SetDropoutRate(0.0);
137  out_nnet_transf.SetDropoutRate(0.0);
138  }
139 
140  kaldi::int64 total_frames = 0;
141 
142  // Initialize feature and target readers,
143  SequentialBaseFloatMatrixReader feature_reader(feature_rspecifier);
144  RandomAccessBaseFloatMatrixReader targets_reader(targets_rspecifier);
145  RandomAccessBaseFloatVectorReader weights_reader;
146  if (frame_weights != "") {
147  weights_reader.Open(frame_weights);
148  }
149 
150 
151  Xent xent(loss_opts);
152  Mse mse(loss_opts);
153 
154  CuMatrix<BaseFloat> feats_transf, nnet_tgt, nnet_out, obj_diff;
155 
156  Timer time;
157  KALDI_LOG << (crossvalidate ? "CROSS-VALIDATION" : "TRAINING")
158  << " STARTED";
159 
160  // Buffer for input features, used for choosing utt's with similar length,
161  MatrixBuffer matrix_buffer;
162  matrix_buffer.Init(&feature_reader);
163 
164  int32 num_done = 0,
165  num_no_tgt_mat = 0,
166  num_other_error = 0;
167 
168  while (!matrix_buffer.Done()) {
169 
170  // Fill the parallel data into 'std::vector',
171  std::vector<Matrix<BaseFloat> > feats_utt;
172  std::vector<Matrix<BaseFloat> > targets_utt;
173  std::vector<Vector<BaseFloat> > weights_utt;
174  std::vector<int32> frame_num_utt;
175  {
176  matrix_buffer.ResetLength();
177  for (matrix_buffer.Next(); !matrix_buffer.Done(); matrix_buffer.Next()) {
178  std::string utt = matrix_buffer.Key();
179  // Check that we have targets,
180  if (!targets_reader.HasKey(utt)) {
181  KALDI_WARN << utt << ", missing targets";
182  num_no_tgt_mat++;
183  continue;
184  }
185  // Do we have frame-weights?
186  if (frame_weights != "" && !weights_reader.HasKey(utt)) {
187  KALDI_WARN << utt << ", missing frame-weights";
188  num_other_error++;
189  continue;
190  }
191 
192  // Get feature / target pair,
193  Matrix<BaseFloat> mat = matrix_buffer.Value();
194  Matrix<BaseFloat> targets = targets_reader.Value(utt);
195 
196  // Skip too long sentences,
197  if (mat.NumRows() > max_frames) continue;
198 
199  Vector<BaseFloat> weights;
200  if (frame_weights != "") {
201  weights = weights_reader.Value(utt);
202  } else { // all per-frame weights are 1.0
203  weights.Resize(mat.NumRows());
204  weights.Set(1.0);
205  }
206 
207  // correct small length mismatch ... or drop sentence
208  {
209  // add lengths to vector
210  std::vector<int32> length;
211  length.push_back(mat.NumRows());
212  length.push_back(targets.NumRows());
213  length.push_back(weights.Dim());
214  // find min, max
215  int32 min = *std::min_element(length.begin(), length.end());
216  int32 max = *std::max_element(length.begin(), length.end());
217  // fix or drop ?
218  if (max - min < length_tolerance) {
219  if (mat.NumRows() != min) mat.Resize(min, mat.NumCols(), kCopyData);
220  if (targets.NumRows() != min) targets.Resize(min, targets.NumCols(), kCopyData);
221  if (weights.Dim() != min) weights.Resize(min, kCopyData);
222  } else {
223  KALDI_WARN << "Length mismatch! Targets " << targets.NumRows()
224  << ", features " << mat.NumRows() << ", " << utt;
225  num_other_error++;
226  continue;
227  }
228  }
229 
230  // input transform may contain splicing,
231  nnet_transf.Feedforward(CuMatrix<BaseFloat>(mat), &feats_transf);
232  out_nnet_transf.Feedforward(CuMatrix<BaseFloat>(targets), &nnet_tgt);
233 
234  // store,
235  feats_utt.push_back(Matrix<BaseFloat>(feats_transf));
236  targets_utt.push_back(Matrix<BaseFloat>(nnet_tgt));
237  weights_utt.push_back(weights);
238  frame_num_utt.push_back(feats_transf.NumRows());
239 
240  if (frame_num_utt.size() == num_streams) break;
241 
242  // See how many frames we'd have (after padding), if we add one more utterance,
243  int32 max = (*std::max_element(frame_num_utt.begin(), frame_num_utt.end()));
244  if (max * (frame_num_utt.size() + 1) > max_frames) break;
245  }
246  }
247  // Having no data? Skip the cycle...
248  if (frame_num_utt.size() == 0) continue;
249 
250  // Pack the parallel data,
251  Matrix<BaseFloat> feat_mat_host;
252  CuMatrix<BaseFloat> target_mat_host;
253  Vector<BaseFloat> weight_host;
254  {
255  // Number of sequences,
256  int32 n_streams = frame_num_utt.size();
257  int32 frame_num_padded = (*std::max_element(frame_num_utt.begin(), frame_num_utt.end()));
258  int32 feat_dim = feats_utt.front().NumCols();
259  int32 target_dim = targets_utt.front().NumCols();
260 
261  // Create the final feature matrix. Every utterance is padded to the max
262  // length within this group of utterances,
263  feat_mat_host.Resize(n_streams * frame_num_padded, feat_dim, kSetZero);
264  target_mat_host.Resize(n_streams * frame_num_padded, target_dim, kSetZero);
265  weight_host.Resize(n_streams * frame_num_padded, kSetZero);
266 
267  for (int32 s = 0; s < n_streams; s++) {
268  const Matrix<BaseFloat>& mat_tmp = feats_utt[s];
269  for (int32 r = 0; r < frame_num_utt[s]; r++) {
270  feat_mat_host.Row(r*n_streams + s).CopyFromVec(mat_tmp.Row(r));
271  }
272  }
273 
274  for (int32 s = 0; s < n_streams; s++) {
275  const Matrix<BaseFloat>& target_mat_tmp = targets_utt[s];
276  for (int32 r = 0; r < frame_num_utt[s]; r++) {
277  target_mat_host.Row(r*n_streams + s).CopyFromVec(target_mat_tmp.Row(r));
278  }
279  }
280 
281  // padded frames will keep initial zero-weight,
282  for (int32 s = 0; s < n_streams; s++) {
283  const Vector<BaseFloat>& weight_tmp = weights_utt[s];
284  for (int32 r = 0; r < frame_num_utt[s]; r++) {
285  weight_host(r*n_streams + s) = weight_tmp(r);
286  }
287  }
288  }
289 
290  // Set the original lengths of utterances before padding,
291  nnet.SetSeqLengths(frame_num_utt);
292  // Show the 'utt' lengths in the VLOG[2],
293  if (GetVerboseLevel() >= 2) {
294  std::ostringstream os;
295  os << "[ ";
296  for (size_t i = 0; i < frame_num_utt.size(); i++) {
297  os << frame_num_utt[i] << " ";
298  }
299  os << "]";
300 
301  KALDI_LOG << "frame_num_utt[" << frame_num_utt.size() << "]" << os.str();
302  }
303  // Reset all the streams (we have new sentences),
304  nnet.ResetStreams(std::vector<int32>(frame_num_utt.size(), 1));
305 
306  // Propagation,
307  nnet.Propagate(CuMatrix<BaseFloat>(feat_mat_host), &nnet_out);
308 
309  // evaluate objective function we've chosen,
310  if (objective_function == "xent") {
311  xent.Eval(weight_host, nnet_out, target_mat_host, &obj_diff);
312  } else if (objective_function == "mse") {
313  mse.Eval(weight_host, nnet_out, target_mat_host, &obj_diff);
314  } else {
315  KALDI_ERR << "Unknown objective function code : "
316  << objective_function;
317  }
318 
319  // Backward pass
320  if (!crossvalidate) {
321  nnet.Backpropagate(obj_diff, NULL);
322  }
323 
324  // 1st model update : show what happens in network,
325  if (total_frames == 0) {
326  KALDI_VLOG(1) << "### After " << total_frames << " frames,";
327  KALDI_VLOG(1) << nnet.Info();
328  KALDI_VLOG(1) << nnet.InfoPropagate();
329  if (!crossvalidate) {
330  KALDI_VLOG(1) << nnet.InfoBackPropagate();
331  KALDI_VLOG(1) << nnet.InfoGradient();
332  }
333  }
334 
335  int32 tmp_done = num_done;
336  kaldi::int64 tmp_frames = total_frames;
337 
338  num_done += frame_num_utt.size();
339  total_frames += std::accumulate(frame_num_utt.begin(), frame_num_utt.end(), 0);
340 
341  // report the speed,
342  int32 N = 5000;
343  if (tmp_done / N != num_done / N) {
344  double time_now = time.Elapsed();
345  KALDI_VLOG(1) << "After " << num_done << " utterances, "
346  << "(" << total_frames/360000.0 << "h), "
347  << "time elapsed = " << time_now / 60 << " min; "
348  << "processed " << total_frames / time_now << " frames per sec.";
349  }
350 
351  // monitor the NN training (--verbose=2),
352  int32 F = 25000;
353  if (GetVerboseLevel() >= 3) {
354  // print every 25k frames,
355  if (tmp_frames / F != total_frames / F) {
356  KALDI_VLOG(2) << "### After " << total_frames << " frames,";
357  KALDI_VLOG(2) << nnet.Info();
358  KALDI_VLOG(2) << nnet.InfoPropagate();
359  if (!crossvalidate) {
360  KALDI_VLOG(2) << nnet.InfoBackPropagate();
361  KALDI_VLOG(2) << nnet.InfoGradient();
362  }
363  }
364  }
365  }
366 
367  // after last model update : show what happens in network,
368  if (GetVerboseLevel() >= 1) { // vlog-1
369  KALDI_VLOG(1) << "### After " << total_frames << " frames,";
370  KALDI_VLOG(1) << nnet.Info();
371  KALDI_VLOG(1) << nnet.InfoPropagate();
372  if (!crossvalidate) {
373  KALDI_VLOG(1) << nnet.InfoBackPropagate();
374  KALDI_VLOG(1) << nnet.InfoGradient();
375  }
376  }
377 
378  if (!crossvalidate) {
379  nnet.Write(target_model_filename, binary);
380  }
381 
382  if (objective_function == "xent") {
383  KALDI_LOG << xent.ReportPerClass();
384  }
385 
386  KALDI_LOG << "Done " << num_done << " files, "
387  << num_no_tgt_mat << " with no tgt_mats, "
388  << num_other_error << " with other errors. "
389  << "[" << (crossvalidate ? "CROSS-VALIDATION" : "TRAINING")
390  << ", " << time.Elapsed() / 60 << " min, processing "
391  << total_frames / time.Elapsed() << " frames per sec.]";
392 
393  if (objective_function == "xent") {
394  KALDI_LOG << xent.Report();
395  } else if (objective_function == "mse") {
396  KALDI_LOG << mse.Report();
397  } else {
398  KALDI_ERR << "Unknown objective function code : " << objective_function;
399  }
400 
401 #if HAVE_CUDA == 1
402  CuDevice::Instantiate().PrintProfile();
403 #endif
404  return 0;
405  } catch(const std::exception &e) {
406  std::cerr << e.what();
407  return -1;
408  }
409 }
