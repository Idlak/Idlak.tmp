29 // This header provides convenient mechanisms for parallelization.
30 //
31 // The class MultiThreader, and the function RunMultiThreaded provide a
32 // mechanism to run a specified number of jobs in parellel and wait for them
33 // all to finish. They accept objects of some class C that derives from the
34 // base class MultiThreadable. C needs to define the operator () that takes
35 // no arguments. See ExampleClass below.
36 //
37 // The class TaskSequencer addresses a different problem typically encountered
38 // in Kaldi command-line programs that process a sequence of items. The items
39 // to be processed are coming in. They are all of different sizes, e.g.
40 // utterances with different numbers of frames. We would like them to be
41 // processed in parallel to make good use of the threads available but they
42 // must be output in the same order they came in. Here, we again accept objects
43 // of some class C with an operator () that takes no arguments. C may also have
44 // a destructor with side effects (typically some kind of output).
45 // TaskSequencer is responsible for running the jobs in parallel. It has a
46 // function Run() that will accept a new object of class C; this will block
47 // until a thread is free, at which time it will spawn a thread that starts
48 // running the operator () of the object. When threads are finished running,
49 // the objects will be deleted. TaskSequencer guarantees that the destructors
50 // will be called sequentially (not in parallel) and in the same order the
51 // objects were given to the Run() function, so that it is safe for the
52 // destructor to have side effects such as outputting data.
53 // Note: the destructor of TaskSequencer will wait for any remaining jobs that
54 // are still running and will call the destructors.

66 class MultiThreadable {
67  // To create a function object that does part of the job, inherit from this
68  // class, implement a copy constructor calling the default copy constructor
69  // of this base class (so that thread_id_ and num_threads_ are copied to new
70  // instances), and finally implement the operator() that does part of the job
71  // based on thread_id_ and num_threads_ variables.
72  // Note: example implementations are in util/kaldi-thread-test.cc
73  public:
74  virtual void operator() () = 0;
75  // Does the main function of the class
76  // Subclasses have to redefine this
77  virtual ~MultiThreadable();
78  // Optional destructor. Note: the destructor of the object passed by the user
79  // will also be called, so watch out.
80 
81  public:
82  // Do not redeclare thread_id_ and num_threads_ in derived classes.
83  int32 thread_id_; // 0 <= thread_id_ < num_threads_
84  int32 num_threads_;
85 
86  private:
87  // Have additional member variables as needed.
88 };

115 template<class C>
116 class MultiThreader {
117  public:
118  MultiThreader(int32 num_threads, const C &c_in) :
119  threads_(std::max<int32>(1, num_threads)),
120  cvec_(std::max<int32>(1, num_threads), c_in) {
121  if (num_threads == 0) {
122  // This is a special case with num_threads == 0, which behaves like with
123  // num_threads == 1 but without creating extra threads. This can be
124  // useful in GPU computations where threads cannot be used.
125  cvec_[0].thread_id_ = 0;
126  cvec_[0].num_threads_ = 1;
127  (cvec_[0])();
128  } else {
129  for (int32 i = 0; i < threads_.size(); i++) {
130  cvec_[i].thread_id_ = i;
131  cvec_[i].num_threads_ = threads_.size();
132  threads_[i] = std::thread(std::ref(cvec_[i]));
133  }
134  }
135  }
136  ~MultiThreader() {
137  for (size_t i = 0; i < threads_.size(); i++)
138  if (threads_[i].joinable())
139  threads_[i].join();
140  }
141  private:
142  std::vector<std::thread> threads_;
143  std::vector<C> cvec_;
144 };

170 // C should have an operator () taking no arguments, that does some kind
171 // of computation, and a destructor that produces some kind of output (the
172 // destructors will be run sequentially in the same order Run as called.
173 template<class C>
174 class TaskSequencer {
175  public:
176  TaskSequencer(const TaskSequencerConfig &config):
177  num_threads_(config.num_threads),
178  threads_avail_(config.num_threads),
179  tot_threads_avail_(config.num_threads_total > 0 ? config.num_threads_total :
180  config.num_threads + 20),
181  thread_list_(NULL) {
182  KALDI_ASSERT((config.num_threads_total <= 0 ||
183  config.num_threads_total >= config.num_threads) &&
184  "num-threads-total, if specified, must be >= num-threads");
185  }
186 
189  void Run(C *c) {
190  // run in main thread
191  if (num_threads_ == 0) {
192  (*c)();
193  delete c;
194  return;
195  }
196 
197  threads_avail_.Wait(); // wait till we have a thread for computation free.
198  tot_threads_avail_.Wait(); // this ensures we don't have too many threads
199  // waiting on I/O, and consume too much memory.
200 
201  // put the new RunTaskArgsList object at head of the singly
202  // linked list thread_list_.
203  thread_list_ = new RunTaskArgsList(this, c, thread_list_);
204  thread_list_->thread = std::thread(TaskSequencer<C>::RunTask,
205  thread_list_);
206  }
207 
208  void Wait() { // You call this at the end if it's more convenient
209  // than waiting for the destructor. It waits for all tasks to finish.
210  if (thread_list_ != NULL) {
211  thread_list_->thread.join();
212  KALDI_ASSERT(thread_list_->tail == NULL); // thread would not
213  // have exited without setting tail to NULL.
214  delete thread_list_;
215  thread_list_ = NULL;
216  }
217  }
218 
220  ~TaskSequencer() {
221  Wait();
222  }
223  private:
224  struct RunTaskArgsList {
225  TaskSequencer *me; // Think of this as a "this" pointer.
226  C *c; // Clist element of the task we're expected
227  std::thread thread;
228  RunTaskArgsList *tail;
229  RunTaskArgsList(TaskSequencer *me, C *c, RunTaskArgsList *tail):
230  me(me), c(c), tail(tail) {}
231  };
232  // This static function gets run in the threads that we create.
233  static void RunTask(RunTaskArgsList *args) {
234  // (1) run the job.
235  (*(args->c))(); // call operator () on args->c, which does the computation.
236  args->me->threads_avail_.Signal(); // Signal that the compute-intensive
237  // part of the thread is done (we want to run no more than
238  // config_.num_threads of these.)
239 
240  // (2) we want to destroy the object "c" now, by deleting it. But for
241  // correct sequencing (this is the whole point of this class, it
242  // is intended to ensure the output of the program is in correct order),
243  // we first wait till the previous thread, whose details will be in "tail",
244  // is finished.
245  if (args->tail != NULL) {
246  args->tail->thread.join();
247  }
248 
249  delete args->c; // delete the object "c". This may cause some output,
250  // e.g. to a stream. We don't need to worry about concurrent access to
251  // the output stream, because each thread waits for the previous thread
252  // to be done, before doing this. So there is no risk of concurrent
253  // access.
254  args->c = NULL;
255 
256  if (args->tail != NULL) {
257  KALDI_ASSERT(args->tail->tail == NULL); // Because we already
258  // did join on args->tail->thread, which means that
259  // thread was done, and before it exited, it would have
260  // deleted and set to NULL its tail (which is the next line of code).
261  delete args->tail;
262  args->tail = NULL;
263  }
264  // At this point we are exiting from the thread. Signal the
265  // "tot_threads_avail_" semaphore which is used to limit the total number of threads that are alive, including
266  // not onlhy those that are in active computation in c->operator (), but those
267  // that are waiting on I/O or other threads.
268  args->me->tot_threads_avail_.Signal();
269  }
270 
271  int32 num_threads_; // copy of config.num_threads (since Semaphore doesn't store original count)
272 
273  Semaphore threads_avail_; // Initialized to the number of threads we are
274  // supposed to run with; the function Run() waits on this.
275 
276  Semaphore tot_threads_avail_; // We use this semaphore to ensure we don't
277  // consume too much memory...
278  RunTaskArgsList *thread_list_;
279 
280 };
