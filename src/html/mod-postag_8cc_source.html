37 bool TxpPosTag::Process(pugi::xml_document* input) {
38  const char *ptag = "", *tag, *set;
39  pugi::xpath_node_set tks = input->document_element().select_nodes("//tk");
40  tks.sort();
41  for (pugi::xpath_node_set::const_iterator it = tks.begin();
42  it != tks.end();
43  ++it) {
44  pugi::xml_node node = (*it).node();
45  if (it == tks.begin()) {
46  ptag = "#";
47  } else {
48  pugi::xml_node pnode = (*(it - 1)).node();
49  ptag = pnode.attribute("pos").value();
50  }
51  tag = tagger_.GetPos(ptag, node.attribute("tknorm").value());
52  if (!node.attribute("pos")) node.append_attribute("pos");
53  node.attribute("pos").set_value(tag);
54  set = posset_.GetPosSet(tag);
55  if (set) {
56  if (!node.attribute("posset")) node.append_attribute("posset");
57  node.attribute("posset").set_value(set);
58  }
59  }
60  return true;
61 }
