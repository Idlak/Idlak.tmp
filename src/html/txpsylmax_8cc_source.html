34 void TxpSylmax::Maxonset(PhoneVector *pronptr) {
35  std::string pat;
36  int32 nlen, olen, pos, i;
37  PhoneVector &pron = *pronptr;
38  StringSet::iterator it;
39  // move forwards trying to find the nucleus
40  for (pos = 0; pos < pron.size(); pos++) {
41  // try largest nuclei first
42  nlen = FindNucleus(pron, pos);
43  if (nlen) {
44  olen = FindOnset(pron, pos);
45  if (olen) {
46  if (pos - olen - 1 >= 0) pron[pos - olen - 1].codab = true;
47  for (i = pos - olen; i < pos; i++)
48  pron[i].type = TXPSYLMAX_TYPE_ONSET;
49  }
50  if (pos > 0) pron[pos - 1].codab = true;
51  // mark nucleus
52  for (i = pos; i < pos + nlen; i++)
53  pron[i].type = TXPSYLMAX_TYPE_NUCLEUS;
54  pron[pos + nlen - 1].codab = true;
55  if (nlen > 1) pos += nlen - 1;
56  }
57  }
58  AddSylBound(&pron);
59 }

61 void TxpSylmax::Writespron(PhoneVector *pronptr, std::string *sylpron) {
62  int32 pos, newwrdb = -1, i;
63  const char* bound;
64  PhoneVector &pron = *pronptr;
65  sylpron->clear();
66  // if wrdb is at a cross word point move it back to the end of
67  // the previous syllable
68  for (pos = 0; pos < pron.size(); pos++) {
69  if (pron[pos].wrdb == true) {
70  if (pron[pos].cross_word == false) {
71  newwrdb = pos;
72  break;
73  } else {
74  // move word boundary back
75  newwrdb = pos;
76  while (newwrdb >=0 && !pron[newwrdb].sylb) {
77  newwrdb--;
78  }
79  if (newwrdb >=0) pron[newwrdb].wrdb = true;
80  pron[pos].wrdb = false;
81  break;
82  }
83  }
84  }
85  // copy coda of previous word if held back
86  // for cross word syllabification
87  // if newwrdb == -1 there is no pron for this token anymore
88  for (i = 0; i <= newwrdb; i++) {
89  if (pron[i].sylb) bound = "|";
90  else if (pron[i].codab)
91  bound = "+";
92  else
93  bound = "_";
94  *sylpron += pron[i].name + pron[i].stress + bound;
95  }
96  // remove processed phones from the front of the array
97  pron.erase(pron.begin(), pron.begin() + newwrdb + 1);
98 }

101 int32 TxpSylmax::AddSylBound(PhoneVector *pronptr) {
102  PhoneVector &pron = *pronptr;
103  enum TXPSYLMAX_TYPE ltype = pron[0].type;
104  bool codab = pron[0].codab;
105  int i, coda;
106  for (i = 1; i < pron.size(); i++) {
107  if (codab &&
108  !((ltype == TXPSYLMAX_TYPE_ONSET &&
109  pron[i].type == TXPSYLMAX_TYPE_NUCLEUS) ||
110  (ltype == TXPSYLMAX_TYPE_NUCLEUS &&
111  pron[i].type == TXPSYLMAX_TYPE_CODA)))
112  pron[i - 1].sylb = true;
113  ltype = pron[i].type;
114  codab = pron[i].codab;
115  }
116  // Returns position of the start of the last coda in the word
117  // if cross word syllabification
118  if (pron[pron.size() - 1].cross_word) {
119  for (i = i - 1; i >= 0; i--) {
120  if (pron[i].type != TXPSYLMAX_TYPE_CODA) {
121  coda = i + 1;
122  break;
123  } else if (i == 0 && pron[i].type == TXPSYLMAX_TYPE_CODA) {
124  coda = 1;
125  }
126  }
127  return coda;
128  } else {
129  pron[pron.size() - 1].sylb = true;
130  }
131  return 0;
132 }

134 int32 TxpSylmax::FindNucleus(const PhoneVector &pron, int32 pos) {
135  int len;
136  std::string pat;
137  StringSet::iterator it;
138  for (len = max_nucleus_; len > 0; len--) {
139  if (GetPhoneNucleusPattern(pron, pos, len, stress_, &pat)) {
140  it = nuclei_.find(pat);

164 int32 TxpSylmax::GetPhoneVector(const char* pron,
165  PhoneVector *phonevectorptr) {
166  PhoneVector &phonevector = *phonevectorptr;
167  std::vector<std::string> array;
168  const char *p, *s;
169  int32 i, len;
170  p = pron;
171  i = 0;
172  if (!p) return 0;
173  while (*p) {
174  s = strstr(p, " ");
175  if (s) {
176  len = static_cast<int32>(s - p);
177  InsertPhone(phonevectorptr, p, len, false);
178  p = s;
179  p++;
180  } else {
181  len = strlen(p);
182  InsertPhone(phonevectorptr, p, len, true);
183  p += len;
184  }
185  i++;
186  }
187  return phonevector.size();

199  item.stress = std::string(p + len - 1, 1);

209 bool TxpSylmax::GetPhoneNucleusPattern(const PhoneVector &pron,
210  int32 pos, int32 len,
211  bool with_stress, std::string *pat) {
212  PhoneVector::const_iterator it;
213  int32 i;
214  pat->clear();
215  for (i = 0, it = pron.begin() + pos; i < len; ++it, ++i) {
216  if (it == pron.end()) {
217  pat->clear();
218  return false;
219  }
220  if (!pat->empty()) *pat += " ";
221  // Remove stress marks if required (forward only for nucleus)
222  if (!with_stress)
223  *pat += it->name;
224  else
225  *pat += it->name + it->stress;
226  }
227  return true;
228 }

230 bool TxpSylmax::GetPhoneOnsetPattern(const PhoneVector &pron,
231  int32 pos, int32 len,
232  std::string *pat) {
233  PhoneVector::const_reverse_iterator it;
234  int32 i;
235  pat->clear();
236  for (i = 0, it = pron.rbegin() + (pron.size() - pos); i < len; ++it, ++i) {
237  if (it == pron.rend()) {
238  pat->clear();
239  return false;
240  }
241  // word boundary blocks onset pattern
242  if (it->wrdb && !it->cross_word) {
243  pat->clear();
244  return false;
245  }
246  if (!pat->empty()) *pat = " " + *pat;
247  *pat = it->name + *pat;
248  }
249  return true;
250 }

258 void TxpSylmax::StartElement(const char *name, const char ** atts) {
259  std::string item;
260  PhoneVector pv;
261  int32 size;
262  if (!strcmp("nuclei", name)) {
263  SetAttribute("stress", atts, &item);
264  stress_ = false;
265  if (item == "true" || item == "True" || item == "TRUE")
266  stress_ = true;
267  } else if (!strcmp("s", name)) {
268  SetAttribute("name", atts, &item);
269  if (item.empty()) {
270  KALDI_ERR << fname_ << ": error missing 'name' attribute on line " << GetCurrentLineNumber();
271  return;
272  }
273  syllabic_.insert(item);
274  } else if (!strcmp("o", name)) {
275  SetAttribute("pat", atts, &item);
276  if (item.empty()) {
277  KALDI_ERR << fname_ << ": error missing 'pat' attribute on line " << GetCurrentLineNumber();
278  return;
279  }
280  onsets_.insert(item);
281  size = GetPhoneVector(item.c_str(), &pv);
282  if (size > max_onset_) max_onset_ = size;
283  } else if (!strcmp("n", name)) {
284  SetAttribute("pat", atts, &item);
285  if (item.empty()) {
286  KALDI_ERR << fname_ << ": error missing 'pat' attribute on line " << GetCurrentLineNumber();
287  return;
288  }
289  nuclei_.insert(item);
290  size = GetPhoneVector(item.c_str(), &pv);
291  if (size > max_nucleus_) max_nucleus_ = size;
292  }
293 }
