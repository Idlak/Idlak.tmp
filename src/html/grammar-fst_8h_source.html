41 // GrammarFstArc is an FST Arc type which differs from the normal StdArc type by
42 // having the state-id be 64 bits, enough to store two indexes: the higher 32
43 // bits for the FST-instance index, and the lower 32 bits for the state within
44 // that FST-instance.
45 // Obviously this leads to very high-numbered state indexes, which might be
46 // a problem in some circumstances, but the decoder code doesn't store arrays
47 // indexed by state, it uses hashes, so this isn't a problem.

162  Weight Final(StateId s) const {
163  // If the fst-id (top 32 bits of s) is nonzero, this state is not final,
164  // because we need to return to the top-level FST before we can be final.
165  if (s != static_cast<StateId>(static_cast<int32>(s))) {
166  return Weight::Zero();

169  Weight ans = top_fst_->Final(base_state);
170  if (ans.Value() == KALDI_GRAMMAR_FST_SPECIAL_WEIGHT) {
171  return Weight::Zero();

173  return ans;
174  }

339  inline ExpandedState *GetExpandedState(int32 instance_id,
340  BaseStateId state_id) {
341  std::unordered_map<BaseStateId, ExpandedState*> &expanded_states =
342  instances_[instance_id].expanded_states;
343 
344  std::unordered_map<BaseStateId, ExpandedState*>::iterator iter =
345  expanded_states.find(state_id);
346  if (iter != expanded_states.end()) {
347  return iter->second;
348  } else {
349  ExpandedState *ans = ExpandState(instance_id, state_id);
350  // Don't use the reference 'expanded_states'; it could have been
351  // invalidated.
352  instances_[instance_id].expanded_states[state_id] = ans;
353  return ans;
354  }
355  }

363  struct ExpandedState {
364  // The final-prob for expanded states is always zero; to avoid
365  // corner cases, we ensure this via adding epsilon arcs where
366  // needed.
367 
368  // fst-instance index of destination state (we will have ensured previously
369  // that this is the same for all outgoing arcs).
370  int32 dest_fst_instance;
371 
372  // List of arcs out of this state, where the 'nextstate' element will be the
373  // lower-order 32 bits of the destination state and the higher order bits
374  // will be given by 'dest_fst_instance'. We do it this way, instead of
375  // constructing a vector<Arc>, in order to simplify the ArcIterator code and
376  // avoid unnecessary branches in loops over arcs.
377  // We guarantee that this 'arcs' array will always be nonempty; this
378  // is to avoid certain hassles on Windows with automated bounds-checking.
379  std::vector<StdArc> arcs;
380  };

383  // An FstInstance is a copy of an FST. The instance numbered zero is for
384  // top_fst_, and (to state it approximately) whenever any FST instance invokes
385  // another FST a new instance will be generated on demand.
386  struct FstInstance {
387  // ifst_index is the index into the ifsts_ vector that corresponds to this
388  // FST instance, or -1 if this is the top-level instance.
389  int32 ifst_index;
390 
391  // Pointer to the FST corresponding to this instance: it will equal top_fst_
392  // if ifst_index == -1, or ifsts_[ifst_index].second otherwise.
393  const ConstFst<StdArc> *fst;
394 
395  // 'expanded_states', which will be populated on demand as states in this
396  // FST instance are accessed, will only contain entries for states in this
397  // FST that the final-prob's value equal to
398  // KALDI_GRAMMAR_FST_SPECIAL_WEIGHT. (That final-prob value is used as a
399  // kind of signal to this code that the state needs expansion).
400  std::unordered_map<BaseStateId, ExpandedState*> expanded_states;
401 
402  // 'child_instances', which is populated on demand as states in this FST
403  // instance are accessed, is logically a map from pair (nonterminal_index,
404  // return_state) to instance_id. When we encounter an arc in our FST with a
405  // user-defined nonterminal indexed 'nonterminal_index' on its ilabel, and
406  // with 'return_state' as its nextstate, we look up that pair
407  // (nonterminal_index, return_state) in this map to see whether there already
408  // exists an FST instance for that. If it exists then the transition goes to
409  // that FST instance; if not, then we create a new one. The 'return_state'
410  // that's part of the key in this map would be the same as the 'parent_state'
411  // in that child FST instance, and of course the 'parent_instance' in
412  // that child FST instance would be the instance_id of this instance.
413  //
414  // In most cases each return_state would only have a single
415  // nonterminal_index, making the 'nonterminal_index' in the key *usually*
416  // redundant, but in principle it could happen that two user-defined
417  // nonterminals might share the same return-state.
418  std::unordered_map<int64, int32> child_instances;
419 
420  // The instance-id of the FST we return to when we are done with this one
421  // (or -1 if this is the top-level FstInstance so there is nowhere to
422  // return).
423  int32 parent_instance;
424 
425  // The state in the FST of 'parent_instance' at which we expanded this FST
426  // instance, and to which we return (actually we return to the next-states
427  // of arcs out of 'parent_state').
428  int32 parent_state;
429 
430  // 'parent_reentry_arcs' is a map from left-context-phone (i.e. either a
431  // phone index or #nonterm_bos), to an arc-index, which we could use to
432  // Seek() in an arc-iterator for state parent_state in the FST-instance
433  // 'parent_instance'. It's set up when we create this FST instance. (The
434  // arcs used to enter this instance are not located here, they can be
435  // located in entry_arcs_[instance_id]). We make use of reentry_arcs when
436  // we expand states in this FST that have #nonterm_end on their arcs,
437  // leading to final-states, which signal a return to the parent
438  // FST-instance.
439  std::unordered_map<int32, int32> parent_reentry_arcs;
440  };

442  // The integer id of the symbol #nonterm_bos in phones.txt.

488 class ArcIterator<GrammarFst> {
489  public:
490  using Arc = typename GrammarFst::Arc;
491  using BaseArc = StdArc;
492  using StateId = typename Arc::StateId; // int64
493  using BaseStateId = typename StdArc::StateId; // int
494  using ExpandedState = GrammarFst::ExpandedState;
495 
496  // Caution: uses const_cast to evade const rules on GrammarFst. This is for
497  // compatibility with how things work in OpenFst.
498  inline ArcIterator(const GrammarFst &fst_in, StateId s) {
499  GrammarFst &fst = const_cast<GrammarFst&>(fst_in);
500  // 'instance_id' is the high order bits of the state.
501  int32 instance_id = s >> 32;
502  // 'base_state' is low order bits of the state. It's important to
503  // explicitly say int32 below, not BaseStateId == int, which might on some
504  // compilers be a 64-bit type.
505  BaseStateId base_state = static_cast<int32>(s);
506  const GrammarFst::FstInstance &instance = fst.instances_[instance_id];
507  const ConstFst<StdArc> *base_fst = instance.fst;
508  if (base_fst->Final(base_state).Value() != KALDI_GRAMMAR_FST_SPECIAL_WEIGHT) {
509  // A normal state
510  dest_instance_ = instance_id;
511  base_fst->InitArcIterator(s, &data_);
512  i_ = 0;
513  } else {
514  // A special state
515  ExpandedState *expanded_state = fst.GetExpandedState(instance_id,
516  base_state);
517  dest_instance_ = expanded_state->dest_fst_instance;
518  // it's ok to leave the other members of data_ uninitialized, as they will
519  // never be interrogated.
520  data_.arcs = &(expanded_state->arcs[0]);
521  data_.narcs = expanded_state->arcs.size();
522  i_ = 0;
523  }
524  // Ideally we want to call CopyArcToTemp() now, but we rely on the fact that
525  // the calling code needs to call Done() before accessing Value(); we call
526  // CopyArcToTemp() from Done(). Of course this is slightly against the
527  // semantics of Done(), but it's more efficient to have Done() call
528  // CopyArcToTemp() than this function or Next(), as Done() already has to
529  // test that the arc-iterator has not reached the end.
530  }
531 
532  inline bool Done() {
533  if (i_ < data_.narcs) {
534  CopyArcToTemp();
535  return false;
536  } else {
537  return true;
538  }
539  }
540 
541  inline void Next() {
542  i_++;
543  // Note: logically, at this point we should do:
544  // if (i_ < data_.size)
545  // CopyArcToTemp();
546  // Instead we move this CopyArcToTemp() invocation into Done(), which we
547  // know will always be called after Next() and before Value(), because the
548  // user has no other way of knowing whether the iterator is still valid.
549  // This is for efficiency.
550  }
551 
552  inline const Arc &Value() const { return arc_; }
553 
554  private:
555 
556  inline void CopyArcToTemp() {
557  const StdArc &src = data_.arcs[i_];
558  arc_.ilabel = src.ilabel;
559  arc_.olabel = src.olabel;
560  arc_.weight = src.weight;
561  arc_.nextstate = (static_cast<int64>(dest_instance_) << 32) |
562  src.nextstate;
563  }
564 
565  // The members of 'data_' that we use are:
566  // const Arc *arcs;
567  // size_t narcs;
568  ArcIteratorData<StdArc> data_;
569 
570 
571  int32 dest_instance_; // The index of the FstInstance that we transition to from
572  // this state.
573  size_t i_; // i_ is the index into the 'arcs' pointer.
574 
575  Arc arc_; // 'Arc' is the current arc in the GrammarFst, that this iterator
576  // is pointing to. It will be a copy of data_.arcs[i], except with
577  // the 'nextstate' modified to encode dest_instance_ in the higher
578  // order bits. Making a copy is of course unnecessary for the most
579  // part, but Value() needs to return a reference; we rely on the
580  // compiler to optimize out any unnecessary moves of data.
581 };
